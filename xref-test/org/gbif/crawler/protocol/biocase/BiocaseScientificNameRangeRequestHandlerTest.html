29      BiocaseScientificNameRangeRequestHandler handler = new BiocaseScientificNameRangeRequestHandler(job);

31      ScientificNameRangeCrawlContext context = new ScientificNameRangeCrawlContext();
32      assertThat(handler.buildRequestUrl(context)).isEqualTo(FIRST_URL);
