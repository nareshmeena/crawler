24    @Test
25    public void testUrlBuilding() {
26      URI targetUrl = URI.create("http://peabody.research.yale.edu/digir/DiGIR.php");
27      DigirCrawlConfiguration job = new DigirCrawlConfiguration(UUID.randomUUID(), 1, targetUrl, "ent", true);
28      DigirScientificNameRangeRequestHandler handler = new DigirScientificNameRangeRequestHandler(job);
29  
30      ScientificNameRangeCrawlContext context = new ScientificNameRangeCrawlContext();
31      String request = handler.buildRequestUrl(context);
32      request = request.replaceAll("sendTime.*sendTime", "");
33      assertThat(request).isEqualTo(FIRST_URL);
34  
35      context.setLowerBound("Aaa");
36      context.setUpperBound("Aba");
37      request = handler.buildRequestUrl(context);
38      request = request.replaceAll("sendTime.*sendTime", "");
39      assertThat(request).isEqualTo(SECOND_URL);
40  
41      context.setLowerBound("Zza");
42      context.setUpperBoundAbsent();
43      context.setOffset(1000);
44      request = handler.buildRequestUrl(context);
45      System.out.println(request);
46      request = request.replaceAll("sendTime.*sendTime", "");
47      assertThat(request).isEqualTo(LAST_URL);
48  
49      // Null jobs are not allowed
50      try {
51        handler.buildRequestUrl(null);
52        fail();
53      } catch (Exception e) {
54  
55      }
