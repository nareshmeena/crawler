118   /**
119    * Sets up a test crawl by generating an entry in the queue list. With a 30% chance an added crawl will also be added
120    * to the running list.
121    *
122    * @param uuid to add
123    */
124   private static void setupTestCrawl(UUID uuid) throws Exception {
125     Random random = new Random();
126 
127     // Simulates a curator queue identifier by generating a random number and then padding with '0' to 19 chars
128     String queueIdentifier = "queue-" + Strings.padStart(String.valueOf(Math.abs(random.nextLong())), 19, '0');
129     curator.create()
130       .creatingParentsIfNeeded()
131       .forPath(buildPath(XML_CRAWL, CrawlerNodePaths.QUEUED_CRAWLS, queueIdentifier), serialize(uuid, UUID_SERIALIZER));
132 
133     // 30% chance of this crawl being a running one
134     boolean running = false;
135     if (random.nextInt(100) < 30) {
136       running = true;
137       curator.create()
138         .creatingParentsIfNeeded()
139         .forPath(buildPath(XML_CRAWL, CrawlerNodePaths.RUNNING_CRAWLS, queueIdentifier),
140                  serialize(uuid, UUID_SERIALIZER));
141     }
142 
143     // get a DatasetCrawlMetrics mock object to populate the first crawl job
144     DatasetProcessStatus status = mockMetrics(uuid, running);
145     // populate the first crawl job with complete data
146     String runningJob = CrawlerNodePaths.getCrawlInfoPath(uuid);
147 
148     curator.create()
149       .creatingParentsIfNeeded()
150       .forPath(buildPath(runningJob, STARTED_CRAWLING), dateToBytes(status.getStartedCrawling()));
151     curator.create()
152       .creatingParentsIfNeeded()
153       .forPath(buildPath(runningJob, CRAWL_CONTEXT), stringToBytes(status.getCrawlContext()));
154 
155     curator.setData().forPath(runningJob, new ObjectMapper().writeValueAsBytes(status.getCrawlJob()));
156 
157     if (status.getDeclaredCount() != null) {
158       curator.create()
159         .forPath(buildPath(runningJob, DECLARED_COUNT), status.getDeclaredCount().toString().getBytes(Charsets.UTF_8));
160     }
161 
162     setCounter(buildPath(runningJob, PAGES_CRAWLED), status.getPagesCrawled());
163     setCounter(buildPath(runningJob, PAGES_FRAGMENTED_SUCCESSFUL), status.getPagesFragmentedSuccessful());
164     setCounter(buildPath(runningJob, PAGES_FRAGMENTED_ERROR), status.getPagesFragmentedError());
165     setCounter(buildPath(runningJob, FRAGMENTS_EMITTED), status.getFragmentsEmitted());
166     setCounter(buildPath(runningJob, FRAGMENTS_RECEIVED), status.getFragmentsReceived());
167     setCounter(buildPath(runningJob, RAW_OCCURRENCES_PERSISTED_NEW), status.getRawOccurrencesPersistedNew());
168     setCounter(buildPath(runningJob, RAW_OCCURRENCES_PERSISTED_UPDATED), status.getRawOccurrencesPersistedUpdated());
169     setCounter(buildPath(runningJob, RAW_OCCURRENCES_PERSISTED_UNCHANGED),
170                status.getRawOccurrencesPersistedUnchanged());
171     setCounter(buildPath(runningJob, RAW_OCCURRENCES_PERSISTED_ERROR), status.getRawOccurrencesPersistedError());
172     setCounter(buildPath(runningJob, FRAGMENTS_PROCESSED), status.getFragmentsProcessed());
173     setCounter(buildPath(runningJob, VERBATIM_OCCURRENCES_PERSISTED_SUCCESSFUL),
174                status.getVerbatimOccurrencesPersistedSuccessful());
175     setCounter(buildPath(runningJob, VERBATIM_OCCURRENCES_PERSISTED_ERROR),
176                status.getVerbatimOccurrencesPersistedError());
177     setCounter(buildPath(runningJob, INTERPRETED_OCCURRENCES_PERSISTED_SUCCESSFUL),
178                status.getInterpretedOccurrencesPersistedSuccessful());
179     setCounter(buildPath(runningJob, INTERPRETED_OCCURRENCES_PERSISTED_ERROR),
180                status.getInterpretedOccurrencesPersistedError());
181 
182     ALL_CRAWLS.add(status);
183     if (running) {
184       RUNNING_CRAWLS.add(status);
185     } else {
186       QUEUED_CRAWLS.add(status);
187       QUEUE_MAP.put(status, queueIdentifier);
188     }
189 
190     Ordering<DatasetProcessStatus> ordering =
191       Ordering.natural().onResultOf(new Function<DatasetProcessStatus, String>() {
192         @Override
193         public String apply(@Nullable DatasetProcessStatus input) {
194           return QUEUE_MAP.get(input);
195         }
196       });
197     QUEUED_CRAWLS = ordering.sortedCopy(QUEUED_CRAWLS);
198 
199   }

201   private static void setCounter(String path, long value) throws Exception {
202     curator.create().creatingParentsIfNeeded().forPath(path);
203     DistributedAtomicLong dal = new DistributedAtomicLong(curator, path, new RetryOneTime(100));
204     dal.trySet(value);
205   }
