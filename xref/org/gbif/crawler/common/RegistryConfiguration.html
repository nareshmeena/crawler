40        .add("registryUser", user)

43    }

45    /**
46     * Convenience method to setup a guice injector with a writable registry client module using the configuration
47     * of this instance.
48     *
49     * @return guice injector with RegistryWsClientModule bound
50     */
51    public Injector newRegistryInjector() {
52      // setup writable registry client
53      Properties properties = ConfigUtils.toProperties(this);
54      RegistryWsClientModule regModule = new RegistryWsClientModule(properties);
55      SingleUserAuthModule authModule = new SingleUserAuthModule(user, password);

57      return Guice.createInjector(regModule, authModule);
