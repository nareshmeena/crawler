15  /**
16   * This class handles a BioCASe response.
17   * <p/>
18   * Every BioCASe response has a few pieces of optional diagnostic information:
19   * <ul>
20   * <li>{@code recordCount} is the number of records that are being returned in this response</li>
21   * <li>{@code recordDropped} is the number of dropped records for this response, this number is <em>not</em> included
22   * in the {@code recordCount}s.</li>
23   * <li>{@code recordStart} is the {@code offset} that was used for this request</li>
24   * <li>{@code totalSearchHits} is the total number of results for this query. It usually doesn't return an exact count
25   * as that is expensive to calculate. If {@code totalSearchHits} is greater than {@code recordCount} + {@code
26   * recordDropped} + {@code recordStart} there should be more results</li>
27   * </ul>
28   * <p/>
29   * As an example. Sending a request with {@code start}=0 and {@code limit}=100 could return a response like this:
30   * <pre>
31   *   {@code <biocase:content recordCount='98' recordDropped='2' recordStart='0' totalSearchHits='101'>}
32   * </pre>
33   * <p/>
34   * Which would mean that the response includes 98 records and 2 had to be dropped on the way out. Because
35   * totalSearchHits is greater than recordCount + recordDropped + recordStart there are more pages and the next request
36   * should be with {@code start}=100 and {@code limit}=100 which could return this:
37   * <pre>
38   *   {@code <biocase:content recordCount='80' recordDropped='5' recordStart='100' totalSearchHits='185'>}
39   * </pre>
40   * <p/>
41   * This class is not thread-safe.
42   */
43  @NotThreadSafe
44  public class BiocaseResponseHandler extends AbstractResponseHandler {
45  
46    private static final Logger LOG = LoggerFactory.getLogger(BiocaseResponseHandler.class);
47  
48    private static final String CONTENT_ELEMENT = "content";
49  
50    private boolean insideContent;
51    private boolean seenContent;
52  
53    @Override
54    protected boolean shouldHash(XMLStreamReader2 reader) {
55      if (isElement(reader, CONTENT_ELEMENT, true)) {
56        insideContent = true;
57        return false;
58      }
59  
60      if (insideContent && reader.getEventType() == XMLStreamConstants.START_ELEMENT) {
61        seenContent = true;
62      }
63  
64      // We found the end element which shouldn't be hashed either
65      if (isElement(reader, CONTENT_ELEMENT, false)) {
66        insideContent = false;
67      }
68  
69      return insideContent && seenContent;
70    }
71  
72    @Override
73    protected void initialize() {
74      insideContent = false;
75      seenContent = false;
76    }
77  
78    @Override
79    protected void process(XMLStreamReader2 reader) throws XMLStreamException, ProtocolException {
80      if (reader.getEventType() == XMLStreamConstants.START_ELEMENT && reader.getLocalName().equals(CONTENT_ELEMENT)) {
81        Optional<Integer> recordStart = getAttribute(reader, "recordStart");
82        Optional<Integer> recordCount = getAttribute(reader, "recordCount");
83        Optional<Integer> recordDropped = getAttribute(reader, "recordDropped");
84        Optional<Integer> totalHits = getAttribute(reader, "totalSearchHits");
85  
86        setRecordCount(recordCount);
87        if (recordCount.isPresent() && recordStart.isPresent() && recordDropped.isPresent() && totalHits.isPresent()) {
88          setEndOfRecords(Optional.of(recordCount.get() + recordStart.get() + recordDropped.get() >= totalHits.get()));
89        }
90  
91        LOG.debug("Found "
92                  + CONTENT_ELEMENT
93                  + " element. [{}] records, [{}] records dropped, [{}] record start, [{}] total search hits",
94                  new Object[] {recordCount, recordDropped, recordStart, totalHits});
95      }
96    }
97  
98    /**
99     * This tries to read an attribute of an element as an integer and returns it if found, {@link Optional#absent()}
100    * otherwise.
101    *
102    * @param reader    to read the attribute from, needs to be set to an element
103    * @param attribute to read from the element
104    *
105    * @return an Optional with the integer value of the attribute {@code absent} otherwise
106    */
107   private Optional<Integer> getAttribute(XMLStreamReader2 reader, String attribute) throws XMLStreamException {
108     int index = reader.getAttributeIndex(null, attribute);
109     if (index == -1) {
110       return Optional.absent();
111     }
112 
113     return Optional.of(reader.getAttributeAsInt(index));
114   }
