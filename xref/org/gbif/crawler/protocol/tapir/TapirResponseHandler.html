13  /**
14   * This class handles a TAPIR response.
15   * <p/>
16   * Every TAPIR response has a few pieces of diagnostic information as defined in the <a
17   * href="http://www.tdwg.org/dav/subgroups/tapir/1.0/docs/tdwg_tapir_specification_2010-05-05.htm">specification</a>:
18   * <ul>
19   * <li>{@code start} index of the first element in this response</li>
20   * <li>{@code totalReturned} number of elements in this response</li>
21   * <li>{@code next} index of the next element that should be retrieved. Used for paging. This attribute is absent if
22   * there are no more records.</li>
23   * </ul>
24   * <p/>
25   * As an example. Sending a request with {@code start}=0 and {@code limit}=2 could return a response like this:
26   * <pre>
27   *   {@code <summary start='0' next='2' totalReturned='2'/>}
28   * </pre>
29   * <p/>
30   * This class is not thread-safe.
31   */
32  @NotThreadSafe
33  public class TapirResponseHandler extends AbstractResponseHandler {
34  
35    private static final String CONTENT_ELEMENT = "search";
36    private static final String INFO_ELEMENT = "summary";
37  
38    private boolean insideContent;
39    private boolean seenContent;
40  
41    @Override
42    protected boolean shouldHash(XMLStreamReader2 reader) {
43      if (isElement(reader, CONTENT_ELEMENT, true)) {
44        insideContent = true;
45        return false;
46      }
47  
48      boolean infoElement = isElement(reader, INFO_ELEMENT, true) || isElement(reader, INFO_ELEMENT, false);
49  
50      if (insideContent && !infoElement && reader.getEventType() == XMLStreamConstants.START_ELEMENT) {
51        seenContent = true;
52      }
53  
54      if (isElement(reader, CONTENT_ELEMENT, false)) {
55        insideContent = false;
56      }
57  
58      return insideContent && seenContent && !infoElement;
59    }
60  
61    @Override
62    protected void initialize() {
63      insideContent = false;
64      seenContent = false;
65    }
66  
67    @Override
68    protected void process(XMLStreamReader2 reader) throws XMLStreamException, ProtocolException {
69      if (isElement(reader, INFO_ELEMENT, true)) {
70        int index = reader.getAttributeIndex(null, "totalReturned");
71        if (index == -1) {
72          setRecordCount(Optional.<Integer>absent());
73        } else {
74          setRecordCount(Optional.of(reader.getAttributeAsInt(index)));
75        }
76  
77        if (reader.getAttributeIndex(null, "next") == -1) {
78          setEndOfRecords(Optional.<Boolean>absent());
79        } else {
80          setEndOfRecords(Optional.of(false));
81        }
82  
83      }
84    }
85  
86  }
