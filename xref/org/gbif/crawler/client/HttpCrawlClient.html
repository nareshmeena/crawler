37  /**
38   * Executes HTTP GET requests in String form using Apache HttpClient.
39   * <p/>
40   * This client does not support HTTPS.
41   * <p/>
42   * {@link ResponseHandler}s must understand how to process a {@link HttpResponse} but they do not need to handle stream
43   * closing as this is done by this class. When not needed anymore {@link #shutdown()} should be called so underlying
44   * resources (e.g. the connection manager) can be closed.
45   * <p/>
46   * This class is thread-safe.
47   */
48  // TODO: Log redirects - implement a RedirectStrategy

51  @ThreadSafe
52  public class HttpCrawlClient implements CrawlClient<String, HttpResponse> {

59    /**
60     * Factory method to create a new instance of this client with a few default settings. This creates an instance that
61     * works with HTTP only (no HTTPS), uses multi-threading for its connections, supports compressed content and sets
62     * some user provided timeouts.
63     *
64     * @param connectionTimeout   this timeout in milliseconds is used for establishing a connection and for waiting for
65     *                            data
66     * @param maxTotalConnections maximum total number of connections
67     * @param maxTotalPerRoute    maximum number of connections per host
68     *
69     * @return new HttpCrawlClient with default settings
70     */
71    public static HttpCrawlClient newInstance(int connectionTimeout, int maxTotalConnections, int maxTotalPerRoute) {
72      checkArgument(connectionTimeout > 0, "connectionTimeout has to be greater than zero");
73      checkArgument(maxTotalConnections > 0, "maxTotalConnections has to be greater than zero");
74      checkArgument(maxTotalPerRoute > 0, "maxTotalPerRoute has to be greater than zero");
75  
76      SchemeRegistry schemeRegistry = new SchemeRegistry();
77      schemeRegistry.register(new Scheme("http", 80, PlainSocketFactory.getSocketFactory()));
78  
79      PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry);
80      connectionManager.setMaxTotal(maxTotalConnections);
81      connectionManager.setDefaultMaxPerRoute(maxTotalPerRoute);
82  
83      HttpParams params = new BasicHttpParams();
84      HttpConnectionParams.setConnectionTimeout(params, connectionTimeout);
85      HttpConnectionParams.setSoTimeout(params, connectionTimeout);
86      params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, connectionTimeout);
87      HttpClient httpClient = new DecompressingHttpClient(new DefaultHttpClient(connectionManager, params));
88  
89      return new HttpCrawlClient(connectionManager, httpClient);
90    }

100   /**
101    * Initiates the request using HTTP and processes the response with the given Response handler.
102    */
103   @Override
104   public <RESULT> RESULT execute(String request, final ResponseHandler<HttpResponse, RESULT> handler)
105     throws FatalCrawlException, TransportException, ProtocolException {
106     HttpUriRequest httpget = new HttpGet(request);
107     LOG.debug("Executing request [{}]", request);
108     try {
109       /*
110         This actually executes the request which can throw either a IOException (or its subclass
111         ClientProtocolException) or the special runtime exception (InnerResponseHandlerException) from the
112         ForwardingResponseHandler. We assume that all exceptions coming from the HttpClient itself are
113         protocol level exceptions.
114        */
115       return httpClient.execute(httpget, new ForwardingResponseHandler<RESULT>(handler));
116     } catch (RuntimeException e) {
117       LOG.debug("Caught exception from underlying ResponseHandler", e.getCause());
118 
119       Throwables.propagateIfPossible(e.getCause(), ProtocolException.class);
120       Throwables.propagateIfPossible(e.getCause(), TransportException.class);
121       Throwables.propagateIfPossible(e.getCause(), FatalCrawlException.class);
122       throw e;
123     } catch (IOException e) {
124       LOG.debug("Caught exception during the HTTP request [{}]", httpget, e);
125       // Both ClientProtocolException as well as IOException are TransportExceptions
126       throw new TransportException(e);
127     }
128   }

138   /**
139    * An implementation of HTTPClient's ResponseHandler interface which forwards to our own ResponseHandler interface.
140    * <p/>
141    * This automatically checks for failed requests (i.e. HTTP response code != 200).
142    *
143    * @param <R> the return type of our ResponseHandler
144    */
145   @VisibleForTesting
146   static class ForwardingResponseHandler<R> implements org.apache.http.client.ResponseHandler<R> {
147 
148     private final ResponseHandler<HttpResponse, R> handler;
149 
150     ForwardingResponseHandler(ResponseHandler<HttpResponse, R> handler) {
151       this.handler = handler;
152     }
153 
154     /**
155      * Handles a response by forwarding it to an inner response handler.
156      */
157     @Override
158     public R handleResponse(HttpResponse response) throws IOException {
159       if (response.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
160         throw new ClientProtocolException(
161           "HTTP status code[" + response.getStatusLine().getStatusCode() + "] indicates failure: [" + response
162             .getStatusLine() + "]");
163       }
164 
165       try {
166         return handler.handleResponse(response);
167       } catch (ProtocolException e) {
168         throw Throwables.propagate(e);
169       } catch (TransportException e) {
170         throw Throwables.propagate(e);
171       } catch (FatalCrawlException e) {
172         throw Throwables.propagate(e);
173       }
174     }
