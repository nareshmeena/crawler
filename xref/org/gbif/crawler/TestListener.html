25    public static void main(String[] args) throws IOException {
26      ConnectionParameters connectionParameters = new ConnectionParameters("localhost", 5672, "guest", "guest", "/");
27      ObjectMapper objectMapper = new ObjectMapper().configure(Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
28  
29      MessageListener listener =
30        new MessageListener(connectionParameters, new DefaultMessageRegistry(),
31          objectMapper);
32  
33      listener.listen("testlistener", "registry.change.#", 100, new AbstractMessageCallback<RegistryChangeMessage>() {
34  
35        @Override
36        public void handleMessage(RegistryChangeMessage message) {
37          switch (message.getChangeType()) {
38            case CREATED:
39              LOG.info("Message type: [{}], class [{}]", message.getChangeType(), message.getObjectClass());
40              if (message.getObjectClass().equals(Dataset.class)) {
41                Dataset dataset = (Dataset) message.getNewObject();
42                LOG.info("Created new Dataset with Key [{}] and Title [{}]", dataset.getKey(), dataset.getTitle());
43              }
44              break;
45            case UPDATED:
46              if (message.getObjectClass().equals(Dataset.class)) {
47                Dataset newDataset = (Dataset) message.getNewObject();
48                Dataset oldDataset = (Dataset) message.getOldObject();
49                LOG.info("Updated dataset with key [{}], old Title [{}], new Title [{}]",
50                  newDataset.getKey(),
51                  oldDataset.getTitle(),
52                  newDataset.getTitle());
53              }
54              break;
55            case DELETED:
56              if (message.getObjectClass().equals(Dataset.class)) {
57                Dataset dataset = (Dataset) message.getOldObject();
58                LOG.info("Deleted Dataset with Key [{}] and Title [{}]", dataset.getKey(), dataset.getTitle());
59              }
60              break;
61          }
62        }
63      });
64    }
