38  /**
39   * This crawler can be used to crawl datasets. It plays an orchestrating role and ties together a lot of other
40   * components that have to be supplied using the constructor.
41   * <p/>
42   * This class is not thread-safe as one cannot rely on all the underlying components being thread safe.
43   *
44   * @param <CTX> the crawl context to be used, this holds the mutable state of a crawl and usually depends on the
45   *        crawl strategy being used
46   * @param <REQ> the type of request being created by the request handler and used by the crawl client
47   * @param <RESP> the type of the response being returned by the crawl client and consumed by the response handler
48   * @param <RES> the type of the result being generated by the response handler
49   */
50  @NotThreadSafe
51  public class Crawler<CTX extends CrawlContext, REQ, RESP, RES> {

172     ResponseHandler<HttpResponse, List<Byte>> responseHandler
173     ) {

175     CrawlStrategy<ScientificNameRangeCrawlContext> strategy = new ScientificNameRangeStrategy(context);

194   /**
195    * This starts a crawl using the provided strategy.
196    * <p/>
197    * It will iterate over the contexts the strategy provides. For each context it will also iterate over all the pages.
198    * It uses the RequestHandler to build a request for each context which will then be retrieved by the CrawlClient and
199    * the response will be handled by the ResponseHandler.
200    */
201   public void crawl() {
202 
203     // Initialization
204     notifyStart();
205     CTX currentContext = null;
206 
207     /*
208      * Main loop, will run as long as we didn't get a user abort, there are more things to crawl
209      * and retries have not been exhausted.
210      * It's using two loops, one to loop over each of the Contexts returned by the CrawlStrategy which will depend
211      * on the implementation and an inner loop that loops over pages for each context. That does not depend on the
212      * strategy because there could be multiple pages for any context, this is dependent on the response handler.
213      */
214     try {
215       while ((currentContext == null || !currentContext.isAborted())
216         && !retryPolicy.abortCrawl()
217         && strategy.hasNext()) {
218         currentContext = strategy.next();
219 
220         // The inner loop
221         boolean hasNextPage;
222         do {
223           notifyProgress(currentContext);
224 
225           REQ req = requestHandler.buildRequestUrl(currentContext);
226 
227           executeRequest(req, lock);
228 
229           hasNextPage = setupNextCrawlContext(currentContext);
230         } while (!currentContext.isAborted() && !retryPolicy.abortCrawl() && hasNextPage);
231 
232       }
233     } catch (FatalCrawlException e) {
234       notifyError(e);
235       notifyFinishedAbnormally();
236       return;
237     }
238 
239     if (currentContext != null && currentContext.isAborted()) {
240       LOG.info("Aborted crawl on request");
241       notifyFinishedOnUserRequest();
242     } else if (retryPolicy.abortCrawl()) {
243       LOG.info("Aborted crawl due to exhausted retries");
244       notifyFinishedAbnormally();
245     } else {
246       notifyFinishedNormally();
247     }
248   }

250   /**
251    * Executes a request while honoring retries. If successful it also reports the response.
252    *
253    * @param req to execute
254    * @return either the result if successful or {@code null}
255    */
256   private RES executeRequest(REQ req, Lock lock) throws FatalCrawlException {
257     boolean complete = false;
258     boolean tryAgain = true;
259     int tryCount = 0;
260 
261     RES res = null;
262     do {
263       try {
264         tryCount++;
265 
266         // First we need to get the lock, we're waiting 10 minutes between tries
267         // TODO: At the moment we never abort this so we may wait forever
268         while (!lock.tryLock(10, TimeUnit.MINUTES)) {
269           LOG.debug("Failed to acquire lock [{}]", lock);
270         }
271 
272         // Then we actually try to execute and handle the request
273         notifyRequest(req, tryCount);
274         stopwatch.reset();
275         stopwatch.start();
276 
277         res = client.execute(req, responseHandler);
278         stopwatch.stop();
279         notifyResponse(res,
280           tryCount,
281           stopwatch.elapsedMillis(),
282           responseHandler.getRecordCount(),
283           responseHandler.isEndOfRecords());
284         retryPolicy.successfulRequest();
285         complete = true;
286       } catch (TransportException e) {
287         notifyError(e);
288         if (retryPolicy.allowAfterTransportException()) {
289           LOG.info("Got transport exception, will retry", e);
290           tryAgain = true;
291         } else {
292           tryAgain = false;
293           LOG.warn("Got transport exception, will give up this request");
294           retryPolicy.giveUpRequest();
295         }
296       } catch (ProtocolException e) {
297         notifyError(e);
298         if (retryPolicy.allowAfterProtocolException()) {
299           LOG.info("Got protocol exception, will retry", e);
300           tryAgain = true;
301         } else {
302           tryAgain = false;
303           LOG.warn("Got protocol exception, will give up this request");
304           retryPolicy.giveUpRequest();
305         }
306       } finally {
307         // We don't generally stop it here because we don't want to measure even more of "non request" time than we
308         // already do. Splitting this code in multiple try/catch/finally blocks seems to heavyweight compared to adding
309         // this extra check.
310         if (stopwatch.isRunning()) {
311           stopwatch.stop();
312         }
313         // We unlock even if the request was not successful to give other crawlers a chance to get this lock
314         lock.unlock();
315       }
316     } while (!complete && tryAgain);
317 
318     return res;
319   }

321   private void nextPage(CTX context, boolean speculative) {
322     context.setSpeculative(speculative);
323 
324     // TODO: This isn't very intelligent and could be improved.
325     // We can either parse the response in the crawler which would be a major architectural change. We could also read
326     // the diagnostics messages the providers provide us with and act on those. Best would be to start a crawl with a
327     // very conservative number and then increase the number of requested records dynamically.
328     int offset = responseHandler.getRecordCount().or(requestHandler.getLimit());
329     context.setOffset(context.getOffset() + offset);
330   }

578    * <em>valid request</em> means that we did not get a {@link ProtocolException} or {@link TransportException}.
579    * <p/>
580    * <p/>
581    * <p/>
582    *
583    * @return whether there is another page for the current crawl context. If not we'll try the next range.
584    */
585   private boolean setupNextCrawlContext(CTX context) {
586 
587     // 16: We did not get a valid reply so we're going to skip to the next range, the main loop outside of this method
588     // will determine if we're going to abort because of too many errors.
589     if (!responseHandler.isValidState()) {
590       nextRange(context);
591       return false;
592     }
593 
594     Optional<Long> currentContentHash = responseHandler.getContentHash();
595     // 13: Check to see if we got the exact same content as for the last request, if so skip to the next range
596     // if not then proceed as normal
597     if (currentContentHash.isPresent()
598       && context.getLastContentHash().isPresent()
599       && currentContentHash.get().equals(context.getLastContentHash().get())) {
600       // Next range, log error
601       notifyError("Got same content twice in a row, hash is [" + currentContentHash.get() + "]");
602       nextRange(context);
603       return false;
604     }
605 
606     // We did check the old content hash so we can now update it with the new one
607     if (currentContentHash.isPresent()) {
608       context.setLastContentHash(currentContentHash);
609     }
610 
611     // 14: Next range (last request was speculative but we didn't get any content)
612     if (!currentContentHash.isPresent() && context.isSpeculative()) {
613       nextRange(context);
614       return false;
615     }
616 
617     if (currentContentHash.isPresent() && context.isSpeculative()) {
618       // 15: Only log previous and last request but otherwise continue with any normal request handling
619       // The last request was speculative and we did get content this time, that is in theory valid behavior for
620       // all protocols but is definitely not normal so we should log this and maybe take a look at the endpoints
621       // in question. The content hash comparison or the normal request handling will be used to eventually
622       // skip to the next range.
623 
624       // TODO: Log speculative request that lead to this
625     }
626 
627 
628     Optional<Integer> recordCount = responseHandler.getRecordCount();
629     if (responseHandler.isEndOfRecords().or(false)) {
630 
631       // 5, 6, 7 & 8
632       nextRange(context);
633 
634       if (recordCount.isPresent()) {
635         if (currentContentHash.isPresent()) {
636           if (recordCount.get() == 0) {
637             // 7: Next range, log error if record count == 0
638             notifyError("Got content but record count says we got nothing");
639           }
640         } else {
641           // 8: Next range, log error if record count != 0
