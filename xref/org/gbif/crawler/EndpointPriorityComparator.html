13  /**
14   * Compares two Endpoints.
15   * <p/>
16   * It does so by using a priority list of Endpoint Types.
17   * This Comparator will throw a {@link ClassCastException} exception if a non supported EndpointType is passed in. It
18   * also does not support {@code null} values.
19   * <p/>
20   * The priority list is as follows (most to least important):
21   * <ol>
22   * <li>DwC-A</li>
23   * <li>TAPIR</li>
24   * <li>BioCASe</li>
25   * <li>DiGIR</li>
26   * <li>DiGIR (Manis)</li>
27   * </ol>
28   */
29  public class EndpointPriorityComparator implements Comparator<Endpoint>, Serializable {
30  
31    /* Note: Due to legacy reasons this is a Comparator itself instead of just using the PRIORITY_COMPARATOR.
32     It should be easy to remove this class or change it into a factory */
33  
34    // Priorities from lowest to highest
35    public static final List<EndpointType> PRIORITIES = ImmutableList.of(EndpointType.DIGIR_MANIS,
36                                                                         EndpointType.DIGIR,
37                                                                         EndpointType.BIOCASE,
38                                                                         EndpointType.TAPIR,
39                                                                         EndpointType.DWC_ARCHIVE);
40    private static final long serialVersionUID = 8085216142750609841L;
41    private static final Ordering<EndpointType> PRIORITY_COMPARATOR = Ordering.explicit(PRIORITIES);
42  
43    @Override
44    public int compare(Endpoint endpoint1, Endpoint endpoint2) {
45      return PRIORITY_COMPARATOR.compare(endpoint1.getType(), endpoint2.getType());
46    }
