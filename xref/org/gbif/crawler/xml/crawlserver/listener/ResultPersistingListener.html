16  package org.gbif.crawler.xml.crawlserver.listener;

18  import org.gbif.crawler.AbstractCrawlListener;
19  import org.gbif.crawler.CrawlConfiguration;
20  import org.gbif.crawler.strategy.ScientificNameRangeCrawlContext;
21  
22  import java.io.File;
23  import java.io.IOException;
24  import java.util.List;
25  
26  import com.google.common.base.Optional;
27  import com.google.common.io.Files;
28  import com.google.common.primitives.Bytes;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;

32  /**
33   * This Crawl Listener will persist every single response we receive to disk.
34   *
35   * It will do so in sub folders of the given base path in a format like this:
36   * /<uuid>/<attempt>/<context>/<uuid>_<attempt>_<context>_<retry>.response
37   */
38  public class ResultPersistingListener
39    extends AbstractCrawlListener<ScientificNameRangeCrawlContext, String, List<Byte>> {
40  
41    private static final Logger LOG = LoggerFactory.getLogger(ResultPersistingListener.class);
42  
43    private final File basePath;
44    private final CrawlConfiguration configuration;
45  
46    public ResultPersistingListener(File basePath, CrawlConfiguration configuration) {
47      this.basePath = basePath;
48      this.configuration = configuration;
49    }
50  
51    @Override
52    public void response(
53      List<Byte> response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords
54    ) {
55      String contextString =
56        getCurrentContext().getLowerBound().or("null") + "-" + getCurrentContext().getUpperBound().or("null");
57      if (response == null || response.isEmpty()) {
58        LOG.info("Received empty response for [{}] [{}] in retry [{}]",
59                 configuration.getDatasetKey(),
60                 contextString,
61                 retry);
62      }
63      // This is guaranteed to be efficient as documented on the AbstractResponseHandler
64      byte[] bytes = Bytes.toArray(response);
65      try {
66        // Creates a format like this: /<uuid>/<attempt>/<context>/<uuid>_<attempt>_<context>_<retry>.response
67        StringBuilder path = new StringBuilder();
68        path.append(configuration.getDatasetKey())
69          .append(File.separator)
70          .append(configuration.getAttempt())
71          .append(File.separator)
72          .append(contextString)
73          .append(File.separator)
74          .append(configuration.getDatasetKey())
75          .append("_")
76          .append(configuration.getAttempt())
77          .append("_")
78          .append(contextString)
79          .append("_")
80          .append(getCurrentContext().getOffset())
81          .append("_")
82          .append(retry)
83          .append(".response");
84  
85        File file = new File(basePath, path.toString());
86        Files.createParentDirs(file);
87        Files.write(bytes, file);
88      } catch (IOException e) {
89        LOG.warn("Could not write response due to exception. [{}]", configuration.getDatasetKey(), e);
90      }
91    }
92  }
