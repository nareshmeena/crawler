34  @NotThreadSafe
35  public class CrawlerZooKeeperUpdatingListener<CTX extends CrawlContext>
36    extends AbstractCrawlListener<CTX, String, List<Byte>> {
37  
38    private static final Logger LOG = LoggerFactory.getLogger(CrawlerZooKeeperUpdatingListener.class);
39  
40    private static final ObjectMapper MAPPER = new ObjectMapper();
41  
42    static {
43      MAPPER.registerModule(new GuavaModule());
44    }
45  
46    private final SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ");
47  
48    private final CrawlConfiguration configuration;
49    private final CuratorFramework curator;
50  
51    public CrawlerZooKeeperUpdatingListener(CrawlConfiguration configuration, CuratorFramework curator) {
52      this.configuration = configuration;
53      this.curator = curator;
54  
55      dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
56    }
57  
58    @Override
59    public void startCrawl() {
60      String crawlPath = CrawlerNodePaths.getCrawlInfoPath(configuration.getDatasetKey(), STARTED_CRAWLING);
61      Date date = new Date();
62      byte[] data = dateFormat.format(date).getBytes(Charsets.UTF_8);
63      createOrUpdateZooKeeper(crawlPath, data);
64    }
65  
66    @Override
67    public void progress(CTX context) {
68      String crawlPath = CrawlerNodePaths.getCrawlInfoPath(configuration.getDatasetKey(), CRAWL_CONTEXT);
69      try {
70        byte[] data = MAPPER.writeValueAsBytes(context);
71        createOrUpdateZooKeeper(crawlPath, data);
72      } catch (IOException e) {
73        LOG.error("Error serializing context object", e);
74      }
75    }
76  
77    @Override
78    public void response(
79      List<Byte> response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords
80    ) {
81      String crawlPath = CrawlerNodePaths.getCrawlInfoPath(configuration.getDatasetKey(), PAGES_CRAWLED);
82      DistributedAtomicLong dal = new DistributedAtomicLong(curator, crawlPath, new RetryNTimes(1, 1000));
83      try {
84        AtomicValue<Long> value = dal.increment();
85        if (value.succeeded()) {
86          LOG.debug("Set counter of pages crawled for [{}] to [{}]", configuration.getDatasetKey(), value.postValue());
87        } else {
88          LOG.error("Failed to update counter of pages crawled for [{}]",
89                    configuration.getDatasetKey(),
90                    value.postValue());
91        }
92      } catch (Exception e) {
93        LOG.error("Failed to update counter of pages crawled for [{}]", configuration.getDatasetKey(), e);
94      }
95    }
96  
97    @Override
98    public void finishCrawlNormally() {
99      finishCrawl(FinishReason.NORMAL);
100   }

112   public void finishCrawl(FinishReason reason) {
113     String crawlPath = CrawlerNodePaths.getCrawlInfoPath(configuration.getDatasetKey(), FINISHED_CRAWLING);
114     Date date = new Date();
115     byte[] data = dateFormat.format(date).getBytes(Charsets.UTF_8);
116     createOrUpdateZooKeeper(crawlPath, data);
117 
118     crawlPath = CrawlerNodePaths.getCrawlInfoPath(configuration.getDatasetKey(), FINISHED_REASON);
119     createOrUpdateZooKeeper(crawlPath, reason.toString().getBytes(Charsets.UTF_8));
120   }
121 
122   private void createOrUpdateZooKeeper(String path, byte[] data) {
123     try {
124       Stat stat = curator.checkExists().forPath(path);
125       if (stat == null) {
126         curator.create().forPath(path, data);
127       } else {
128         curator.setData().forPath(path, data);
129       }
130     } catch (Exception e) {
131       LOG.error("Exception while updating ZooKeeper", e);
132     }
133   }
