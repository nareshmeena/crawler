19   * are at WARN. It also make two things available in the MDC: datasetKey and attempt.

32  
33    public LoggingCrawlListener(CrawlConfiguration configuration) {
34      MDC.put("datasetKey", configuration.getDatasetKey().toString());
35      MDC.put("attempt", String.valueOf(configuration.getAttempt()));
36    }

38    @Override

41      LOG.info("started crawl");
42    }
43  
44    @Override

47    }
48

52    }

54    @Override
55    public void response(
56      List<Byte> response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords
57    ) {

60               lastContext,
61               recordCount,
62               endOfRecords,
63               retry,
64               TimeUnit.MILLISECONDS.toSeconds(duration));
65    }

71

77    @Override

82    private void finishCrawl(FinishReason reason) {
83      Date finishDate = new Date();
84      long minutes = (finishDate.getTime() - startDate.getTime()) / (60 * 1000);
85          LOG.info(
86        "finished crawling with a total of [{}] records, reason [{}], started at [{}], finished at [{}], took [{}] minutes",
87        totalRecordCount,
88        reason,
89        startDate,
90        finishDate,
91        minutes);
92  
93      MDC.remove("datasetKey");

95    }
96

98    public void request(String req, int retry) {

100     lastRequest = req;
101   }
