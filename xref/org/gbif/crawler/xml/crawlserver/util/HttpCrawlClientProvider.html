19    private static final int DEFAULT_HTTP_PORT = 80;
20  
21    private static final int CONNECTION_TIMEOUT_MSEC = 600000; // 10 mins
22    private static final int MAX_TOTAL_CONNECTIONS = 500;
23    private static final int MAX_TOTAL_PER_ROUTE = 20;
24  
25    public static HttpCrawlClient newHttpCrawlClient() {
26      SchemeRegistry schemeRegistry = new SchemeRegistry();
27      schemeRegistry.register(new Scheme("http", DEFAULT_HTTP_PORT, PlainSocketFactory.getSocketFactory()));
28  
29      PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry);
30      connectionManager.setMaxTotal(MAX_TOTAL_CONNECTIONS);
31      connectionManager.setDefaultMaxPerRoute(MAX_TOTAL_PER_ROUTE);
32  
33      HttpParams params = new BasicHttpParams();
34      HttpConnectionParams.setConnectionTimeout(params, CONNECTION_TIMEOUT_MSEC);
35      HttpConnectionParams.setSoTimeout(params, CONNECTION_TIMEOUT_MSEC);
36      params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, CONNECTION_TIMEOUT_MSEC);
37      HttpClient httpClient = new DecompressingHttpClient(new DefaultHttpClient(connectionManager, params));
38  
39      return new HttpCrawlClient(connectionManager, httpClient);
40    }
