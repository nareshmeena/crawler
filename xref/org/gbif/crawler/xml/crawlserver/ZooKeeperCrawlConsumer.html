39    private static final Logger LOG = LoggerFactory.getLogger(ZooKeeperCrawlConsumer.class);

42  
43    static {
44      MAPPER.registerModule(new GuavaModule());
45    }

47    private static final int DEFAULT_CONCURRENT_CONNECTIONS = 2;
48    private static final String CONCURRENT_CONNECTIONS_PROPERTY = "concurrentConnections";

52    private final Optional<Long> minDelay;
53    private final Optional<Long> maxDelay;
54    private final File responseArchive;

56    /**
57     * @param minDelay is the "grace" period in milliseconds we allow between calls to a URL implemented by waiting to
58     *        release the lock on the URL for this amount of time
59     */
60    ZooKeeperCrawlConsumer(
61      CuratorFramework curator,
62      MessagePublisher publisher,
63      Optional<Long> minDelay,
64      Optional<Long> maxDelay,
65      File responseArchive) {
66      this.curator = curator;
67      this.publisher = publisher;
68      this.minDelay = minDelay;
69      this.maxDelay = maxDelay;
70      this.responseArchive = responseArchive;
71    }

73    @Override
74    public void stateChanged(CuratorFramework client, ConnectionState newState) {
75      LOG.info("State changed to [{}]", newState);
76    }

78    @Override
79    public void consumeMessage(UUID message) throws Exception {
80      LOG.debug("Got crawl job for UUID [{}]", message);
81      MDC.put("datasetKey", message.toString());
82  
83      try {
84        byte[] bytes = curator.getData().forPath(CrawlerNodePaths.getCrawlInfoPath(message));
85        CrawlJob crawlJob = MAPPER.readValue(bytes, CrawlJob.class);
86        LOG.debug("Crawl job detail [{}]", crawlJob);
87  
88        CrawlerBuilder builder =
89          CrawlerBuilder.buildFor(crawlJob).withDefaultRetryStrategy().withScientificNameRangeCrawlContext();
90  
91        int concurrentConnections = DEFAULT_CONCURRENT_CONNECTIONS;
92        if (crawlJob.getProperties().containsKey(CONCURRENT_CONNECTIONS_PROPERTY)) {
93          concurrentConnections = Integer.parseInt(crawlJob.getProperty(CONCURRENT_CONNECTIONS_PROPERTY));
94        }
95  
96        builder.withZooKeeperLock(curator, concurrentConnections);
97  
98        if (minDelay.isPresent()) {
99          builder.withDelayedLock(minDelay.get(), maxDelay.get());
100       }
101 
102       Crawler<ScientificNameRangeCrawlContext, String, HttpResponse, List<Byte>> crawler = builder.build();
103 
104       crawler.addListener(new CrawlerZooKeeperUpdatingListener<ScientificNameRangeCrawlContext>(builder
105         .getCrawlConfiguration(),
106         curator));
107       crawler.addListener(new LoggingCrawlListener<ScientificNameRangeCrawlContext>(builder.getCrawlConfiguration()));
108       crawler.addListener(new MessagingCrawlListener<ScientificNameRangeCrawlContext>(publisher,
109         builder.getCrawlConfiguration()));
110       if (responseArchive != null) {
111         crawler.addListener(new ResultPersistingListener(responseArchive, builder.getCrawlConfiguration()));
112       }
113 
114       // This blocks until the crawl is finished.
115       crawler.crawl();
116     } catch (Exception e) {
117       // If we catch an exception here that wasn't handled before then something bad happened and we'll abort
118       // TODO: This is a dirty copy & paste job and should be replaced with a library but we don't have access to the
119       // CrawlerZooKeeperUpdatingListener at this point.
120       LOG.warn("Caught exception while crawling. Aborting crawl.", e);
121 
122       String crawlPath = CrawlerNodePaths.getCrawlInfoPath(message, FINISHED_CRAWLING);
123       Date date = new Date();
124       SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ");
125       byte[] data = dateFormat.format(date).getBytes(Charsets.UTF_8);
126       createOrUpdateZooKeeper(crawlPath, data);
127       crawlPath = CrawlerNodePaths.getCrawlInfoPath(message, FINISHED_REASON);
128       createOrUpdateZooKeeper(crawlPath, FinishReason.ABORT.toString().getBytes(Charsets.UTF_8));
129     } finally {
130       LOG.debug("Finished crawling [{}]", message);
131       MDC.remove("datasetKey");
132     }
133   }
134 
135   private void createOrUpdateZooKeeper(String crawlPath, byte[] data) {
136     try {
137       Stat stat = curator.checkExists().forPath(crawlPath);
138       if (stat == null) {
139         curator.create().forPath(crawlPath, data);
140       } else {
141         curator.setData().forPath(crawlPath, data);
142       }
143     } catch (Exception e1) {
144       LOG.error("Exception while updating ZooKeeper", e1);
145     }
146   }
