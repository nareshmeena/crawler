59    @Override
60    protected void runOneIteration() throws Exception {
61      LOG.info("Checking if any crawls have finished now.");
62      Set<DatasetProcessStatus> statuses;
63      try {
64        statuses = service.getRunningDatasetProcesses();
65        LOG.debug("All DatasetProcessStatuses retrieved: {}", statuses);
66      } catch (ServiceUnavailableException e) {
67        LOG.warn("Caught exception while trying to retrieve all running datasets, will try again", e);
68        return;
69      }
70  
71      for (DatasetProcessStatus status : statuses) {
72        LOG.info("Checking DatasetProcessStatus with UUID [{}] now", status.getDatasetKey());
73  
74        try {
75          updateRegistry(status);
76        } catch (Exception e) {
77          LOG.error("Unable to callback and update the registry. Aborting this cleanup round, will try again later.", e);
78          return;
79        }
80  
81        if (!checkDoneProcessing(status)) {
82          continue;
83        }
84  
85        LOG.debug(MAPPER.writeValueAsString(status));
86        try {
87          File file = new File(configuration.archiveDirectory,
88                               status.getDatasetKey() + "_" + status.getCrawlJob().getAttempt() + "_result.json");
89          Files.createParentDirs(file);
90          Files.write(MAPPER.writeValueAsBytes(status), file);
91        } catch (IOException e) {
92          LOG.warn("Could not write status due to exception. [{}]", status.getDatasetKey(), e);
93        }
94  
95        // If all of these things are true we can delete this dataset from ZK and dump info to disc
96        delete(status.getDatasetKey());
97      }
98      LOG.info("Done checking for finished crawls");
99    }

128   /**
129    * Updates the registry with up to date information about the status of processing this dataset.
130    *
131    * @throws Exception when there was an error updating the registry. This can happen when it's offline.
132    */
133   private void updateRegistry(DatasetProcessStatus status) throws Exception {
134     UUID datasetKey = status.getDatasetKey();
135     int attempt = status.getCrawlJob().getAttempt();
136     DatasetProcessStatus persisted = registryService.getDatasetProcessStatus(datasetKey, attempt);
137     LOG.info(MAPPER.writeValueAsString(status));
138     if (persisted == null) {
139       registryService.createDatasetProcessStatus(status);
140     } else {
141       registryService.updateDatasetProcessStatus(status);
142     }
143   }

145   /**
146    * This method checks whether a certain Dataset has been fully processed (crawled, occurrences processed etc.)
147    *
148    * @param status of the dataset in question to check
149    *
150    * @return {@code true} if we are done processing, {@code false} otherwise
151    */
152   private boolean checkDoneProcessing(DatasetProcessStatus status) {
153     if (status.getFinishedCrawling() == null) {
154       return false;
155     }
156 
157     // Done fragmenting?
158     // We are done when we have as many pages fragmented (in error or successful) as we did crawl
159     if (status.getPagesCrawled() != status.getPagesFragmentedError() + status.getPagesFragmentedSuccessful()) {
160       return false;
161     }
162 
163     // Done persisting fragments?
164     // We are done when we have processed as many fragments as the fragmenter emitted. During this processing we could
165     // generate more raw occurrence records than we got fragments due to ABCD2
166     if (status.getFragmentsProcessed() != status.getFragmentsEmitted()) {
167       return false;
168     }
169 
170     // Are we done persisting verbatim occurrences?
171     // We are done when we have persisted (in error or successful) as many verbatim records as there were new or
172     // updated raw occurrences in the previous steps
173     if (status.getVerbatimOccurrencesPersistedSuccessful() + status.getVerbatimOccurrencesPersistedError()
174         != status.getRawOccurrencesPersistedNew() + status.getRawOccurrencesPersistedUpdated()) {
175       return false;
176     }
177 
178     // Are we done interpreting occurrences?
179     // We are done when we have interpreted (in error or successful) as many occurrences as there were successful
180     // verbatim occurrences persisted
181     if (status.getInterpretedOccurrencesPersistedSuccessful() + status.getInterpretedOccurrencesPersistedError()
182         != status.getVerbatimOccurrencesPersistedSuccessful()) {
183       return false;
184     }
185 
186     return true;
187   }

189   private void delete(UUID datasetKey) {
190     LOG.info("Done with [{}]", datasetKey);
191     String path = CrawlerNodePaths.getCrawlInfoPath(datasetKey);
192     try {
193       recursiveDelete(path, curator.getChildren().forPath(path));
194       curator.delete().forPath("crawls/" + datasetKey);
195     } catch (Exception e) {
196       LOG.info("Couldn't delete [{}]", datasetKey, e);
197     }
198   }
199 
200   private void recursiveDelete(String parentPath, Collection<String> paths) throws Exception {
201     if (paths.isEmpty()) {
202       return;
203     }
204 
205     for (String path : paths) {
206       String fullPath = parentPath + "/" + path;
207       recursiveDelete(fullPath, curator.getChildren().forPath(fullPath));
208       curator.delete().forPath(fullPath);
209     }
210   }
