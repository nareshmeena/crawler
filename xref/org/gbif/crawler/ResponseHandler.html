9   /**
10   * Implementations of this interface will handle the response received by {@link CrawlClient}s and produce a result but
11   * it also plays a part in the paging through resources.
12   * <p/>
13   * They have three jobs:
14   * <ul>
15   * <li>Convert the response into a format that is suitable for shipping as a message</li>
16   * <li>Extract information from the response that is important for paging</li>
17   * <li>Extract enough information from the response to see if it is a duplicate response</li>
18   * </ul>
19   *
20   * @param <RESPONSE> the type of response to handle. This is dependent on the {@link CrawlClient} implementation
21   * @param <RESULT>   the type of result this response handler produces when processing a response
22   */
23  public interface ResponseHandler<RESPONSE, RESULT> {
24  
25    /**
26     * Handles a response and produces a result.
27     *
28     * @param response to be handled
29     *
30     * @throws ProtocolException  Implementation specific
31     * @throws TransportException Implementation specific
32     */
33    RESULT handleResponse(RESPONSE response) throws FatalCrawlException, ProtocolException, TransportException;
34  
35    /**
36     * This can be used to see if this response handler is in an illegal state or not. This can happen after parsing a
37     * wrong response or if there has been a transport exception. A proper response needs to be handled for this to be
38     * cleared.
39     *
40     * @return false if the other getter methods will throw a {@link IllegalStateException}
41     */
42    boolean isValidState();
43  
44    /**
45     * @return the number of records in this page
46     */
47    Optional<Integer> getRecordCount();
48  
49    /**
50     * @return true if there are more records to extract
51     */
52    Optional<Boolean> isEndOfRecords();
53  
54    /**
55     * @return a hash value for the current page. This can be used to detect duplicate pages.

57    Optional<Long> getContentHash();
