055  @Before

057    server = new TestingServer();
058    curator = CuratorFrameworkFactory.builder()
059      .connectString(server.getConnectString())
060      .namespace("crawler")
061      .retryPolicy(new RetryOneTime(1))
062      .build();
063    curator.start();
064    ZKPaths.mkdirs(curator.getZookeeperClient().getZooKeeper(), "/crawler/crawls");
065
066    service = new CrawlerCoordinatorServiceImpl(curator, datasetService);
067    dataset.setType(DatasetType.OCCURRENCE);
068    when(datasetService.get(uuid)).thenReturn(dataset);
069  }

077  @Test
078  public void testPrioritySort() {
079    List<Endpoint> endpoints = Lists.newArrayList();
080
081    Endpoint endpoint1 = new Endpoint();
082    endpoint1.setType(EndpointType.DWC_ARCHIVE);
083    endpoints.add(endpoint1);
084
085    Endpoint endpoint2 = new Endpoint();
086    endpoint2.setType(EndpointType.FEED);
087    endpoints.add(endpoint2);
088
089    Endpoint endpoint3 = new Endpoint();
090    endpoint3.setType(EndpointType.DIGIR);
091    endpoints.add(endpoint3);
092
093    Endpoint endpoint4 = new Endpoint();
094    endpoint4.setType(EndpointType.BIOCASE);
095    endpoints.add(endpoint4);
096
097    List<Endpoint> sortedEndpoints = service.prioritySortEndpoints(endpoints);
098
099    assertThat(sortedEndpoints.size(), equalTo(3));
100    assertThat(sortedEndpoints, contains(endpoint1, endpoint4, endpoint3));
101  }

154  @Test
155  public void testSchedule() throws Exception {
156    URI url = URI.create("http://gbif.org/index.html");
157    Endpoint endpoint = new Endpoint();
158    endpoint.setType(EndpointType.TAPIR);
159    endpoint.setUrl(url);
160
161    dataset.getEndpoints().add(endpoint);
162    dataset.getMachineTags().add(MachineTag.newInstance("metasync.gbif.org", "conceptualSchema", "foo"));
163    MachineTag tag = MachineTag.newInstance(CRAWL_ATTEMPT.getNamespace().getNamespace(), CRAWL_ATTEMPT.getName(), "10");
164    tag.setKey(123);
165    dataset.getMachineTags().add(tag);
166    dataset.getMachineTags()
167      .add(MachineTag.newInstance(DECLARED_RECORD_COUNT.getNamespace().getNamespace(),
168                                  DECLARED_RECORD_COUNT.getName(),
169                                  "1234"));
170    service.initiateCrawl(uuid, 5);
171
172    List<String> children = curator.getChildren().forPath("/crawls");
173    assertThat(children.size(), equalTo(1));
174    assertThat(children.get(0), equalTo(uuid.toString()));
175
176    byte[] bytes = curator.getData().forPath("/crawls/" + uuid);
177    ObjectMapper mapper = new ObjectMapper();
178    Map<?, ?> map = mapper.readValue(bytes, Map.class);
179
180    Integer attempt = (Integer) map.get("attempt");
181    assertThat(attempt, equalTo(11));
182
183    bytes = curator.getData().forPath("/crawls/" + uuid + "/" + CrawlerNodePaths.DECLARED_COUNT);
184    assertThat("1234", equalTo(new String(bytes)));
185  }
