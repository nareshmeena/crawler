001package org.gbif.crawler;
002
003import org.gbif.common.messaging.api.messages.StartCrawlMessage;
004
005import java.util.UUID;
006
007import org.junit.Test;
008
009import static org.mockito.Matchers.eq;
010import static org.mockito.Matchers.isA;
011import static org.mockito.Mockito.doThrow;
012import static org.mockito.Mockito.mock;
013import static org.mockito.Mockito.verify;
014
015public class StartCrawlMessageCallbackTest {
016
017  @Test
018  public void testCallback() {
019    CrawlerCoordinatorService service = mock(CrawlerCoordinatorService.class);
020
021    StartCrawlMessageCallback callback = new StartCrawlMessageCallback(service);
022
023    StartCrawlMessage message = new StartCrawlMessage(UUID.randomUUID(), 5);
024    callback.handleMessage(message);
025
026    verify(service).initiateCrawl(message.getDatasetUuid(), message.getPriority().get());
027  }
028
029  @Test
030  public void testFailedCallback() {
031    CrawlerCoordinatorService service = mock(CrawlerCoordinatorService.class);
032    doThrow(new IllegalArgumentException("foo")).when(service).initiateCrawl(isA(UUID.class), eq(5));
033
034    StartCrawlMessageCallback callback = new StartCrawlMessageCallback(service);
035
036    StartCrawlMessage message = new StartCrawlMessage(UUID.randomUUID(), 5);
037    callback.handleMessage(message);
038  }
039}
