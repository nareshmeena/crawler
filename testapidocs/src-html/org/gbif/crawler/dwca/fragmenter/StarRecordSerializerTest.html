001package org.gbif.crawler.dwca.fragmenter;
002
003import org.gbif.crawler.dwca.util.DwcaTestUtil;
004import org.gbif.dwc.text.Archive;
005import org.gbif.dwc.text.ArchiveFactory;
006import org.gbif.dwc.text.StarRecord;
007
008import java.io.File;
009import java.io.IOException;
010import java.util.Arrays;
011
012import org.junit.Test;
013
014import static org.junit.Assert.assertTrue;
015
016public class StarRecordSerializerTest {
017
018  @Test
019  public void testIdenticalJson() throws IOException {
020    String archiveDir1 = DwcaTestUtil.openArchive("/dwca/dwca-one-hundred-good-triplets-good-ids.zip");
021    Archive archive1 = ArchiveFactory.openArchive(new File(archiveDir1));
022    StarRecord record1 = archive1.iterator().next();
023    DwcaTestUtil.cleanupArchive(archiveDir1);
024
025    String archiveDir2 = DwcaTestUtil.openArchive("/dwca/dwca-one-hundred-good-triplets-good-ids.zip");
026    Archive archive2 = ArchiveFactory.openArchive(new File(archiveDir2));
027    StarRecord record2 = archive2.iterator().next();
028    DwcaTestUtil.cleanupArchive(archiveDir2);
029
030    byte[] json1 = StarRecordSerializer.toJson(record1);
031    byte[] json2 = StarRecordSerializer.toJson(record2);
032    assertTrue(Arrays.equals(json1, json2));
033  }
034}
