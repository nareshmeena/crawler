001package org.gbif.crawler.dwca.util;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.util.UUID;
008import java.util.zip.ZipEntry;
009import java.util.zip.ZipInputStream;
010
011public class DwcaTestUtil {
012
013  public static String openArchive(String archiveFilePath) throws IOException {
014    String outputDir = UUID.randomUUID().toString();
015
016    File zipFile = new File(DwcaTestUtil.class.getResource(archiveFilePath).getFile());
017    FileInputStream fis = new FileInputStream(zipFile);
018    ZipInputStream zis = new ZipInputStream(fis);
019    ZipEntry ze = zis.getNextEntry();
020
021    while (ze != null) {
022      String fileName = ze.getName();
023      File newFile = new File(zipFile.getParent(), outputDir + File.separator + fileName);
024
025      new File(newFile.getParent()).mkdirs();
026      FileOutputStream fos = new FileOutputStream(newFile);
027
028      int len;
029      byte[] buffer = new byte[1024];
030      while ((len = zis.read(buffer)) > 0) {
031        fos.write(buffer, 0, len);
032      }
033
034      fos.close();
035      ze = zis.getNextEntry();
036    }
037
038    zis.closeEntry();
039    zis.close();
040    fis.close();
041
042    return zipFile.getParent() + File.separator + outputDir;
043  }
044
045  public static void cleanupArchive(String dirName) {
046    File openArchive = new File(dirName);
047    for (File file : openArchive.listFiles()) {
048      file.delete();
049    }
050  }
051}
