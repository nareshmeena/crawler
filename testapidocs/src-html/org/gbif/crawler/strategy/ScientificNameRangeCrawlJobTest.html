001package org.gbif.crawler.strategy;
002
003import org.junit.Test;
004
005import static org.fest.assertions.api.Assertions.assertThat;
006import static org.junit.Assert.fail;
007
008public class ScientificNameRangeCrawlJobTest {
009
010  @Test
011  public void testDefaultConstructor() {
012    ScientificNameRangeCrawlContext job = new ScientificNameRangeCrawlContext();
013
014    assertThat(job.getLowerBound().isPresent()).isFalse();
015    assertThat(job.getUpperBound().get()).isEqualTo("Aaa");
016    assertThat(job.getOffset()).isZero();
017  }
018
019  @Test
020  public void testCustomConstructor() {
021    ScientificNameRangeCrawlContext job;
022
023    job = new ScientificNameRangeCrawlContext(0, null, null);
024    assertThat(job.getLowerBound().isPresent()).isFalse();
025    assertThat(job.getUpperBound().isPresent()).isFalse();
026    assertThat(job.getOffset()).isZero();
027
028    job = new ScientificNameRangeCrawlContext(0, "aaa", null);
029    assertThat(job.getLowerBound().isPresent()).isTrue();
030    assertThat(job.getLowerBound().get()).isEqualTo("aaa");
031    assertThat(job.getUpperBound().isPresent()).isFalse();
032    assertThat(job.getOffset()).isZero();
033
034    job = new ScientificNameRangeCrawlContext(10, "aaa", "zzz");
035    assertThat(job.getLowerBound().isPresent()).isTrue();
036    assertThat(job.getLowerBound().get()).isEqualTo("aaa");
037    assertThat(job.getUpperBound().isPresent()).isTrue();
038    assertThat(job.getUpperBound().get()).isEqualTo("zzz");
039    assertThat(job.getOffset()).isEqualTo(10);
040
041    try {
042      new ScientificNameRangeCrawlContext(0, "", null);
043      fail();
044    } catch (IllegalArgumentException e) {
045    }
046
047    try {
048      new ScientificNameRangeCrawlContext(-10, null, null);
049      fail();
050    } catch (IllegalArgumentException e) {
051    }
052
053    try {
054      new ScientificNameRangeCrawlContext(0, null, "abcd");
055      fail();
056    } catch (IllegalArgumentException e) {
057    }
058
059  }
060
061  @Test
062  public void testSetAbsent() {
063    ScientificNameRangeCrawlContext job = new ScientificNameRangeCrawlContext(0, "aaa", "zzz");
064
065    job.setLowerBoundAbsent();
066    assertThat(job.getLowerBound().isPresent()).isFalse();
067    assertThat(job.getUpperBound().isPresent()).isTrue();
068
069    job.setUpperBoundAbsent();
070    assertThat(job.getLowerBound().isPresent()).isFalse();
071    assertThat(job.getUpperBound().isPresent()).isFalse();
072  }
073}
