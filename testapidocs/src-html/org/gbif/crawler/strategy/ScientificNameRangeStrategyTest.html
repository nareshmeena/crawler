001package org.gbif.crawler.strategy;
002
003import java.util.NoSuchElementException;
004
005import org.junit.Test;
006
007import static org.fest.assertions.api.Assertions.assertThat;
008import static org.junit.Assert.fail;
009
010public class ScientificNameRangeStrategyTest {
011
012  @Test
013  public void testDefaultJob() {
014    ScientificNameRangeStrategy strategy = new ScientificNameRangeStrategy(new ScientificNameRangeCrawlContext());
015    ScientificNameRangeCrawlContext next = strategy.next();
016
017    // Test if null...Aaa works
018    assertThat(next.getOffset()).isZero();
019    assertThat(next.getLowerBound().isPresent()).isFalse();
020    assertThat(next.getUpperBound().isPresent()).isTrue();
021    assertThat(next.getUpperBound().get()).isEqualTo("Aaa");
022
023    // Aaa...Aba
024    assertThat(strategy.hasNext()).isTrue();
025    next = strategy.next();
026    assertThat(next.getOffset()).isZero();
027    assertThat(next.getLowerBound().isPresent()).isTrue();
028    assertThat(next.getLowerBound().get()).isEqualTo("Aaa");
029    assertThat(next.getUpperBound().isPresent()).isTrue();
030    assertThat(next.getUpperBound().get()).isEqualTo("Aba");
031
032    // Aza...Baa
033    next.setLowerBound("Aya");
034    next.setUpperBound("Aza");
035    assertThat(strategy.hasNext()).isTrue();
036    next = strategy.next();
037    assertThat(next.getOffset()).isZero();
038    assertThat(next.getLowerBound().isPresent()).isTrue();
039    assertThat(next.getLowerBound().get()).isEqualTo("Aza");
040    assertThat(next.getUpperBound().isPresent()).isTrue();
041    assertThat(next.getUpperBound().get()).isEqualTo("Baa");
042
043    // Zxa...Zya, Zya...Zza
044    next.setLowerBound("Zxa");
045    next.setUpperBound("Zya");
046    assertThat(strategy.hasNext()).isTrue();
047    next = strategy.next();
048    assertThat(next.getOffset()).isZero();
049    assertThat(next.getLowerBound().isPresent()).isTrue();
050    assertThat(next.getLowerBound().get()).isEqualTo("Zya");
051    assertThat(next.getUpperBound().isPresent()).isTrue();
052    assertThat(next.getUpperBound().get()).isEqualTo("Zza");
053
054    // Zza...null
055    assertThat(strategy.hasNext()).isTrue();
056    next = strategy.next();
057    assertThat(next.getOffset()).isZero();
058    assertThat(next.getLowerBound().isPresent()).isTrue();
059    assertThat(next.getLowerBound().get()).isEqualTo("Zza");
060    assertThat(next.getUpperBound().isPresent()).isFalse();
061  }
062
063  @Test
064  public void testFailure() {
065    ScientificNameRangeStrategy strategy = new ScientificNameRangeStrategy(new ScientificNameRangeCrawlContext());
066    ScientificNameRangeCrawlContext next = strategy.next();
067
068    next.setLowerBound("Zza");
069    next.setUpperBoundAbsent();
070    // Make sure there are no more
071    assertThat(strategy.hasNext()).isFalse();
072
073    try {
074      strategy.next();
075      fail();
076    } catch (NoSuchElementException e) {
077    }
078  }
079
080}
