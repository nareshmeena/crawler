001package org.gbif.crawler.xml.crawlserver.util;
002
003import org.gbif.wrangler.lock.Lock;
004import org.gbif.wrangler.lock.LockFactory;
005import org.gbif.wrangler.lock.NoLockFactory;
006
007import org.junit.Test;
008
009import static org.fest.assertions.api.Assertions.assertThat;
010
011public class DelayingLockFactoryTest {
012
013  /*
014   * This test is not ideal because it depends on actual passing of time etc. The Guava library uses the concept of
015   * {@code Ticker} for this very reason but it seems overkill for our usage.
016   */
017  @Test
018  public void testDelay() {
019    LockFactory lockFactory = new DelayingLockFactory(new NoLockFactory(), 2000);
020    Lock lock = lockFactory.makeLock("foo");
021    long start = System.currentTimeMillis();
022    lock.lock();
023    lock.unlock();
024    long end = System.currentTimeMillis();
025    assertThat(end).isGreaterThan(start);
026    assertThat(start + 2000).isLessThanOrEqualTo(end);
027  }
028
029}
