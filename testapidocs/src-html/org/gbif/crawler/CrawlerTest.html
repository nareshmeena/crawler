044  /**
045   * This Mockito Answer returns a different optional long every time it is called. It can be initialized by providing
046   * a
047   * Iterator of Optional longs. Every time a present value is encountered a random long is returned. So the passed in
048   * sequence serves as a pattern which is cycled over.
049   */
050  private static class CyclingHashAnswer implements Answer<Optional<Long>> {
051
052    private final Random rnd = new Random();
053    private final Iterator<Optional<Long>> cycle;
054
055
056    /**
057     * The default is a cycle of present-absent-absent.
058     */
059    private CyclingHashAnswer() {
060      cycle = Iterables.cycle(Optional.of(1L), Optional.<Long>absent(), Optional.<Long>absent()).iterator();
061    }
062
063    private CyclingHashAnswer(Optional<Long>... values) {
064      cycle = Iterables.cycle(values).iterator();
065    }
066
067    @Override
068    public Optional<Long> answer(InvocationOnMock invocation) throws Throwable {
069      Optional<Long> next = cycle.next();
070      if (next.isPresent()) {
071        return Optional.of(rnd.nextLong());
072      }
073      return next;
074    }
075
076  }

103  @Before

105    context = new ScientificNameRangeCrawlContext();
106    strategy = new ScientificNameRangeStrategy(context);
107    retryPolicy = new LimitedRetryPolicy(1, 1, 1, 1);
108    crawler =
109      Crawler.newInstance(strategy, requestHandler, responseHandler, client, retryPolicy, NoLockFactory.getLock());

114  @Test
115  public void testAbortFatalCrawlException() throws Exception {
116    when(responseHandler.isEndOfRecords()).thenReturn(Optional.<Boolean>absent());
117    when(responseHandler.getRecordCount()).thenReturn(Optional.<Integer>absent());
118    when(responseHandler.getContentHash()).thenReturn(Optional.<Long>absent());
119    when(client.execute(anyString(), eq(responseHandler))).thenThrow(new FatalCrawlException("foo"));
120    crawler.crawl();
121    verify(client, times(1)).execute(anyString(), eq(responseHandler));
122  }
