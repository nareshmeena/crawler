001package org.gbif.crawler.protocol;
002
003import org.gbif.crawler.ResponseHandler;
004import org.gbif.crawler.exception.FatalCrawlException;
005import org.gbif.crawler.exception.ProtocolException;
006
007import java.io.IOException;
008import java.net.URL;
009import java.util.ArrayList;
010import java.util.Collection;
011
012import com.google.common.base.Optional;
013import com.google.common.io.Resources;
014import org.apache.http.HttpResponse;
015import org.codehaus.jackson.JsonNode;
016import org.codehaus.jackson.map.ObjectMapper;
017import org.junit.Test;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import static org.fest.assertions.api.Assertions.assertThat;
022import static org.junit.Assert.fail;
023import static org.mockito.Mockito.RETURNS_DEEP_STUBS;
024import static org.mockito.Mockito.mock;
025import static org.mockito.Mockito.when;
026
027public abstract class BaseParameterizedResponseHandlerTest {
028
029  private static final Logger LOG = LoggerFactory.getLogger(BaseParameterizedResponseHandlerTest.class);
030
031  /**
032   * This loads test data from JSON files and returns them in a way suitable for JUnit's {@link
033   * org.junit.runners.Parameterized} runner.
034   * <p/>
035   * All JSON files must consist of a top level array of objects with the following fields:
036   * <ul>
037   * <li>file_name (string)</li>
038   * <li>record_count (number), optional</li>
039   * <li>hash (number), optional</li>
040   * <li>end_of_records (boolean), optional</li>
041   * <li>exception_expected (boolean), optional</li>
042   * </ul>
043   *
044   * @param fileName to read the test data from
045   *
046   * @return a collection of Object arrays containing the test data
047   */
048  public static Collection<Object[]> getTestData(String fileName) throws IOException {
049    Collection<Object[]> objects = new ArrayList<Object[]>();
050    ObjectMapper mapper = new ObjectMapper();
051    JsonNode root = mapper.readValue(Resources.getResource(fileName), JsonNode.class);
052
053    for (JsonNode node : root) {
054      objects.add(new Object[] {
055        node.get("file_name").getTextValue(),
056        node.get("record_count") == null ? Optional.absent() : Optional.of(node.get("record_count").asInt()),
057        node.get("hash") == null ? Optional.absent() : Optional.of(node.get("hash").asLong()),
058        node.get("end_of_records") == null ? Optional.absent() : Optional.of(node.get("end_of_records").asBoolean()),
059        node.path("exception_expected").asBoolean(false)});
060    }
061
062    return objects;
063  }
064
065  public BaseParameterizedResponseHandlerTest(
066    String fileName,
067    Optional<Integer> recordCount,
068    Optional<Long> contentHash,
069    Optional<Boolean> endOfRecords,
070    boolean exceptionExpected
071  ) {
072    this.fileName = fileName;
073    this.recordCount = recordCount;
074    this.contentHash = contentHash;
075    this.endOfRecords = endOfRecords;
076    this.exceptionExpected = exceptionExpected;
077  }
078
079  private final String fileName;
080  private final Optional<Integer> recordCount;
081  private final Optional<Long> contentHash;
082  private final Optional<Boolean> endOfRecords;
083  private final boolean exceptionExpected;
084
085  protected abstract ResponseHandler<HttpResponse, ?> getResponseHandler();
086
087  @Test
088  public void testHandlingResponses() throws Exception {
089    LOG.info("Testing [{}]. Expecting [{}] records, [{}] hash, [{}] end of records",
090             new Object[] {fileName, recordCount, contentHash, endOfRecords});
091
092    HttpResponse response = mock(HttpResponse.class, RETURNS_DEEP_STUBS);
093
094    URL resource = Resources.getResource(fileName);
095    when(response.getEntity().getContent()).thenReturn(resource.openStream());
096
097    ResponseHandler<HttpResponse, ?> handler = getResponseHandler();
098
099    try {
100      handler.handleResponse(response);
101      if (exceptionExpected) {
102        assertThat(handler.isValidState()).isFalse();
103        fail();
104      }
105    } catch (ProtocolException e) {
106      if (exceptionExpected) {
107        assertThat(handler.isValidState()).isFalse();
108        return;
109      }
110
111      throw e;
112    } catch (FatalCrawlException e) {
113      if (exceptionExpected) {
114        assertThat(handler.isValidState()).isFalse();
115        return;
116      }
117
118      throw e;
119    }
120
121    assertThat(handler.isValidState()).as("Valid state").isTrue();
122    assertThat(handler.getContentHash()).as("Content hash").isEqualTo(contentHash);
123    assertThat(handler.getRecordCount()).as("Record count").isEqualTo(recordCount);
124    assertThat(handler.isEndOfRecords()).as("End of records").isEqualTo(endOfRecords);
125  }
126}
