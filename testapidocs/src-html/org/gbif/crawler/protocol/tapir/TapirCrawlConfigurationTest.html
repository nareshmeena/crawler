001package org.gbif.crawler.protocol.tapir;
002
003import java.net.URI;
004import java.util.UUID;
005
006import org.junit.Test;
007
008import static org.fest.assertions.api.Assertions.assertThat;
009import static org.junit.Assert.fail;
010
011public class TapirCrawlConfigurationTest {
012
013  @Test
014  public void testJob() {
015    URI targetUrl = URI.create("http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus");
016    UUID uuid = UUID.randomUUID();
017    String contentNamespace = "http://rs.tdwg.org/dwc/dwcore/";
018
019    testFailure(null, 1, targetUrl, contentNamespace, "datasetKey");
020    testFailure(uuid, 0, targetUrl, contentNamespace, "attempt");
021    testFailure(uuid, -10, targetUrl, contentNamespace, "attempt");
022    testFailure(uuid, 1, null, contentNamespace, "url");
023    testFailure(uuid, 1, targetUrl, null, "Namespace");
024    testFailure(uuid, 1, targetUrl, "foobar", "support");
025
026    TapirCrawlConfiguration job = new TapirCrawlConfiguration(uuid, 1, targetUrl, contentNamespace);
027
028    assertThat(job.getContentNamespace()).isEqualTo(contentNamespace);
029    assertThat(job.getUrl()).isEqualTo(targetUrl);
030    assertThat(job.getAttempt()).isEqualTo(1);
031    assertThat(job.getDatasetKey()).isEqualTo(uuid);
032  }
033
034  private void testFailure(UUID uuid, int attempt, URI url, String contentNamespace, String expectedString) {
035    try {
036      new TapirCrawlConfiguration(uuid, attempt, url, contentNamespace);
037      fail();
038    } catch (Exception ex) {
039      assertThat(ex).hasMessageContaining(expectedString);
040    }
041  }
042
043}
