001package org.gbif.crawler.protocol.tapir;
002
003import org.gbif.crawler.strategy.ScientificNameRangeCrawlContext;
004
005import java.net.URI;
006import java.util.UUID;
007
008import org.junit.Test;
009
010import static org.fest.assertions.api.Assertions.assertThat;
011import static org.junit.Assert.fail;
012
013public class TapirScientificNameRangeRequestHandlerTest {
014
015  private static final String FIRST_URL =
016    "http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus?limit=1000&upper=Aaa&lower&t=http%3A%2F%2Frs.gbif.org%2Ftemplates%2Ftapir%2Fdwc%2F1.4%2Fsci_name_range.xml&op=s&start=0";
017  private static final String SECOND_URL =
018    "http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus?limit=1000&upper=Aba&lower=Aaa&t=http%3A%2F%2Frs.gbif.org%2Ftemplates%2Ftapir%2Fdwc%2F1.4%2Fsci_name_range.xml&op=s&start=0";
019  private static final String LAST_URL =
020    "http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus?limit=1000&upper&lower=Zza&t=http%3A%2F%2Frs.gbif.org%2Ftemplates%2Ftapir%2Fdwc%2F1.4%2Fsci_name_range.xml&op=s&start=1000";
021
022  @Test
023  public void testUrlBuilding() {
024    URI targetUrl = URI.create("http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus");
025    TapirCrawlConfiguration
026      job = new TapirCrawlConfiguration(UUID.randomUUID(), 1, targetUrl, "http://rs.tdwg.org/dwc/dwcore/");
027    TapirScientificNameRangeRequestHandler handler = new TapirScientificNameRangeRequestHandler(job);
028
029    ScientificNameRangeCrawlContext context = new ScientificNameRangeCrawlContext();
030
031    assertThat(handler.buildRequestUrl(context)).isEqualTo(FIRST_URL);
032    context.setLowerBound("Aaa");
033    context.setUpperBound("Aba");
034    assertThat(handler.buildRequestUrl(context)).isEqualTo(SECOND_URL);
035
036    context.setLowerBound("Zza");
037    context.setUpperBoundAbsent();
038    context.setOffset(1000);
039    assertThat(handler.buildRequestUrl(context)).isEqualTo(LAST_URL);
040
041    // Null jobs are not allowed
042    try {
043      handler.buildRequestUrl(null);
044      fail();
045    } catch (Exception e) {
046
047    }
048  }
049
050  @Test
051  public void testConstructor() {
052    try {
053      new TapirScientificNameRangeRequestHandler(null);
054      fail();
055    } catch (Exception e) {
056    }
057  }
058}
