001package org.gbif.crawler.protocol.biocase;
002
003import java.net.URI;
004import java.util.UUID;
005
006import org.junit.Test;
007
008import static org.fest.assertions.api.Assertions.assertThat;
009import static org.junit.Assert.fail;
010
011public class BiocaseCrawlConfigurationTest {
012
013  @Test
014  public void testJob() {
015    URI targetUrl = URI.create("http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus");
016    UUID uuid = UUID.randomUUID();
017    String contentNamespace = "http://www.tdwg.org/schemas/abcd/1.2";
018    String datasetTitle = "Pontaurus";
019
020    testFailure(null, 1, targetUrl, contentNamespace, datasetTitle, "datasetKey");
021    testFailure(uuid, 0, targetUrl, contentNamespace, datasetTitle, "attempt");
022    testFailure(uuid, -10, targetUrl, contentNamespace, datasetTitle, "attempt");
023    testFailure(uuid, 1, null, contentNamespace, datasetTitle, "url");
024    testFailure(uuid, 1, targetUrl, null, datasetTitle, "Namespace");
025    testFailure(uuid, 1, targetUrl, "", datasetTitle, "Namespace");
026    testFailure(uuid, 1, targetUrl, "foobar", datasetTitle, "Namespace");
027
028    BiocaseCrawlConfiguration job = new BiocaseCrawlConfiguration(uuid, 1, targetUrl, contentNamespace, datasetTitle);
029
030    assertThat(job.getContentNamespace()).isEqualTo(contentNamespace);
031    assertThat(job.getUrl()).isEqualTo(targetUrl);
032    assertThat(job.getAttempt()).isEqualTo(1);
033    assertThat(job.getDatasetKey()).isEqualTo(uuid);
034    assertThat(job.getDatasetTitle()).isEqualTo(datasetTitle);
035  }
036
037
038  private void testFailure(UUID uuid, int attempt, URI url, String contentNamespace, String datasetTitle,
039    String expectedString) {
040    try {
041      new BiocaseCrawlConfiguration(uuid, attempt, url, contentNamespace, datasetTitle);
042      fail();
043    } catch (Exception ex) {
044      assertThat(ex).hasMessageContaining(expectedString);
045    }
046  }
047
048}
