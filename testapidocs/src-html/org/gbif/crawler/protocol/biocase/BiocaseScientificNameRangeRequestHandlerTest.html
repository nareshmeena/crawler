024  @Test
025  public void testUrlBuilding() {
026    URI targetUrl = URI.create("http://mockhost1.gbif.org/biocase/pywrapper.cgi?dsa=pontaurus");
027    BiocaseCrawlConfiguration job =
028      new BiocaseCrawlConfiguration(UUID.randomUUID(), 1, targetUrl, "http://www.tdwg.org/schemas/abcd/2.06", "Pontaurus");
029    BiocaseScientificNameRangeRequestHandler handler = new BiocaseScientificNameRangeRequestHandler(job);
030
031    ScientificNameRangeCrawlContext context = new ScientificNameRangeCrawlContext();
032    assertThat(handler.buildRequestUrl(context)).isEqualTo(FIRST_URL);
033
034    context.setLowerBound("Aaa");
035    context.setUpperBound("Aba");
036    assertThat(handler.buildRequestUrl(context)).isEqualTo(SECOND_URL);
037
038    context.setLowerBound("Zza");
039    context.setUpperBoundAbsent();
040    context.setOffset(1000);
041    assertThat(handler.buildRequestUrl(context)).isEqualTo(LAST_URL);
042
043    // Null jobs are not allowed
044    try {
045      handler.buildRequestUrl(null);
046      fail();
047    } catch (Exception e) {
048
049    }
050  }
