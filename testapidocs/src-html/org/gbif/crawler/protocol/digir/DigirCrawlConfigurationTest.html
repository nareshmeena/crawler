001package org.gbif.crawler.protocol.digir;
002
003import java.net.URI;
004import java.util.UUID;
005
006import org.junit.Test;
007
008import static org.fest.assertions.api.Assertions.assertThat;
009import static org.junit.Assert.fail;
010
011public class DigirCrawlConfigurationTest {
012
013  @Test
014  public void testJob() {
015    URI targetUrl = URI.create("http://mockhost1.gbif.org/tapirlink/tapir.php/pontaurus");
016    UUID uuid = UUID.randomUUID();
017    String resourceCode = "ent";
018
019    testFailure(null, 1, targetUrl, resourceCode, true, "datasetKey");
020    testFailure(uuid, 0, targetUrl, resourceCode, true, "attempt");
021    testFailure(uuid, -10, targetUrl, resourceCode, true, "attempt");
022    testFailure(uuid, 1, null, resourceCode, true, "url");
023    testFailure(uuid, 1, targetUrl, null, true, "resource");
024    testFailure(uuid, 1, targetUrl, "", true, "resource");
025
026    DigirCrawlConfiguration job = new DigirCrawlConfiguration(uuid, 1, targetUrl, resourceCode, true);
027
028    assertThat(job.getResourceCode()).isEqualTo(resourceCode);
029    assertThat(job.getUrl()).isEqualTo(targetUrl);
030    assertThat(job.getAttempt()).isEqualTo(1);
031    assertThat(job.getDatasetKey()).isEqualTo(uuid);
032    assertThat(job.isManis()).isTrue();
033  }
034
035  private void testFailure(UUID uuid, int attempt, URI url, String resourceCode, boolean manis,
036    String expectedString) {
037    try {
038      new DigirCrawlConfiguration(uuid, attempt, url, resourceCode, manis);
039      fail();
040    } catch (Exception ex) {
041      assertThat(ex).hasMessageContaining(expectedString);
042    }
043  }
044
045}
