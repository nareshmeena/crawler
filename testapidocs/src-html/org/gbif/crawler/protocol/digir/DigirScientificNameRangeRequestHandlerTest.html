024  @Test
025  public void testUrlBuilding() {
026    URI targetUrl = URI.create("http://peabody.research.yale.edu/digir/DiGIR.php");
027    DigirCrawlConfiguration job = new DigirCrawlConfiguration(UUID.randomUUID(), 1, targetUrl, "ent", true);
028    DigirScientificNameRangeRequestHandler handler = new DigirScientificNameRangeRequestHandler(job);
029
030    ScientificNameRangeCrawlContext context = new ScientificNameRangeCrawlContext();
031    String request = handler.buildRequestUrl(context);
032    request = request.replaceAll("sendTime.*sendTime", "");
033    assertThat(request).isEqualTo(FIRST_URL);
034
035    context.setLowerBound("Aaa");
036    context.setUpperBound("Aba");
037    request = handler.buildRequestUrl(context);
038    request = request.replaceAll("sendTime.*sendTime", "");
039    assertThat(request).isEqualTo(SECOND_URL);
040
041    context.setLowerBound("Zza");
042    context.setUpperBoundAbsent();
043    context.setOffset(1000);
044    request = handler.buildRequestUrl(context);
045    System.out.println(request);
046    request = request.replaceAll("sendTime.*sendTime", "");
047    assertThat(request).isEqualTo(LAST_URL);
048
049    // Null jobs are not allowed
050    try {
051      handler.buildRequestUrl(null);
052      fail();
053    } catch (Exception e) {
054
055    }
056  }
