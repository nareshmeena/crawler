013package org.gbif.crawler.constants;
014
015import java.util.UUID;
016import javax.annotation.Nullable;
017
018import com.google.common.base.Joiner;
019
020import static com.google.common.base.Preconditions.checkNotNull;
021
022/**
023 * Constant class for storing the Zookeeper node paths.
024 */
025public class CrawlerNodePaths {
026
027  public static final String SUCCESSFUL = "successful";
028  public static final String ERROR = "error";
029  public static final String CRAWL_INFO = "crawls";
030  public static final String QUEUED_CRAWLS = "queuedCrawls";
031  public static final String RUNNING_CRAWLS = "runningCrawls";
032  public static final String LOCKED_URLS = "lockedUrls";
033  public static final String XML_CRAWL = "xml";
034  public static final String DWCA_CRAWL = "dwca";
035  public static final String STARTED_CRAWLING = "startedCrawling";
036  public static final String FINISHED_CRAWLING = "finishedCrawling";
037  public static final String FINISHED_REASON = "finishedReason";
038  public static final String CRAWL_CONTEXT = "crawlContext";
039  public static final String DECLARED_COUNT = "declaredCount";
040  public static final String PAGES_CRAWLED = "pagesCrawled";
041  public static final String PAGES_FRAGMENTED_SUCCESSFUL = "pagesFragmented/" + SUCCESSFUL;
042  public static final String PAGES_FRAGMENTED_ERROR = "pagesFragmented/" + ERROR;
043  public static final String FRAGMENTS_EMITTED = "fragmentsEmitted";
044  public static final String FRAGMENTS_RECEIVED = "fragmentsReceived";
045  public static final String RAW_OCCURRENCES_PERSISTED_NEW = "rawOccurrencesPersisted/new";
046  public static final String RAW_OCCURRENCES_PERSISTED_UPDATED = "rawOccurrencesPersisted/updated";
047  public static final String RAW_OCCURRENCES_PERSISTED_UNCHANGED = "rawOccurrencesPersisted/unchanged";
048  public static final String RAW_OCCURRENCES_PERSISTED_ERROR = "rawOccurrencesPersisted/" + ERROR;
049  public static final String FRAGMENTS_PROCESSED = "fragmentsProcessed";
050  public static final String VERBATIM_OCCURRENCES_PERSISTED_SUCCESSFUL = "verbatimOccurrencesPersisted/" + SUCCESSFUL;
051  public static final String VERBATIM_OCCURRENCES_PERSISTED_ERROR = "verbatimOccurrencesPersisted/" + ERROR;
052  public static final String INTERPRETED_OCCURRENCES_PERSISTED_SUCCESSFUL =
053    "interpretedOccurrencesPersisted/" + SUCCESSFUL;
054  public static final String INTERPRETED_OCCURRENCES_PERSISTED_ERROR = "interpretedOccurrencesPersisted/" + ERROR;
055  private static final Joiner JOINER = Joiner.on('/').skipNulls();
056
057  /**
058   * Helper method to retrieve the path in ZooKeeper where all information about a specific dataset can be found.
059   *
060   * @param uuid to retrieve path for
061   *
062   * @return crawl info path for dataset
063   */
064  public static String getCrawlInfoPath(UUID uuid) {
065    return getCrawlInfoPath(uuid, null);
066  }
067
068  /**
069   * Helper method to retrieve a path under the {@link #CRAWL_INFO} node.
070   *
071   * @param uuid of the dataset to get the path for
072   * @param path if null we retrieve the path of the {@link #CRAWL_INFO} node itself otherwise we append this path
073   */
074  public static String getCrawlInfoPath(UUID uuid, @Nullable String path) {
075    checkNotNull(uuid, "uuid can't be null");
076    String resultPath = CRAWL_INFO + "/" + uuid.toString();
077    if (path != null) {
078      resultPath += "/" + path;
079    }
080
081    return resultPath;
082  }
083
084  /**
085   * Builds a "/" separated path out of path elements. The returned string will not have a "/" as the first and last
086   * character.
087   *
088   * @param paths to concatenate
089   */
090  public static String buildPath(String... paths) {
091    return JOINER.join(paths);
092  }
093
094  private CrawlerNodePaths() {
095  }
096
097}
