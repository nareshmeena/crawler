047/**
048 * This implementation stores crawls in ZooKeeper in a node structure like this:
049 * <ul>
050 * <li>{@code /crawls/<uuid>} holds general information about this crawl. The data attached to this node is supposed to
051 * be immutable throughout the crawl but subnodes will hold information about the current state of the job.</li>
052 * <li>{@code /queuedCrawls/<queuename>} is used as a priority queue. The children can be ordered lexicographically to
053 * get the order in which crawl jobs are supposed to be picked up</li>
054 * <li>{@code /runningCrawls/<queuename>} is used as a lock for {@code /queuedCrawls}. Every running crawler creates a
055 * lock here as long as it's running so that in case of failure the job can be picked up</li>
056 * </ul>
057 * <p/>
058 * This uses a framework named Curator (by Netflix) to help implement the Queue in ZooKeeper. For more information see
059 * the <a href="https://github.com/Netflix/curator/wiki/Distributed-Priority-Queue>documentation</a>. Please note that
060 * this means that we rely on Curator's serializationf format for the <em>queuedCrawls</em> and <em>runningCrawls</em>
061 * nodes.
062 */
063public class CrawlerCoordinatorServiceImpl implements CrawlerCoordinatorService {

068  private static final Comparator<Endpoint> ENDPOINT_COMPARATOR =
069    Collections.reverseOrder(new EndpointPriorityComparator());
070  private static final ObjectMapper MAPPER = new ObjectMapper();
071  private static final int DEFAULT_PRIORITY = 0;
072  private final CuratorFramework curator;

074  private final DistributedPriorityQueue<UUID> dwcaQueue;
075  private final DatasetService datasetService;
076  private final CrawlerCoordinatorServiceMetrics metrics = new CrawlerCoordinatorServiceMetrics();

078  /**
079   * Creates a CrawlerCoordinatorService for a specific ZooKeeper instance, pointing to a remote Registry WS.
080   *
081   * @param curator        to access ZooKeeper
082   * @param datasetService to access the Registry
083   *
084   * @throws ServiceUnavailableException when there was a failure during initialization of ZooKeeper
085   */
086  public CrawlerCoordinatorServiceImpl(
087    CuratorFramework curator, DatasetService datasetService
088  ) {
089    this.curator = checkNotNull(curator, "curator can't be null");
090    this.datasetService = checkNotNull(datasetService, "datasetService can't be null");
091
092    xmlQueue = buildQueue(curator, XML_CRAWL);
093    dwcaQueue = buildQueue(curator, DWCA_CRAWL);
094  }

096  @Override
097  public void initiateCrawl(UUID datasetKey, int priority) {
098    metrics.timerStart();
099    try {
100      doScheduleCrawl(datasetKey, priority);
101    } catch (RuntimeException e) {
102      metrics.unsuccessfulSchedule();
103      throw e;
104    } finally {
105      metrics.timerStop();
106    }
107
108    metrics.successfulSchedule(datasetKey);
109  }
110
111  @Override
112  public void initiateCrawl(UUID datasetKey) {
113    initiateCrawl(datasetKey, DEFAULT_PRIORITY);
114  }

116  /**
117   * Returns a list of valid Endpoints for the currently available Crawler implementations in a priority sorted order.
118   *
119   * @param endpoints to sort
120   *
121   * @return sorted and filtered list of Endpoints with the <em>best</em> ones at the beginning of the list
122   *
123   * @see EndpointPriorityComparator
124   */
125  @VisibleForTesting
126  List<Endpoint> prioritySortEndpoints(List<Endpoint> endpoints) {
127    checkNotNull(endpoints, "endpoints can't be null");
128
129    // Filter out all Endpoints that we can't crawl
130    List<Endpoint> result = Lists.newArrayList();
131    for (Endpoint endpoint : endpoints) {
132      if (EndpointPriorityComparator.PRIORITIES.contains(endpoint.getType())) {
133        result.add(endpoint);
134      }
135    }
136
137    // Sort the remaining ones
138    Collections.sort(result, ENDPOINT_COMPARATOR);
139    return result;
140  }

142  /**
143   * Builds a Curator queue for the given path. The path will be a subpath to {@link CrawlerNodePaths#QUEUED_CRAWLS}.
144   */
145  private DistributedPriorityQueue<UUID> buildQueue(CuratorFramework curator, String path) {
146    QueueBuilder<UUID> builder =
147      QueueBuilder.builder(curator, null, new UuidSerializer(), CrawlerNodePaths.buildPath(path, QUEUED_CRAWLS));
148    DistributedPriorityQueue<UUID> queue =
149      builder.lockPath(CrawlerNodePaths.buildPath(path, RUNNING_CRAWLS)).buildPriorityQueue(1);
150
151    try {
152      curator.newNamespaceAwareEnsurePath(CRAWL_INFO).ensure(curator.getZookeeperClient());
153      queue.start();
154    } catch (Exception e) {
155      throw new ServiceUnavailableException("Error starting up Priority queue", e);
156    }
157
158    return queue;
159  }

161  /**
162   * This is the main entry point into this service and should be called by all methods wanting to schedule a crawl.
163   */
164  private void doScheduleCrawl(UUID datasetKey, int priority) {
165    checkNotNull(datasetKey, "datasetKey can't be null");
166    MDC.put("datasetKey", datasetKey.toString());
167
168    // We first have to check if the dataset can be crawled.
169    Dataset dataset = checkDataset(datasetService.get(datasetKey), datasetKey);
170    Optional<Endpoint> endpoint = getEndpointToCrawl(dataset);
171    if (!endpoint.isPresent()) {
172      metrics.noValidEndpoint();
173      throw new IllegalArgumentException("No eligible endpoints for dataset [" + datasetKey + "]");
174    }
175
176    // This object holds all information needed by Crawlers to crawl the endpoint
177    CrawlJob crawlJob = getCrawlJob(datasetKey, dataset, endpoint.get());
178
179    byte[] dataBytes = serializeCrawlJob(crawlJob);
180    queueCrawlJob(datasetKey, DWCA_ENDPOINTS.contains(endpoint.get().getType()), priority, dataBytes);
181    LOG.info("Crawling endpoint [{}] for dataset [{}]", endpoint.get().getUrl(), datasetKey);
182    writeDeclaredRecordCount(dataset, endpoint.get(), datasetKey);
183    MDC.remove("datasetKey");
184  }

186  /**
187   * Some of our datatsets declare how many records they hold. We save this count in ZooKeeper to later display it on
188   * the admin console.
189   */
190  private void writeDeclaredRecordCount(Dataset dataset, Endpoint endpoint, UUID datasetKey) {
191    List<MachineTag> allTags = dataset.getMachineTags();
192    allTags.addAll(endpoint.getMachineTags());
193    List<MachineTag> filteredTags = Lists.newArrayList();
194    for (MachineTag tag : allTags) {
195      if (tag.getNamespace().equals(DECLARED_RECORD_COUNT.getNamespace().getNamespace()) && tag.getName().equals(DECLARED_RECORD_COUNT.getName())) {
196        filteredTags.add(tag);
197      }
198    }
199
200    if (filteredTags.size() == 1) {
201      Long declaredCount = Long.parseLong(filteredTags.get(0).getValue());
202      try {
203        curator.create()
204          .creatingParentsIfNeeded()
205          .forPath(CrawlerNodePaths.getCrawlInfoPath(datasetKey, CrawlerNodePaths.DECLARED_COUNT),
206                   declaredCount.toString().getBytes(Charsets.UTF_8));
207      } catch (Exception e) {
208        throw new ServiceUnavailableException("Error communicating with ZooKeeper", e);
209      }
210    } else if (filteredTags.size() > 1) {
211      LOG.warn("Found more than one declaredRecordCount for dataset [{}]. Ignoring.", dataset.getKey());
212    }
213  }

215  /**
216   * Checks if a dataset is eligible to be crawled throwing {@link IllegalArgumentException} if not.
217   * <p/>
218   * <ul>
219   * <li>It needs to exist</li>
220   * <li>It mustn't be scheduled already</li>
221   * <li>It must have endpoints</li>
222   * </ul>
223   * <p/>
224   * This method does not check the validity of the endpoints itself. And this check is dependent on external state so
225   * the output might change between calls.
226   *
227   * @param dataset     to check
228   * @param datasetKey of the dataset to check, passed in separately to generate error messages in case the dataset
229   *                    doesn't even exist
230   *
231   * @return the dataset
232   *
233   * @throws RuntimeException when there is a problem communicating with ZooKeeper
234   */
235  private Dataset checkDataset(Dataset dataset, UUID datasetKey) {
236    // Does the dataset exist?
237    if (dataset == null) {
238      throw new IllegalArgumentException("Dataset [" + datasetKey + "] does not exist");
239    }
240
241    if (dataset.getDeleted() != null) {
242      throw new IllegalArgumentException("Dataset [" + datasetKey + "] is deleted");
243    }
244
245    // Is the dataset already scheduled to be crawled or currently being crawled?
246    Stat crawlNode;
247    try {
248      crawlNode = curator.checkExists().forPath(CrawlerNodePaths.getCrawlInfoPath(datasetKey));
249    } catch (Exception e) {
250      throw new ServiceUnavailableException("Exception while verifying dataset", e);
251    }
252    if (crawlNode != null) {
253      metrics.alreadyScheduled();
254      throw new AlreadyCrawlingException("Requested crawl for dataset ["
255                                         + datasetKey
256                                         + "] but crawl already scheduled or running, ignoring");
257    }
258
259    // Does the dataset have any endpoints?
260    if (dataset.getEndpoints().isEmpty()) {
261      metrics.noValidEndpoint();
262      throw new IllegalArgumentException("Dataset [" + datasetKey + "] does not have any endpoints");
263    }
264
265    return dataset;
266  }

268  /**
269   * Gets the endpoint that we want to crawl from the passed in dataset.
270   * <p/>
271   * We take into account a list of supported and prioritized endpoint types and verify that the declared dataset type
272   * matches a supported endpoint type.
273   *
274   * @param dataset to get the endpoint for
275   *
276   * @return will be present if we found an eligible endpoint
277   *
278   * @see EndpointPriorityComparator
279   */
280  private Optional<Endpoint> getEndpointToCrawl(Dataset dataset) {
281    // Are any of the endpoints eligible to be crawled
282    List<Endpoint> sortedEndpoints = prioritySortEndpoints(dataset.getEndpoints());
283    if (sortedEndpoints.isEmpty()) {
284      return Optional.absent();
285    }
286    Endpoint ep = sortedEndpoints.get(0);
287    return Optional.fromNullable(ep);
288  }

290  /**
291   * Populates a CrawlJob object with various protocol specific properties (using key-value pairs).
292   * <p/>
293   * These key-value pairs are using magic strings so whatever reads these things from ZooKeeper must use the same
294   * terminology. This was done to avoid polymorphic deserialization.
295   *
296   * @param datasetKey of the dataset to crawl,
297   * @param dataset     to crawl
298   * @param endpoint    to crawl
299   *
300   * @return populated object ready to be put into ZooKeeper
301   */
302  private CrawlJob getCrawlJob(UUID datasetKey, Dataset dataset, Endpoint endpoint) {
303    checkNotNull(dataset, "dataset can't be null");
304    checkNotNull(endpoint, "endpoint can't be null");
305
306    Map<String, String> properties = Maps.newHashMap();
307    switch (endpoint.getType()) {
308      case DIGIR:
309        fillPropertyFromTags(datasetKey, dataset.getMachineTags(), properties, METADATA_NAMESPACE, "code");
310        properties.put("manis", "false");
311        break;
312      case DIGIR_MANIS:
313        fillPropertyFromTags(datasetKey, dataset.getMachineTags(), properties, METADATA_NAMESPACE, "code");
314        properties.put("manis", "true");
315        break;
316      case TAPIR:
317        fillPropertyFromTags(datasetKey, dataset.getMachineTags(), properties, METADATA_NAMESPACE, "conceptualSchema");
318        break;
319      case BIOCASE:
320        properties.put("datasetTitle", dataset.getTitle());
321        fillPropertyFromTags(datasetKey,
322                             endpoint.getMachineTags(),
323                             properties,
324                             METADATA_NAMESPACE,
325                             "conceptualSchema");
326        break;
327      case DWC_ARCHIVE:
328        break;
329      default:
330        // This should never happen as it should've been caught earlier in the process
331        // probably a wrongly registered dataset
332        LOG.warn("Unsupported endpoint for occurrence dataset [{}] - [{}]", datasetKey, endpoint.getType());
333        break;
334    }
335
336    /* NOTE/TODO: This introduces a race-condition. If two coordinators were to do this at the same time they'd get a
337       new attempt number but only one of them could later create the node.
338
339       There is no easy solution to this: We could first take out a lock by creating the path for this UUID but then
340       we'd need to change the DatasetProcessService to ignore everything that doesn't have content which means an extra
341       check for each dataset. Another solution would be to write a dummy attempt number (like 0 or -1)but if a crawler
342       then picks up this crawl before we can change it to the proper number the crawler will use the wrong number.
343       All other solutions I could come up with were relatively complicated as well.
344
345       I've therefore opted to not change this. That means that the attempt number is not necessarily always increased
346       by one but might in very rare cases be incremented by more than one even though only one of these will then be
347       used.
348
349       This is definitely a design flaw and should be addressed at some point.
350      */
351    int attempt = getAttempt(datasetKey, dataset, endpoint);
352
353    return new CrawlJob(datasetKey, endpoint.getType(), URI.create(endpoint.getUrl().toString()), attempt, properties);
354  }

356  /**
357   * This retrieves the current attempt of crawling for this dataset, increments it by one (in the registry) and
358   * returns this new number.
359   */
360  private int getAttempt(UUID datasetKey, Dataset dataset, Endpoint endpoint) {
361    // Find the current attempt if it exists, if it doesn't exist it will be initialized to 1
362    List<MachineTag> allTags = dataset.getMachineTags();
363    List<MachineTag> filteredTags = Lists.newArrayList();
364    for (MachineTag tag : allTags) {
365      if (tag.getNamespace().equals(CRAWL_ATTEMPT.getNamespace().getNamespace()) && tag.getName()
366        .equals(CRAWL_ATTEMPT.getName())) {
367        filteredTags.add(tag);
368      }
369    }
370
371    int attempt = 0;
372    int tagKey = 0;
373    if (filteredTags.size() == 1) {
374      attempt = Integer.parseInt(filteredTags.get(0).getValue());
375      tagKey = filteredTags.get(0).getKey();
376    } else if (filteredTags.size() > 1) {
377      throw new ServiceUnavailableException("crawl_attempt tag exists more than once, clean up first and retry"
378                                            + " ["
379                                            + datasetKey
380                                            + "]");
381    }
382
383    attempt++;
384
385    MachineTag tag = new MachineTag(CRAWL_ATTEMPT.getNamespace().getNamespace(), CRAWL_ATTEMPT.getName(), null);
386    datasetService.deleteMachineTag(datasetKey, tagKey);
387    tag.setValue(String.valueOf(attempt));
388    datasetService.addMachineTag(datasetKey, tag);
389
390    metrics.registerCrawl(endpoint);
391    return attempt;
392  }

394  /**
395   * Looks for a specific predicate in a namespace and populates the properties map with the value if found, throwing
396   * an IllegalArgumentException otherwise.
397   *
398   * @param datasetKey these tags are from, used for error message
399   * @param tags        to search through
400   * @param properties  to populate
401   * @param namespace   to look for in tags
402   * @param predicate   to look for in tags
403   */
404  private void fillPropertyFromTags(
405    UUID datasetKey, Iterable<MachineTag> tags, Map<String, String> properties, String namespace, String predicate
406  ) {
407    Optional<String> value = findTag(tags, namespace, predicate);
408    if (!value.isPresent()) {
409      throw new IllegalArgumentException("Could not find ["
410                                         + namespace
411                                         + ":"
412                                         + predicate
413                                         + "] tag for dataset ["
414                                         + datasetKey
415                                         + "]");
416    }
417    properties.put(predicate, value.get());
418  }

420  /**
421   * Can be used to find the value of a certain namespace-name pair for a collection of tags.
422   *
423   * @param tags      to search in
424   * @param namespace to look for
425   * @param name      to look for
426   *
427   * @return {@code absent} if no such tag could be found, the value otherwise
428   */
429  private Optional<String> findTag(Iterable<MachineTag> tags, String namespace, String name) {
430    for (MachineTag tag : tags) {
431      if (namespace.equals(tag.getNamespace()) && name.equals(tag.getName())) {
432        // This assumes that all tags have a non-null value
433        return Optional.of(tag.getValue());
434      }
435    }
436    return Optional.absent();
437  }

439  /**
440   * Serializes a {@link CrawlJob} into JSON
441   *
442   * @param crawlJob to serialize
443   *
444   * @return byte array of the JSON format of the passed in object
445   */
446  private byte[] serializeCrawlJob(CrawlJob crawlJob) {
447    byte[] dataBytes;
448    try {
449      dataBytes = MAPPER.writeValueAsBytes(crawlJob);
450    } catch (IOException e) {
451      throw new ServiceUnavailableException("Error serializing the crawl status object", e);
452    }
453    return dataBytes;
454  }

456  /**
457   * This puts the passed in byte array in our crawl queue as well as in the general crawl info node.
458   *
459   * @param datasetKey of the dataset to enqueue
460   * @param dataBytes   to put into ZooKeeper
461   */
462  private void queueCrawlJob(UUID datasetKey, boolean isDwca, int priority, byte[] dataBytes) {
463    try {
464      // This could in theory fail when two coordinators are running at the same time. They'll have confirmed if this
465      // node exists or not earlier but in the meantime another Coordinator might have created it. That means the
466      // Exception we throw does not necessarily tell us enough. But this is an edge-case and I feel it's not worth
467      // trying to figure out why this failed.
468      curator.create().creatingParentsIfNeeded().forPath(CrawlerNodePaths.getCrawlInfoPath(datasetKey), dataBytes);
469      if (isDwca) {
470        dwcaQueue.put(datasetKey, priority);
471      } else {
472        xmlQueue.put(datasetKey, priority);
473      }
474    } catch (Exception e) {
475      throw new ServiceUnavailableException("Error communicating with ZooKeeper", e);
476    }
477  }
