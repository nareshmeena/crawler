001package org.gbif.crawler;
002
003/**
004 * This builds a protocol specific request for a certain crawl context.
005 * <p/>
006 * One example would be a request for a scientific name range for BioCASe.
007 *
008 * @param <CTX> type of context to handle. This could be a scientific name range filter or a request without any filter
009 * @param <REQ> type of the request that is being created. This is configurable so we can allow direct construction of
010 *              a complex query object if necessary but is expected to be mostly {@link String} for now.
011 */
012public interface RequestHandler<CTX extends CrawlContext, REQ> {
013
014  /**
015   * Builds a request for a context.
016   * <p/>
017   * This is dependent on the protocol being used as well as the {@link CrawlStrategy} through its {@link CrawlContext}
018   * implementation.
019   *
020   * @param context to build the request for
021   *
022   * @return request
023   */
024  REQ buildRequestUrl(CTX context);
025
026  /**
027   * Returns the number of records that this request handler will request at a time.
028   *
029   * @return number of records this request handler will request
030   */
031  int getLimit();
032
033}
