041/**
042 * Service that listens to DwcaDownloadFinishedMessages and puts found metadata documents into the metadata
043 * repository thereby updating the registered datasets information. Is aware of constituent datasets within an
044 * archive and therefore knows how to process Catalogue of Life GSD information.
045 */
046public class DwcaMetasyncService extends DwcaService {
047
048  private static final Logger LOG = LoggerFactory.getLogger(DwcaMetasyncService.class);
049  private final DwcaMetasyncConfiguration configuration;
050
051  public DwcaMetasyncService(DwcaMetasyncConfiguration configuration) {
052    super(configuration);
053    this.configuration = configuration;
054  }
055
056  @Override
057  protected void bindListeners() throws IOException {
058    CuratorFramework curator = configuration.zooKeeper.getCuratorFramework();
059
060    // listen to DwcaDownloadFinishedMessage messages
061    listener.listen(config.queueName, config.poolSize,
062                            new DwcaValidationFinishedMessageCallback(datasetService,
063                                                                    config.archiveRepository,
064                                                                    publisher,
065                                                                    curator));
066  }
067
068  private static class DwcaValidationFinishedMessageCallback extends AbstractMessageCallback<DwcaValidationFinishedMessage> {
069
070    private final DatasetService datasetService;
071    private final File archiveRepository;
072    private final MessagePublisher publisher;
073    private final CuratorFramework curator;
074
075    private final Counter messageCount = Metrics.newCounter(DwcaMetasyncService.class, "messageCount");
076    private final Counter datasetsUpdated = Metrics.newCounter(DwcaMetasyncService.class, "datasetsUpdated");
077    private final Counter constituentsAdded = Metrics.newCounter(DwcaMetasyncService.class, "constituentsAdded");
078    private final Counter constituentsDeleted = Metrics.newCounter(DwcaMetasyncService.class, "constituentsDeleted");
079    private final Counter constituentsUpdated = Metrics.newCounter(DwcaMetasyncService.class, "constituentsUpdated");
080
081    private DwcaValidationFinishedMessageCallback(
082      DatasetService datasetService, File archiveRepository, MessagePublisher publisher, CuratorFramework curator
083    ) {
084      this.archiveRepository = archiveRepository;
085      this.datasetService = datasetService;
086      this.publisher = publisher;
087      this.curator = curator;
088    }
089
090    @Override
091    public void handleMessage(DwcaValidationFinishedMessage message) {
092      MDC.put("datasetKey", message.getDatasetUuid().toString());
093      messageCount.inc();
094      UUID uuid = message.getDatasetUuid();
095      LOG.info("Updating metadata from DwC-A for dataset [{}]", uuid);
096
097      try {
098        handleMessageInternal(message, uuid);
099      } catch (Exception e) {
100        LOG.error("Exception caught during metasyncing DwC-A [{}]", uuid, e);
101        updateZookeeper(uuid);
102      } finally {
103        MDC.remove("datasetKey");
104      }
105    }
106
107    private void handleMessageInternal(DwcaValidationFinishedMessage message, UUID uuid) throws IOException {
108      if (!message.getValidationReport().isValid()) {
109        LOG.info("Ignoring message for invalid DwC-A for dataset [{}]", uuid);
110        return;
111      }
112
113      Dataset dataset = datasetService.get(uuid);
114
115      if (dataset == null) {
116        // exception, we don't know this dataset
117        throw new IllegalArgumentException("The requested dataset " + message.getDatasetUuid() + " is not registered");
118      }
119
120        Archive archive = LenientArchiveFactory.openArchive(new File(archiveRepository, uuid.toString()));
121        File metaFile = archive.getMetadataLocationFile();
122        if (metaFile != null && metaFile.exists()) {
123          // metadata found, put into repository thereby updating the dataset
124          setMetaDocument(metaFile, uuid);
125          datasetsUpdated.inc();
126        }
127
128        // process dataset constituents
129        Map<String, UUID> constituents = processConstituents(dataset, archive);
130
131        LOG.info("Finished updating metadata from DwC-A for dataset [{}]", uuid);
132
133        // send success message
134        publisher.send(new DwcaMetasyncFinishedMessage(uuid,
135                                                       dataset.getType(),
136                                                       message.getSource(),
137                                                       message.getAttempt(),
138                                                       constituents,
139                                                       message.getValidationReport()));
140    }
141
142    private Map<String, UUID> processConstituents(Dataset parent, Archive archive) {
143      Map<String, UUID> constituents = Maps.newHashMap();
144      // we don't expect to reach more than a few hundred constituents - so ignore paging here by using a 2000 p size
145      Pageable page = new PagingRequest(0, 2000);
146      List<Dataset> existingConstituents = datasetService.listConstituents(parent.getKey(), page).getResults();
147      Map<String, File> archiveConstituents = archive.getConstituentMetadata();
148      LOG.info("{} constituents metadata found in archive {}", archiveConstituents.size(), parent.getKey());
149
150      // go through each existing constituent and update its metadata or delete it from the registry
151      for (Dataset constituent : existingConstituents) {
152        try {
153          // we keep the datasetID as a tag, get it
154          String datasetId = getTagValue(constituent.getKey(), TagName.DATASET_ID);
155          if (datasetId == null) {
156            LOG.warn("Existing registered constituent found without a tagged datasetID. Please adjust manually {}",
157                     constituent.getKey());
158
159          } else {
160            // do we still have this constituent in the archive?
161            if (archiveConstituents.containsKey(datasetId)) {
162              // remove from archive map so we have only the new ones at the end
163              File metaFile = archiveConstituents.remove(datasetId);
164              setMetaDocument(metaFile, constituent.getKey());
165              constituentsUpdated.inc();
166              constituents.put(datasetId, constituent.getKey());
167
168            } else {
169              // constituent has been removed. Delete in registry
170              datasetService.delete(constituent.getKey());
171              constituentsDeleted.inc();
172            }
173          }
174        } catch (FileNotFoundException e) {
175          LOG.error("Failed to read archive constituent metadata file for already registered dataset {}",
176                    constituent.getKey(),
177                    e);
178        } catch (IllegalArgumentException e) {
179          LOG.error("Constituent dataset with UUID key expected, but got [{}]", parent.getKey(), e);
180        }
181      }
182
183      // now see if there are any new constituents left and create them
184      for (Map.Entry<String, File> constituent : archiveConstituents.entrySet()) {
185        String datasetId = constituent.getKey();
186        File metaFile = constituent.getValue();
187        UUID constituentKey = addNewConstituent(parent, parent.getKey(), datasetId, metaFile);
188        constituents.put(datasetId, constituentKey);
189      }
190
191      return constituents;
192    }
193
194    private boolean setMetaDocument(File metaDoc, UUID datasetKey) throws FileNotFoundException {
195      InputStream stream = new FileInputStream(metaDoc);
196      try {
197        datasetService.insertMetadata(datasetKey, stream);
198        return true;
199
200      } catch (IllegalArgumentException e) {
201        LOG.warn("Metadata document {} for dataset {} not understood", metaDoc.getAbsolutePath(), datasetKey, e);
202        return false;
203
204      } finally {
205        Closeables.closeQuietly(stream);
206      }
207    }
208
209    /**
210     * Creates a new constituent dataset of the same type as the parent in the registry which will be linked to the
211     * parent dataset and tagged with a datasetID.
212     * The metadata file will be uploaded into the repository and its information used to update the constituent
213     * dataset.
214     *
215     * @return the newly created constituent dataset key
216     */
217    private UUID addNewConstituent(Dataset parent, UUID parentKey, String datasetID, File metaFile) {
218      Dataset constituent = new Dataset();
219      // use temporary required title that will get overwritten when submitting the metadata file
220      constituent.setTitle("Constituent " + datasetID + " of " + parent.getTitle());
221      constituent.setParentDatasetKey(parentKey);
222      constituent.setOwningOrganizationKey(parent.getOwningOrganizationKey());
223      constituent.setInstallationKey(parent.getInstallationKey());
224      constituent.setType(parent.getType());
225      constituent.setSubtype(parent.getSubtype());
226
227      UUID key = null;
228      try {
229        key = datasetService.create(constituent);
230        LOG.info("Created new constituent {} with key {} for dataset {}", datasetID, key, parentKey);
231        constituentsAdded.inc();
232
233        MachineTag idTag = MachineTag.newInstance(TagName.DATASET_ID.getNamespace().getNamespace(), TagName.DATASET_ID.getName(), datasetID);
234        datasetService.addMachineTag(key, idTag);
235
236        setMetaDocument(metaFile, key);
237
238      } catch (IllegalArgumentException e) {
239        // sth went wrong with this constituent. Maybe bad metadata?
240        LOG.warn("Failed to create new constituent {} for dataset {}", datasetID, parentKey, e);
241
242      } catch (FileNotFoundException e) {
243        LOG.warn("Failed to upload metadata file for new constituent {} of dataset {}", datasetID, parentKey, e);
244      }
245
246      return key;
247    }
248
249    /**
250     * Convenience method to retrieve a single tag for a given name enum or null.
251     * If there are multiple tags with the same name only the first is returned.
252     *
253     * @param name the name to look for
254     */
255    // TODO(lfrancke): I think this is bad behavior as it usually indicates something wrong in the registry if there are
256    // multiple tags with the same name (when we expect only one) and we should rather fail fast and clean up
257    private String getTagValue(UUID datasetKey, TagName name) {
258      List<MachineTag> tags = datasetService.listMachineTags(datasetKey);
259      for (MachineTag tag : tags) {
260        if (tag.getNamespace().equals(name.getNamespace().getNamespace()) && name.getName().equals(tag.getName())) {
261          return tag.getValue();
262        }
263      }
264      return null;
265    }
266
267    private void updateZookeeper(UUID uuid) {RetryPolicy retryPolicy = new RetryNTimes(5, 1000);
268      String path = CrawlerNodePaths.getCrawlInfoPath(uuid, PAGES_FRAGMENTED_ERROR);
269      DistributedAtomicLong dal = new DistributedAtomicLong(curator, path, retryPolicy);
270      try {
271        dal.trySet(1L);
272      } catch (Exception e) {
273        LOG.error("Failed to update counter for successful DwC-A fragmenting", e);
274      }
275    }
