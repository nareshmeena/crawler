001package org.gbif.crawler.dwca.validator;
002
003import org.gbif.api.model.crawler.DwcaValidationReport;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.service.registry.DatasetService;
006import org.gbif.common.messaging.AbstractMessageCallback;
007import org.gbif.common.messaging.api.MessagePublisher;
008import org.gbif.common.messaging.api.messages.DwcaDownloadFinishedMessage;
009import org.gbif.common.messaging.api.messages.DwcaValidationFinishedMessage;
010import org.gbif.crawler.constants.CrawlerNodePaths;
011import org.gbif.crawler.dwca.DwcaService;
012import org.gbif.crawler.dwca.LenientArchiveFactory;
013import org.gbif.dwc.text.Archive;
014import org.gbif.dwc.text.UnsupportedArchiveException;
015
016import java.io.File;
017import java.io.IOException;
018import java.util.UUID;
019
020import com.yammer.metrics.Metrics;
021import com.yammer.metrics.core.Counter;
022import org.apache.curator.RetryPolicy;
023import org.apache.curator.framework.CuratorFramework;
024import org.apache.curator.framework.recipes.atomic.DistributedAtomicLong;
025import org.apache.curator.retry.RetryNTimes;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029import static org.gbif.crawler.constants.CrawlerNodePaths.PAGES_FRAGMENTED_ERROR;
030
031public class ValidatorService extends DwcaService {
032
033  private static final Logger LOG = LoggerFactory.getLogger(ValidatorService.class);
034
035  private final DwcaValidatorConfiguration configuration;
036
037
038  public ValidatorService(DwcaValidatorConfiguration configuration) {
039    super(configuration);
040    this.configuration = configuration;
041  }
042
043  @Override
044  protected void bindListeners() throws IOException {
045    CuratorFramework curator = configuration.zooKeeper.getCuratorFramework();
046
047    // listen to DwcaDownloadFinishedMessage messages
048    listener.listen(config.queueName,
049                    config.poolSize,
050                    new DwcaDownloadFinishedMessageCallback(datasetService,
051                                                            config.archiveRepository,
052                                                            publisher,
053                                                            curator));
054  }
055
056  private static class DwcaDownloadFinishedMessageCallback extends AbstractMessageCallback<DwcaDownloadFinishedMessage> {
057
058    private final DatasetService datasetService;
059    private final File archiveRepository;
060    private final MessagePublisher publisher;
061    private final CuratorFramework curator;
062
063    private final Counter messageCount = Metrics.newCounter(ValidatorService.class, "messageCount");
064    private final Counter failedValidations = Metrics.newCounter(ValidatorService.class, "failedValidations");
065
066    private DwcaDownloadFinishedMessageCallback(
067      DatasetService datasetService, File archiveRepository, MessagePublisher publisher, CuratorFramework curator
068    ) {
069      this.datasetService = datasetService;
070      this.archiveRepository = archiveRepository;
071      this.publisher = publisher;
072      this.curator = curator;
073    }
074
075    @Override
076    public void handleMessage(DwcaDownloadFinishedMessage message) {
077      messageCount.inc();
078
079      UUID uuid = message.getDatasetUuid();
080
081      if (!message.isModified()) {
082        LOG.info("DwC-A for dataset [{}] was not modified, skipping validation", uuid);
083        return;
084      }
085
086      LOG.info("Now validating DwC-A for dataset [{}]", uuid);
087      Dataset dataset = datasetService.get(uuid);
088      if (dataset == null) {
089        // exception, we don't know this dataset
090        throw new IllegalArgumentException("The requested dataset " + uuid + " is not registered");
091      }
092
093      boolean valid = false;
094
095
096      DwcaValidationReport validationReport = null;
097      try {
098        Archive archive = LenientArchiveFactory.openArchive(new File(archiveRepository, uuid.toString()));
099        validationReport = DwcaValidator.validate(dataset, archive);
100        valid = validationReport.isValid();
101
102      } catch (UnsupportedArchiveException e) {
103        LOG.warn("Invalid Dwc archive for dataset {}", uuid, e);
104
105      } catch (IOException e) {
106        LOG.warn("IOException when reading dwc archive for dataset {}", uuid, e);
107      }
108
109      if (!valid) {
110        failedValidations.inc();
111        if (validationReport == null) {
112          // archive couldn't be opened and/or read
113          validationReport = new DwcaValidationReport(dataset.getKey(), 0, 0, 0, 0, 0, false);
114        }
115
116        // If invalid we also need to update ZooKeeper
117        RetryPolicy retryPolicy = new RetryNTimes(5, 1000);
118
119        String path = CrawlerNodePaths.getCrawlInfoPath(uuid, PAGES_FRAGMENTED_ERROR);
120        DistributedAtomicLong dal = new DistributedAtomicLong(curator, path, retryPolicy);
121        try {
122          dal.trySet(1L);
123        } catch (Exception e) {
124          LOG.error("Failed to update counter for successful DwC-A fragmenting", e);
125        }
126      }
127
128      LOG.info("Finished validating DwC-A for dataset [{}], valid? is [{}]. Full report [{}]",
129               uuid,
130               valid,
131               validationReport);
132
133      // send validation finished message
134      try {
135        publisher.send(new DwcaValidationFinishedMessage(uuid,
136                                                         dataset.getType(),
137                                                         message.getSource(),
138                                                         message.getAttempt(),
139                                                         validationReport));
140      } catch (IOException e) {
141        LOG.warn("Failed to send validation finished message for dataset {}", uuid, e);
142      }
143    }
144  }
145
146}
