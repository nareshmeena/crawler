001package org.gbif.crawler.dwca;
002
003import org.gbif.dwc.text.Archive;
004import org.gbif.dwc.text.ArchiveFactory;
005import org.gbif.dwc.text.UnsupportedArchiveException;
006import org.gbif.file.DownloadUtil;
007import org.gbif.utils.file.CompressionUtil;
008
009import java.io.File;
010import java.io.IOException;
011import java.net.URL;
012import java.util.regex.Pattern;
013
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019/**
020 * A utility that wraps up the ArchiveFactory, but addresses a critical IPT bug with a workaround.
021 * https://code.google.com/p/gbif-providertoolkit/issues/detail?id=1015
022 */
023public class LenientArchiveFactory {
024
025  private static final Logger LOG = LoggerFactory.getLogger(LenientArchiveFactory.class);
026  private static final Pattern DOUBLE_BACKSLASH = Pattern.compile("\\\\");
027
028  public static Archive openArchive(File unzippedFolderLocation) throws IOException, UnsupportedArchiveException {
029    checkNotNull(unzippedFolderLocation, "unzippedFolderLocation can not be null");
030
031    for (File f : unzippedFolderLocation.listFiles()) {
032      if (f.getName().startsWith("\\")) {
033        String cleanedName = DOUBLE_BACKSLASH.matcher(f.getName()).replaceFirst("");
034        LOG.info("Handling IPT windows bug[code issue #1015], renaming {} to {}", f.getName(), cleanedName);
035        if (!f.renameTo(new File(unzippedFolderLocation, cleanedName))) {
036          LOG.warn("Failed to rename file with backslash, archive {} probably won't work as expected", f.getName());
037        }
038      }
039    }
040    return ArchiveFactory.openArchive(unzippedFolderLocation);
041  }
042
043  /**
044   * See {@link ArchiveFactory}
045   */
046  public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
047    try {
048      CompressionUtil.decompressFile(archiveDir, archiveFile);
049      return openArchive(archiveDir);
050
051    } catch (CompressionUtil.UnsupportedCompressionType e) {
052      return openArchive(archiveFile);
053    }
054  }
055
056  /**
057   * See {@link ArchiveFactory}
058   */
059  public static Archive openArchive(URL archiveUrl, File workingDir) throws IOException, UnsupportedArchiveException {
060    File downloadTo = new File(workingDir, "dwca-download");
061    File dwca = new File(workingDir, "dwca");
062    DownloadUtil.download(archiveUrl, downloadTo);
063    return openArchive(downloadTo, dwca);
064  }
065}
