025import com.google.common.cache.CacheBuilder;
026import com.google.common.cache.CacheLoader;
027import com.google.common.cache.LoadingCache;
028import com.google.common.util.concurrent.AbstractIdleService;
029import org.apache.curator.RetryPolicy;
030import org.apache.curator.framework.CuratorFramework;
031import org.apache.curator.framework.recipes.atomic.AtomicValue;
032import org.apache.curator.framework.recipes.atomic.DistributedAtomicLong;

037/**
038 * This service will listen for {@link DwcaValidationFinishedMessage} from RabbitMQ and for every valid archive it will
039 * extract all {@link StarRecord} and hand them back to RabbitMQ as {@link OccurrenceFragmentedMessage}s. While it does
040 * this it'll update appropriate counters in ZooKeeper.
041 */
042public class DwcaFragmenterService extends AbstractIdleService {
043
044  private static final Logger LOG = LoggerFactory.getLogger(DwcaFragmenterService.class);
045  private final DwcaFragmenterConfiguration configuration;
046  private MessagePublisher publisher;
047  private MessageListener listener;
048  private CuratorFramework curator;
049
050  public DwcaFragmenterService(DwcaFragmenterConfiguration configuration) {
051    this.configuration = configuration;
052  }
053
054  @Override
055  protected void startUp() throws Exception {
056
057    if (!configuration.archiveRepository.exists() && !configuration.archiveRepository.isDirectory()) {
058      throw new IllegalArgumentException("Archive repository needs to be an existing directory: "
059                                         + configuration.archiveRepository.getAbsolutePath());
060    }
061    if (!configuration.archiveRepository.canWrite()) {
062      throw new IllegalArgumentException("Archive repository directory not writable: " + configuration.archiveRepository
063        .getAbsolutePath());
064    }
065
066    // A cache for DistributedAtomicLongs to update ZooKeeper. NOTE: I'm not sure if this is actually worth it
067    // TODO: Unhardcode the paths here in favor of CrawlerNodePaths
068    LoadingCache<String, DistributedAtomicLong> cache =
069      CacheBuilder.newBuilder().maximumSize(1000).build(new CacheLoader<String, DistributedAtomicLong>() {
070
071        private final RetryPolicy retryPolicy = new RetryNTimes(5, 1000);
072
073        @Override
074        public DistributedAtomicLong load(String key) throws Exception {
075          return new DistributedAtomicLong(curator, "/" + CrawlerNodePaths.CRAWL_INFO + "/" + key, retryPolicy);
076        }
077      });
078
079    curator = configuration.zooKeeper.getCuratorFramework();
080
081    publisher = new DefaultMessagePublisher(configuration.messaging.getConnectionParameters());
082    listener = new MessageListener(configuration.messaging.getConnectionParameters());
083    listener.listen(configuration.queueName,
084                    configuration.poolSize,
085                    new DwCaCallback(cache, publisher, configuration.archiveRepository));
086  }
087
088  @Override
089  protected void shutDown() throws Exception {
090    publisher.close();
091    listener.close();
092    curator.close();
093
094  }
095
096  /**
097   * This callback handles {@link DwcaValidationFinishedMessage}s and extracts {@link StarRecord}s out of the archives
098   * and sends them on as messages.
099   */
100  private static class DwCaCallback extends AbstractMessageCallback<DwcaMetasyncFinishedMessage> {
101
102    private static final int COUNTER_UPDATE_FREQUENCY = 1000;
103    private final LoadingCache<String, DistributedAtomicLong> counterCache;
104    private final MessagePublisher publisher;
105    private final File archiveRepository;
106
107    private DwCaCallback(
108      LoadingCache<String, DistributedAtomicLong> cache, MessagePublisher publisher, File archiveRepository
109    ) {
110      counterCache = cache;
111      this.publisher = publisher;
112      this.archiveRepository = archiveRepository;
113    }
114
115    @Override
116    public void handleMessage(DwcaMetasyncFinishedMessage message) {
117      UUID uuid = message.getDatasetUuid();
118
119      Archive archive;
120      try {
121        archive = LenientArchiveFactory.openArchive(new File(archiveRepository, uuid.toString()));
122      } catch (IOException e) {
123        LOG.error("Could not open archive for dataset [{}]", uuid, e);
124        DistributedAtomicLong distributedAtomicLong = counterCache.getUnchecked(CrawlerNodePaths.getCrawlInfoPath(
125          message.getDatasetUuid(),
126          CrawlerNodePaths.PAGES_FRAGMENTED_ERROR));
127        try {
128          distributedAtomicLong.trySet(1L);
129        } catch (Exception e1) {
130          LOG.warn("Failed to update counter for failed DwC-A fragment", e1);
131        }
132        return;
133      }
134
135      if (message.getDatasetType() == DatasetType.OCCURRENCE) {
136        handleOccurrenceCore(uuid, archive, message);
137
138      } else if (archive.getExtension(DwcTerm.Occurrence) != null) {
139        handleOccurrenceExtension(uuid, archive, message);
140
141      } else {
142        LOG.info("Ignoring DwC-A for dataset [{}] because it does not have Occurrence information.",
143                 message.getDatasetUuid());
144      }
145
146      setPagesFragmentedSuccessful(message);
147    }
148
149    private void handleOccurrenceCore(UUID uuid, Archive archive, DwcaMetasyncFinishedMessage message) {
150      LOG.info("Fragmenting DwC-A for dataset [{}] with occurrence core", uuid);
151
152      ClosableIterator<StarRecord> iterator = archive.iteratorRaw();
153      int counter = 0;
154      while (iterator.hasNext()) {
155        StarRecord record = iterator.next();
156        counter++;
157
158        byte[] serializedRecord = StarRecordSerializer.toJson(record);
159        try {
160          publisher.send(new OccurrenceFragmentedMessage(uuid,
161                                                         message.getAttempt(),
162                                                         serializedRecord,
163                                                         OccurrenceSchemaType.DWCA,
164                                                         EndpointType.DWC_ARCHIVE,
165                                                         message.getValidationReport()));
166        } catch (IOException e) {
167          LOG.error("Could not send message for dataset [{}]", uuid, e);
168        }
169        if (counter % COUNTER_UPDATE_FREQUENCY == 0) {
170          incrementCounters(message, COUNTER_UPDATE_FREQUENCY);
171        }
172      }
173
174      incrementCounters(message, counter % COUNTER_UPDATE_FREQUENCY);
175      LOG.info("Successfully extracted [{}] records out of DwC-A for dataset [{}]", counter, uuid);
176    }
177
178    private void handleOccurrenceExtension(UUID uuid, Archive archive, DwcaMetasyncFinishedMessage message) {
179      LOG.info("Fragmenting DwC-A for dataset [{}] with occurrence extension", uuid);
180      //TODO: implement extension processing, see http://dev.gbif.org/issues/browse/POR-1722
181      LOG.warn("Processing of occurrence extension records not implemented yet. Ignoring dataset [{}]", uuid);
182    }
183
184    private void setPagesFragmentedSuccessful(DwcaMetasyncFinishedMessage message) {
185      DistributedAtomicLong dal =
186        counterCache.getUnchecked(message.getDatasetUuid() + "/" + CrawlerNodePaths.PAGES_FRAGMENTED_SUCCESSFUL);
187      try {
188        dal.trySet(1L);
189      } catch (Exception e) {
190        LOG.error("Failed to update counter for successful DwC-A fragmenting", e);
191      }
192    }
193
194    private void incrementCounters(DwcaMetasyncFinishedMessage message, long count) {
195      DistributedAtomicLong fragmentsEmittedCounter =
196        counterCache.getUnchecked(message.getDatasetUuid() + "/" + CrawlerNodePaths.FRAGMENTS_EMITTED);
197      try {
198        if (count > 0) {
199          handleCounterResponse(message, fragmentsEmittedCounter.add(count));
200        }
201      } catch (Exception e) {
202        LOG.warn("Failed to update counter of pages crawled for [{}]", message.getDatasetUuid(), e);
203      }
204    }
205
206    private void handleCounterResponse(DatasetBasedMessage message, AtomicValue<Long> value) {
207      if (value.succeeded()) {
208        LOG.debug("Set counter of pages crawled for [{}] to [{}]", message.getDatasetUuid(), value.postValue());
209      } else {
210        LOG.warn("Failed to update counter of pages crawled for [{}]", message.getDatasetUuid(), value.postValue());
211      }
212    }
213
214  }
