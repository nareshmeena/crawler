041import static org.gbif.crawler.constants.CrawlerNodePaths.FINISHED_REASON;
042import static org.gbif.crawler.constants.CrawlerNodePaths.getCrawlInfoPath;
043
044/**
045 * Consumer of the crawler queue that runs the actual dwc archive download and emits a DwcaDownloadFinishedMessage
046 * when done.
047 */
048public class CrawlConsumer implements QueueConsumer<UUID> {
049
050  private static final Logger LOG = LoggerFactory.getLogger(CrawlConsumer.class);
051  private static final String DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSZ";
052  private static final ObjectMapper MAPPER = new ObjectMapper();
053
054  static {
055    MAPPER.registerModule(new GuavaModule());
056  }
057
058  private final File archiveRepository;
059  private final Counter startedDownloads = Metrics.newCounter(DownloaderService.class, "startedDownloads");
060  private final Counter failedDownloads = Metrics.newCounter(DownloaderService.class, "failedDownloads");
061  private final Counter notModified = Metrics.newCounter(DownloaderService.class, "notModified");
062  private final HttpUtil client = new HttpUtil(HttpUtil.newMultithreadedClient(10 * 60 * 1000, 20, 2));
063  private final CuratorFramework curator;
064  private final MessagePublisher publisher;
065  private final RetryPolicy retryPolicy = new ExponentialBackoffRetry(100, 10);
066
067  public CrawlConsumer(CuratorFramework curator, MessagePublisher publisher, File archiveRepository) {
068    this.curator = curator;
069    this.publisher = publisher;
070    this.archiveRepository = archiveRepository;
071    if (!archiveRepository.exists() && !archiveRepository.isDirectory()) {
072      throw new IllegalArgumentException("Archive repository needs to be an existing directory: "
073                                         + archiveRepository.getAbsolutePath());
074    }
075    if (!archiveRepository.canWrite()) {
076      throw new IllegalArgumentException("Archive repository directory not writable: "
077                                         + archiveRepository.getAbsolutePath());
078    }
079  }
080
081  @Override
082  public void stateChanged(CuratorFramework client, ConnectionState newState) {
083    LOG.info("State changed to [{}]", newState);
084  }
085
086  @Override
087  public void consumeMessage(UUID message) throws Exception {
088    LOG.debug("Got crawl job for UUID [{}]", message);
089    DwcaDownloadFinishedMessage finishedMessage;
090    try {
091      byte[] bytes = curator.getData().forPath(getCrawlInfoPath(message));
092      CrawlJob crawlJob = MAPPER.readValue(bytes, CrawlJob.class);
093
094      startCrawl(crawlJob.getDatasetKey());
095      MDC.put("datasetKey", message.toString());
096      MDC.put("attempt", String.valueOf(crawlJob.getAttempt()));
097      finishedMessage = doCrawl(crawlJob);
098    } catch (RuntimeException e) {
099      createOrUpdateZooKeeper(getCrawlInfoPath(message, FINISHED_REASON),
100                              FinishReason.ABORT.toString().getBytes(Charsets.UTF_8));
101      LOG.warn("Finished failed download attempt of DwC-A for dataset [{}]", message);
102      // TODO: Send a message on failure, see CRAWLER-72
103      return;
104    } finally {
105      MDC.remove("datasetKey");
106      MDC.remove("attempt");
107      updateDate(message, CrawlerNodePaths.FINISHED_CRAWLING);
108    }
109
110    // This will only be reached when the crawl was successful
111    createOrUpdateZooKeeper(getCrawlInfoPath(message, FINISHED_REASON),
112                            FinishReason.NORMAL.toString().getBytes(Charsets.UTF_8));
113    DistributedAtomicLong dal =
114      new DistributedAtomicLong(curator, getCrawlInfoPath(message, CrawlerNodePaths.PAGES_CRAWLED), retryPolicy);
115    dal.trySet(1L);
116
117    // If the archive wasn't modified we are done processing so we need to update ZooKeeper to reflect this
118    if (!finishedMessage.isModified()) {
119      dal = new DistributedAtomicLong(curator,
120                                      getCrawlInfoPath(message, CrawlerNodePaths.PAGES_FRAGMENTED_SUCCESSFUL),
121                                      retryPolicy);
122      dal.trySet(1L);
123    }
124    LOG.debug("Finished download attempt of DwC-A for dataset [{}]", message);
125  }
126
127  /**
128   * The started crawl is declared in zookeeper by putting the date into
129   * /crawls/UUID/startedCrawling
130   */
131  private void startCrawl(UUID datasetKey) {
132    updateDate(datasetKey, CrawlerNodePaths.STARTED_CRAWLING);
133  }
134
135  /**
136   * Updates a node in ZooKeeper saving the current date in time in there.
137   *
138   * @param datasetKey designates the first bit of the path to update
139   * @param path        the path to update within the dataset node
140   */
141  private void updateDate(UUID datasetKey, String path) {
142    String crawlPath = getCrawlInfoPath(datasetKey, path);
143    Date date = new Date();
144
145    SimpleDateFormat dateFormat = new SimpleDateFormat(DATE_FORMAT);
146    dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
147    byte[] data = dateFormat.format(date).getBytes(Charsets.UTF_8);
148    createOrUpdateZooKeeper(crawlPath, data);
149  }
150
151  private void createOrUpdateZooKeeper(String path, byte[] data) {
152    try {
153      Stat stat = curator.checkExists().forPath(path);
154      if (stat == null) {
155        curator.create().forPath(path, data);
156      } else {
157        curator.setData().forPath(path, data);
158      }
159    } catch (Exception e) {
160      LOG.warn("Exception while updating ZooKeeper", e);
161    }
162  }
163
164  private DwcaDownloadFinishedMessage doCrawl(CrawlJob crawlRequest) {
165    Preconditions.checkNotNull("crawlRequest required", crawlRequest);
166    UUID uuid = crawlRequest.getDatasetKey();
167
168    startedDownloads.inc();
169
170    // we keep the compressed file forever and use it to retrieve the last modified for conditional gets
171    File localFile = new File(archiveRepository, uuid + ".dwca");
172    // the decompressed archive folder is created by this consumer and then gets deleted at the very end of the processing
173    File archiveDir = new File(archiveRepository, uuid.toString());
174    if (archiveDir.exists()) {
175      // clean up any existing folder
176      LOG.debug("Deleting existing archive folder [{}]", archiveDir.getAbsolutePath());
177      FileUtils.deleteDirectoryRecursively(archiveDir);
178    }
179
180    boolean modified = false;
181    Date lastModified = null;
182
183    try {
184      StatusLine status = client.downloadIfModifiedSince(crawlRequest.getTargetUrl().toURL(), localFile);
185
186      if (status.getStatusCode() == HttpStatus.SC_NOT_MODIFIED) {
187        notModified.inc();
188        // use last modified from previous file
189        lastModified = new Date(localFile.lastModified());
190
191      } else if (HttpUtil.success(status)) {
192        modified = true;
193        lastModified = new Date();
194        try {
195          CompressionUtil.decompressFile(archiveDir, localFile, true);
196        } catch (CompressionUtil.UnsupportedCompressionType unsupportedCompressionType) {
197          LOG.debug("Could not uncompress archive for dataset [{}]", uuid, unsupportedCompressionType);
198          try {
199            ArchiveFactory.openArchive(localFile);
200            LOG.debug("Was able to read plain text file for dataset [{}]", uuid);
201            if (!archiveDir.mkdir()) {
202              LOG.error("Failed creating directory [{}], aborting dataset [{}]", archiveDir, uuid);
203              return null;
204            }
205            Files.copy(localFile, new File(archiveDir, Files.getNameWithoutExtension(localFile.getName()) + ".csv"));
206          } catch (UnsupportedArchiveException ignored) {
207            LOG.warn("Not a valid archive or plain text file for dataset [{}]", uuid);
208            failedDownloads.inc();
209            throw new IllegalStateException("Was able to download but received invalid archive for dataset "
210                                            + "[" + uuid + "]");
211          }
212        }
213      } else {
214        failedDownloads.inc();
215        throw new IllegalStateException("HTTP " + status.getStatusCode() +
216                                        ". Failed to download DwC-A for dataset " + uuid +
217                                        " from " + crawlRequest.getTargetUrl());
218      }
219
220    } catch (IOException e) {
221      failedDownloads.inc();
222      LOG.error("Failed to download dwc archive for dataset [{}] from [{}]",
223                crawlRequest.getDatasetKey(),
224                crawlRequest.getTargetUrl(),
225                e);
226      throw new RuntimeException(e);
227    }
228
229    LOG.info("Successfully downloaded DwC-A for dataset, modified [{}], lastModified [{}], uuid [{}]",
230             modified,
231             lastModified,
232             uuid);
233
234    // send download success message
235    try {
236      DwcaDownloadFinishedMessage message = new DwcaDownloadFinishedMessage(crawlRequest.getDatasetKey(),
237                                                                            crawlRequest.getTargetUrl(),
238                                                                            crawlRequest.getAttempt(),
239                                                                            lastModified,
240                                                                            modified);
241      publisher.send(message);
242      return message;
243    } catch (IOException e) {
244      LOG.error("Failed to send download finished message for crawl [{}]", crawlRequest.getDatasetKey(), e);
245    }
246    return null;
247  }
