001package org.gbif.crawler.dwca;
002
003import org.gbif.api.service.registry.DatasetService;
004import org.gbif.common.messaging.DefaultMessagePublisher;
005import org.gbif.common.messaging.MessageListener;
006import org.gbif.common.messaging.api.MessagePublisher;
007
008import java.io.IOException;
009import java.util.concurrent.Executors;
010
011import com.google.common.base.Preconditions;
012import com.google.common.util.concurrent.AbstractIdleService;
013
014/**
015 * Basic service that connects to RabbitMQ and binds a writable DatasetService for the GBIF registry.
016 * Implement {@link #bindListeners} to listen to any messages or configure anything else on service startup.
017 */
018public abstract class DwcaService extends AbstractIdleService {
019
020  protected final DwcaConfiguration config;
021  protected MessagePublisher publisher;
022  protected MessageListener listener;
023  protected DatasetService datasetService;
024
025  protected DwcaService(DwcaConfiguration config) {
026    this.config = Preconditions.checkNotNull(config, "Configuration required");
027  }
028
029  /**
030   * Does all the registry client bindings and messaging startup, binding listener via the abstract bindListeners()
031   * method.
032   */
033  @Override
034  protected void startUp() throws Exception {
035    publisher = new DefaultMessagePublisher(config.messaging.getConnectionParameters());
036    listener = new MessageListener(config.messaging.getConnectionParameters());
037    datasetService = config.registry.newRegistryInjector().getInstance(DatasetService.class);
038    bindListeners();
039  }
040
041  @Override
042  protected void shutDown() throws Exception {
043    publisher.close();
044    listener.close();
045  }
046
047  /**
048   * Implement this to bind any messaging listeners once after the messaging service has been started up.
049   */
050  protected abstract void bindListeners() throws IOException;
051
052}
