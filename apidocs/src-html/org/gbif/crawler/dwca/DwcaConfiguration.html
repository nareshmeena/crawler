001package org.gbif.crawler.dwca;
002
003import org.gbif.cli.IgnoreProperty;
004import org.gbif.common.messaging.config.MessagingConfiguration;
005import org.gbif.crawler.common.RegistryConfiguration;
006
007import java.io.File;
008
009import javax.validation.Valid;
010import javax.validation.constraints.Min;
011import javax.validation.constraints.NotNull;
012
013import com.beust.jcommander.Parameter;
014import com.beust.jcommander.ParametersDelegate;
015import com.google.common.base.Objects;
016
017public class DwcaConfiguration {
018
019  @ParametersDelegate
020  @Valid
021  @NotNull
022  @IgnoreProperty
023  public MessagingConfiguration messaging = new MessagingConfiguration();
024
025  @ParametersDelegate
026  @Valid
027  @NotNull
028  public RegistryConfiguration registry = new RegistryConfiguration();
029
030  @Parameter(names = "--pool-size")
031  @Min(1)
032  public int poolSize = 10;
033
034  @Parameter(names = "--queue-name")
035  @NotNull
036  public String queueName;
037
038  @Parameter(names = "--archive-repository")
039  @NotNull
040  public File archiveRepository;
041
042
043  @Override
044  public String toString() {
045    return Objects.toStringHelper(this)
046      .add("registry", registry)
047      .add("messaging", messaging)
048      .add("queueName", queueName)
049      .add("poolSize", poolSize)
050      .add("archiveRepository", archiveRepository)
051      .toString();
052  }
053
054}
