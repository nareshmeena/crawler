001package org.gbif.crawler;
002
003import com.google.common.base.Optional;
004
005/**
006 * Used to listen for crawl events.
007 *
008 * Any exceptions thrown are logged but ignored for further processing.
009 *
010 * @param <CTX>  the crawl context this listener supports
011 * @param <REQ>  the type of request this listener supports
012 * @param <RESP> the type of response (from response handlers) this listener supports
013 */
014public interface CrawlListener<CTX extends CrawlContext, REQ, RESP> {
015
016  /**
017   * Reports that a Crawler will now actively begin to crawl.
018   */
019  void startCrawl();
020
021  /**
022   * Reports that we are making progress. This method may be called multiple times with the same context.
023   *
024   * @param context current context the crawler is working on
025   */
026  void progress(CTX context);
027
028  /**
029   * Reports the verbatim request as it will be sent to the host.
030   *
031   * @param req   the request
032   * @param retry we might try a request multiple times if there was a recoverable error, this is the retry number
033   *              beginning at 1
034   */
035  void request(REQ req, int retry);
036
037  /**
038   * Reports a response we received.
039   *
040   * @param response     we received
041   * @param retry        which retry is this
042   * @param duration     how long did it take to receive the response and process it
043   * @param recordCount  how many records does the response claim to contain
044   * @param endOfRecords are there more pages coming
045   */
046  void response(RESP response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords);
047
048  /**
049   * Reports that we are done crawling a dataset and there were no errors that lead us to abort it prematurely.
050   */
051  void finishCrawlNormally();
052
053  /**
054   * The user requested us to abort the crawl.
055   */
056  void finishCrawlOnUserRequest();
057
058  /**
059   * Reports that the crawl is finished due to an abnormal reason (e.g. retries exhausted)
060   */
061  void finishCrawlAbnormally();
062
063  void error(Throwable e);
064
065  void error(String msg);
066
067}
