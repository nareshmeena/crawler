001package org.gbif.crawler.ws.client;
002
003import org.gbif.api.model.crawler.DatasetProcessStatus;
004import org.gbif.api.service.crawler.DatasetProcessService;
005import org.gbif.ws.client.BaseWsClient;
006
007import java.util.List;
008import java.util.Set;
009import java.util.UUID;
010import javax.ws.rs.core.MediaType;
011
012import com.google.inject.Inject;
013import com.sun.jersey.api.client.GenericType;
014import com.sun.jersey.api.client.WebResource;
015
016/**
017 * Client-side implementation to the crawler service.
018 */
019public class CrawlerWsClient extends BaseWsClient implements DatasetProcessService {
020
021  private static final GenericType<Set<DatasetProcessStatus>> CRAWL_SET_TYPE =
022    new GenericType<Set<DatasetProcessStatus>>() {};
023
024  private static final GenericType<List<DatasetProcessStatus>> CRAWL_LIST_TYPE =
025    new GenericType<List<DatasetProcessStatus>>() {};
026
027  private static final String DATASET_PATH = "dataset/process";
028
029  @Inject
030  public CrawlerWsClient(WebResource resource) {
031    super(resource.path(DATASET_PATH));
032  }
033
034
035  @Override
036  public DatasetProcessStatus getDatasetProcessStatus(UUID datasetKey) {
037    return resource.path("detail")
038      .path(datasetKey.toString())
039      .type(MediaType.APPLICATION_JSON)
040      .get(DatasetProcessStatus.class);
041  }
042
043  @Override
044  public Set<DatasetProcessStatus> getRunningDatasetProcesses() {
045    return resource.path("running").type(MediaType.APPLICATION_JSON).get(CRAWL_SET_TYPE);
046  }
047
048  @Override
049  public List<DatasetProcessStatus> getPendingXmlDatasetProcesses() {
050    return resource.path("xml/pending").type(MediaType.APPLICATION_JSON).get(CRAWL_LIST_TYPE);
051  }
052
053  @Override
054  public List<DatasetProcessStatus> getPendingDwcaDatasetProcesses() {
055    return resource.path("dwca/pending").type(MediaType.APPLICATION_JSON).get(CRAWL_LIST_TYPE);
056  }
057
058}
