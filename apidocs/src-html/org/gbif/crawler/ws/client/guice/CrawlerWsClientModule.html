001package org.gbif.crawler.ws.client.guice;
002
003import org.gbif.api.service.crawler.DatasetProcessService;
004import org.gbif.crawler.ws.client.CrawlerWsClient;
005import org.gbif.ws.client.guice.GbifWsClientModule;
006
007import java.util.Properties;
008
009import com.google.inject.Provides;
010import com.google.inject.Singleton;
011import com.google.inject.name.Named;
012import com.sun.jersey.api.client.Client;
013import com.sun.jersey.api.client.WebResource;
014
015/**
016 * Guice module that exposes the Crawler Service.
017 */
018public class CrawlerWsClientModule extends GbifWsClientModule {
019
020  public CrawlerWsClientModule(Properties properties) {
021    super(properties, CrawlerWsClient.class.getPackage());
022  }
023
024  @Override
025  protected void configureClient() {
026    bind(DatasetProcessService.class).to(CrawlerWsClient.class);
027    expose(DatasetProcessService.class);
028  }
029
030  /**
031   * @return the web resource to the webapp
032   */
033  @Provides
034  @Singleton
035  public WebResource provideWebResource(Client client, @Named("crawler.ws.url") String url) {
036    return client.resource(url);
037  }
038}
