001package org.gbif.crawler.ws.guice;
002
003import org.gbif.ws.server.guice.GbifServletListener;
004
005import java.util.List;
006import java.util.Properties;
007import java.util.concurrent.Executor;
008import java.util.concurrent.ExecutorService;
009import java.util.concurrent.TimeUnit;
010import javax.servlet.ServletContextEvent;
011
012import com.google.common.collect.Lists;
013import com.google.inject.Module;
014import org.apache.curator.framework.CuratorFramework;
015
016/**
017 * The Crawler-Coordinator WS production module.
018 */
019public class CrawlWsServletListener extends GbifServletListener {
020
021  public static final String APPLICATION_PROPERTIES = "crawler.properties";
022
023  public CrawlWsServletListener() {
024    super(APPLICATION_PROPERTIES, "org.gbif.crawler.ws", false);
025  }
026
027  @Override
028  protected List<Module> getModules(Properties properties) {
029    List<Module> modules = Lists.newArrayList();
030    modules.add(new CrawlerModule(properties));
031    return modules;
032  }
033
034  @Override
035  public void contextDestroyed(ServletContextEvent servletContextEvent) {
036    super.contextDestroyed(servletContextEvent);
037    ExecutorService es = (ExecutorService) getInjector().getInstance(Executor.class);
038    es.shutdown();
039    try {
040      es.awaitTermination(20, TimeUnit.SECONDS);
041    } catch (InterruptedException e) {
042    }
043
044    CuratorFramework curatorFramework = getInjector().getInstance(CuratorFramework.class);
045    curatorFramework.close();
046  }
047}
