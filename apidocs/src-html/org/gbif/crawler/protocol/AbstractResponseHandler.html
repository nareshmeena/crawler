001package org.gbif.crawler.protocol;

003import org.gbif.crawler.ResponseHandler;
004import org.gbif.crawler.exception.FatalCrawlException;
005import org.gbif.crawler.exception.ProtocolException;
006import org.gbif.crawler.exception.TransportException;
007
008import java.io.ByteArrayInputStream;
009import java.io.IOException;
010import java.util.List;
011import javax.annotation.Nullable;
012import javax.annotation.concurrent.NotThreadSafe;
013import javax.xml.stream.XMLInputFactory;
014import javax.xml.stream.XMLOutputFactory;
015import javax.xml.stream.XMLStreamConstants;
016import javax.xml.stream.XMLStreamException;
017import javax.xml.stream.XMLStreamReader;
018import javax.xml.stream.XMLStreamWriter;
019
020import com.google.common.base.Optional;
021import com.google.common.hash.Funnels;
022import com.google.common.hash.Hasher;
023import com.google.common.hash.Hashing;
024import com.google.common.io.ByteStreams;
025import com.google.common.primitives.Bytes;
026import org.apache.http.HttpResponse;
027import org.codehaus.stax2.XMLInputFactory2;
028import org.codehaus.stax2.XMLOutputFactory2;
029import org.codehaus.stax2.XMLStreamReader2;
030import org.codehaus.stax2.XMLStreamWriter2;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;

034import static com.google.common.base.Preconditions.checkNotNull;
035import static com.google.common.base.Preconditions.checkState;

037/**
038 * A base class for response handlers.
039 * <p/>
040 * This class takes care of converting a {@link HttpResponse} into a byte[] which is then converted to a List of bytes.
041 * You can get the underlying array back by using Guava: {@link Bytes#toArray(java.util.Collection)}}.
042 * <p/>
043 * A hash code is automatically calculated for the content of the response. The various getters throw an {@link
044 * IllegalStateException} if the last response was not valid so check {@link #isValidState} before calling any of
045 * those. An alternative indicator are the exceptions thrown by {@link #handleResponse(HttpResponse)}. If any is thrown
046 * during processing the Response handler is in an illegal state until a successful response has been handled.
047 * <p/>
048 * This class is not thread-safe.
049 */
050@NotThreadSafe
051public abstract class AbstractResponseHandler implements ResponseHandler<HttpResponse, List<Byte>> {
052
053  private static final Logger LOG = LoggerFactory.getLogger(AbstractResponseHandler.class);
054
055  private static final XMLInputFactory2 XIF = (XMLInputFactory2) XMLInputFactory.newInstance();
056  private static final XMLOutputFactory2 XOF = (XMLOutputFactory2) XMLOutputFactory.newInstance();
057
058  private Optional<Integer> recordCount = Optional.absent();
059  private Optional<Boolean> endOfRecords = Optional.absent();
060  private Optional<Long> contentHash;
061
062  private boolean validState = false;
063
064  /**
065   * This needs to be implemented by subclasses to do actual processing on the XML data. This method is called for
066   * every
067   * single stream event (which includes whitespace etc.) so implementations need to check the event type and make sure
068   * to work on the correct elements.
069   *
070   * @param reader to use
071   *
072   * @throws XMLStreamException if there was an error with the XML parsing (e.g. invalid XML)
073   * @throws ProtocolException  if there is a problem with the (well-formed) XML response (e.g. missing attribute)
074   */
075  protected abstract void process(XMLStreamReader2 reader)
076    throws XMLStreamException, FatalCrawlException, ProtocolException;
077
078  /**
079   * Needs to be implemented by subclasses to indicate to the parent class what the XML element is which delimits the
080   * actual content of a result. Everything inside this element including the element itself is hashed.
081   *
082   * @return with the content element name
083   */
084  protected abstract boolean shouldHash(XMLStreamReader2 reader);
085
086  @Override
087  public boolean isValidState() {
088    return validState;
089  }
090
091  private void initializeInternal() {
092    recordCount = Optional.absent();
093    endOfRecords = Optional.absent();
094    contentHash = Optional.absent();
095    validState = false;
096  }
097
098  /**
099   * To be implemented to initialize internal state before every new response is being handled.
100   */
101  protected abstract void initialize();
102
103  /**
104   * This is an implementation that processes a {@link HttpResponse} and produces a List of Bytes.
105   *
106   * @param response to be handled
107   *
108   * @return the byte array corresponding to the response we received. Wrapped in a List of Bytes because this is
109   *         extending a typed interface. This list is guaranteed to be convertable to a primitive {@code byte[]} using
110   *         Guava's {@link Bytes#toArray(java.util.Collection)}}.
111   *
112   * @throws ProtocolException  if there are any problems with the content received
113   * @throws TransportException if there are any problems receiving the response
114   */
115  @Override
116  public final List<Byte> handleResponse(HttpResponse response)
117    throws FatalCrawlException, ProtocolException, TransportException {
118    checkNotNull(response);
119
120    initializeInternal();
121
122    initialize();
123
124    // First of all we convert the response into a byte array
125    byte[] bytes;
126    try {
127      bytes = ByteStreams.toByteArray(response.getEntity().getContent());
128    } catch (IOException e) {
129      throw new TransportException(e);
130    }
131
132    Hasher hasher = Hashing.md5().newHasher();
133
134    // The reader is used to parse the XML response and the writer is used to hash parts of the document
135    XMLStreamReader2 reader = null;
136    XMLStreamWriter2 writer = null;
137    boolean didHash = false;
138    try {
139      reader = (XMLStreamReader2) XIF.createXMLStreamReader(new ByteArrayInputStream(bytes));
140      writer = (XMLStreamWriter2) XOF.createXMLStreamWriter(Funnels.asOutputStream(hasher));
141      writer.writeStartElement("wrapper");
142
143      while (reader.next() != XMLStreamConstants.END_DOCUMENT) {
144        // Hand off processing to subclasses
145        process(reader);
146
147        if (shouldHash(reader)) {
148          didHash = true;
149          writer.copyEventFromReader(reader, true);
150        }
151      }
152
153      writer.writeEndElement();
154
155    } catch (XMLStreamException e) {
156      throw new ProtocolException(e);
157    } finally {
158      closeQuietly(reader);
159      closeQuietly(writer);
160    }
161
162    if (didHash) {
163      contentHash = Optional.of(hasher.hash().asLong());
164    }
165
166    validState = true;
167
168    return Bytes.asList(bytes);
169  }
170
171  @Override
172  public final Optional<Integer> getRecordCount() {
173    checkState(validState);
174
175    return recordCount;
176  }
177
178  protected final void setRecordCount(Optional<Integer> recordCount) {
179    this.recordCount = checkNotNull(recordCount);
180  }
181
182  protected final void setEndOfRecords(Optional<Boolean> endOfRecords) {
183    this.endOfRecords = checkNotNull(endOfRecords);
184  }
185
186  @Override
187  public final Optional<Boolean> isEndOfRecords() {
188    checkState(validState, "Last response was not valid");
189
190    return endOfRecords;
191  }
192
193  @Override
194  public final Optional<Long> getContentHash() {
195    checkState(validState, "Last response was not valid");
196
197    return contentHash;
198  }
199
200  /**
201   * Closes a XMLStreamReader quietly. Exceptions are being swallowed and logged.
202   *
203   * @param reader to close
204   */
205  private static void closeQuietly(@Nullable XMLStreamReader reader) {
206    if (reader == null) {
207      return;
208    }
209    try {
210      reader.close();
211    } catch (XMLStreamException e) {
212      LOG.warn("Error closing XML reader", e);
213    }
214  }
215
216  /**
217   * Closes a XMLStreamWriter quietly. Exceptions are being swallowed and logged.
218   *
219   * @param writer to close
220   */
221  private static void closeQuietly(@Nullable XMLStreamWriter writer) {
222    if (writer == null) {
223      return;
224    }
225    try {
226      writer.close();
227    } catch (XMLStreamException e) {
228      LOG.debug("Error closing XML writer", e);
229    }
230  }
231
232  protected static boolean isElement(XMLStreamReader reader, String name, boolean start) {
233    return reader.getEventType() == (start ? XMLStreamConstants.START_ELEMENT : XMLStreamConstants.END_ELEMENT)
234           && reader.getLocalName().equals(name);
235  }
