001package org.gbif.crawler.protocol.biocase;
002
003import org.gbif.crawler.exception.ProtocolException;
004import org.gbif.crawler.protocol.AbstractResponseHandler;
005
006import javax.annotation.concurrent.NotThreadSafe;
007import javax.xml.stream.XMLStreamConstants;
008import javax.xml.stream.XMLStreamException;
009
010import com.google.common.base.Optional;
011import org.codehaus.stax2.XMLStreamReader2;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * This class handles a BioCASe response.
017 * <p/>
018 * Every BioCASe response has a few pieces of optional diagnostic information:
019 * <ul>
020 * <li>{@code recordCount} is the number of records that are being returned in this response</li>
021 * <li>{@code recordDropped} is the number of dropped records for this response, this number is <em>not</em> included
022 * in the {@code recordCount}s.</li>
023 * <li>{@code recordStart} is the {@code offset} that was used for this request</li>
024 * <li>{@code totalSearchHits} is the total number of results for this query. It usually doesn't return an exact count
025 * as that is expensive to calculate. If {@code totalSearchHits} is greater than {@code recordCount} + {@code
026 * recordDropped} + {@code recordStart} there should be more results</li>
027 * </ul>
028 * <p/>
029 * As an example. Sending a request with {@code start}=0 and {@code limit}=100 could return a response like this:
030 * <pre>
031 *   {@code <biocase:content recordCount='98' recordDropped='2' recordStart='0' totalSearchHits='101'>}
032 * </pre>
033 * <p/>
034 * Which would mean that the response includes 98 records and 2 had to be dropped on the way out. Because
035 * totalSearchHits is greater than recordCount + recordDropped + recordStart there are more pages and the next request
036 * should be with {@code start}=100 and {@code limit}=100 which could return this:
037 * <pre>
038 *   {@code <biocase:content recordCount='80' recordDropped='5' recordStart='100' totalSearchHits='185'>}
039 * </pre>
040 * <p/>
041 * This class is not thread-safe.
042 */
043@NotThreadSafe
044public class BiocaseResponseHandler extends AbstractResponseHandler {
045
046  private static final Logger LOG = LoggerFactory.getLogger(BiocaseResponseHandler.class);
047
048  private static final String CONTENT_ELEMENT = "content";
049
050  private boolean insideContent;
051  private boolean seenContent;
052
053  @Override
054  protected boolean shouldHash(XMLStreamReader2 reader) {
055    if (isElement(reader, CONTENT_ELEMENT, true)) {
056      insideContent = true;
057      return false;
058    }
059
060    if (insideContent && reader.getEventType() == XMLStreamConstants.START_ELEMENT) {
061      seenContent = true;
062    }
063
064    // We found the end element which shouldn't be hashed either
065    if (isElement(reader, CONTENT_ELEMENT, false)) {
066      insideContent = false;
067    }
068
069    return insideContent && seenContent;
070  }
071
072  @Override
073  protected void initialize() {
074    insideContent = false;
075    seenContent = false;
076  }
077
078  @Override
079  protected void process(XMLStreamReader2 reader) throws XMLStreamException, ProtocolException {
080    if (reader.getEventType() == XMLStreamConstants.START_ELEMENT && reader.getLocalName().equals(CONTENT_ELEMENT)) {
081      Optional<Integer> recordStart = getAttribute(reader, "recordStart");
082      Optional<Integer> recordCount = getAttribute(reader, "recordCount");
083      Optional<Integer> recordDropped = getAttribute(reader, "recordDropped");
084      Optional<Integer> totalHits = getAttribute(reader, "totalSearchHits");
085
086      setRecordCount(recordCount);
087      if (recordCount.isPresent() && recordStart.isPresent() && recordDropped.isPresent() && totalHits.isPresent()) {
088        setEndOfRecords(Optional.of(recordCount.get() + recordStart.get() + recordDropped.get() >= totalHits.get()));
089      }
090
091      LOG.debug("Found "
092                + CONTENT_ELEMENT
093                + " element. [{}] records, [{}] records dropped, [{}] record start, [{}] total search hits",
094                new Object[] {recordCount, recordDropped, recordStart, totalHits});
095    }
096  }
097
098  /**
099   * This tries to read an attribute of an element as an integer and returns it if found, {@link Optional#absent()}
100   * otherwise.
101   *
102   * @param reader    to read the attribute from, needs to be set to an element
103   * @param attribute to read from the element
104   *
105   * @return an Optional with the integer value of the attribute {@code absent} otherwise
106   */
107  private Optional<Integer> getAttribute(XMLStreamReader2 reader, String attribute) throws XMLStreamException {
108    int index = reader.getAttributeIndex(null, attribute);
109    if (index == -1) {
110      return Optional.absent();
111    }
112
113    return Optional.of(reader.getAttributeAsInt(index));
114  }
115
116}
