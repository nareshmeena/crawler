001package org.gbif.crawler.protocol.biocase;
002
003import org.gbif.crawler.protocol.AbstractScientificNameRangeRequestHandler;
004
005import javax.annotation.concurrent.NotThreadSafe;
006
007import com.google.common.collect.ImmutableMap;
008
009/**
010 * This is a request handler for BioCASe queries using a range of scientific names.
011 * <p/>
012 * It sends search requests using BioCASe protocol version 1.3 and supports ABCD in version 1.2 and 2.06 and requests
013 * 1000 results per page by default.
014 * <p/>
015 * This class is not thread-safe.
016 */
017@NotThreadSafe
018public class BiocaseScientificNameRangeRequestHandler extends AbstractScientificNameRangeRequestHandler {
019
020  private static final String TEMPLATE_LOCATION = "template/biocase/scientificNameRangeSearch.ftl";
021  private static final String REQUEST_PARAM_KEY = "request";
022
023  private static final int MAX_RESULTS = 1000;
024
025  private static final String TITLE_KEY = "titlePath";
026  private static final String SUBJECT_KEY = "subjectPath";
027
028  private static final ImmutableMap<String, ImmutableMap<String, String>> SCHEMA_INFOS;
029
030  static {
031    ImmutableMap<String, String> abcd12 = ImmutableMap.<String, String>builder()
032      .put(TITLE_KEY, "/DataSets/DataSet/OriginalSource/SourceName")
033      .put(SUBJECT_KEY,
034        "/DataSets/DataSet/Units/Unit/Identifications/Identification/TaxonIdentified/NameAuthorYearString")
035      .build();
036
037    ImmutableMap<String, String> abcd206 = ImmutableMap.<String, String>builder()
038      .put(TITLE_KEY, "/DataSets/DataSet/Metadata/Description/Representation/Title")
039      .put(SUBJECT_KEY,
040        "/DataSets/DataSet/Units/Unit/Identifications/Identification/Result/TaxonIdentified/ScientificName/FullScientificNameString")
041      .build();
042
043    SCHEMA_INFOS = ImmutableMap.<String, ImmutableMap<String, String>>builder()
044      .put("http://www.tdwg.org/schemas/abcd/1.2", abcd12)
045      .put("http://www.tdwg.org/schemas/abcd/2.06", abcd206)
046      .build();
047  }
048
049  private final ImmutableMap<String, Object> defaultContext;
050
051  /**
052   * Initializes this request handler.
053   */
054  public BiocaseScientificNameRangeRequestHandler(BiocaseCrawlConfiguration configuration) {
055    super(configuration.getUrl(), TEMPLATE_LOCATION, REQUEST_PARAM_KEY);
056
057    defaultContext = ImmutableMap.<String, Object>builder()
058      .put("limit", Integer.toString(MAX_RESULTS))
059      .put("contentNamespace", configuration.getContentNamespace())
060      .put("datasetTitle", configuration.getDatasetTitle())
061      .put(TITLE_KEY, SCHEMA_INFOS.get(configuration.getContentNamespace()).get(TITLE_KEY))
062      .put(SUBJECT_KEY, SCHEMA_INFOS.get(configuration.getContentNamespace()).get(SUBJECT_KEY))
063      .build();
064  }
065
066  @Override
067  protected ImmutableMap<String, Object> getDefaultContext() {
068    return defaultContext;
069  }
070
071  @Override
072  public int getLimit() {
073    return MAX_RESULTS;
074  }
075}
