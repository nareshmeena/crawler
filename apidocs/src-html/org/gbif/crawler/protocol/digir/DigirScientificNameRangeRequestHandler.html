001package org.gbif.crawler.protocol.digir;
002
003import org.gbif.crawler.protocol.AbstractScientificNameRangeRequestHandler;
004
005import javax.annotation.concurrent.NotThreadSafe;
006
007import com.google.common.collect.ImmutableMap;
008
009/**
010 * This is a request handler for DiGIR queries using a range of scientific names.
011 * <p/>
012 * It supports MANIS and normal DiGIR endpoints and requests 1000 results per page by default.
013 * <p/>
014 * This class is not thread-safe.
015 */
016@NotThreadSafe
017public class DigirScientificNameRangeRequestHandler extends AbstractScientificNameRangeRequestHandler {
018
019  private static final String TEMPLATE_LOCATION = "template/digir/scientificNameRangeSearch.ftl";
020  private static final String REQUEST_PARAM_KEY = "request";
021
022  private static final int MAX_RESULTS = 1000;
023
024  private static final String SCHEMA_LOCATION_KEY = "schemaLocation";
025  private static final String RECORD_SCHEMA_LOCATION_KEY = "recordSchemaLocation";
026
027  private static final ImmutableMap<Boolean, ImmutableMap<String, String>> SCHEMA_INFOS;
028
029  static {
030    ImmutableMap<String, String> normalDigir = ImmutableMap.<String, String>builder()
031      .put(SCHEMA_LOCATION_KEY, "http://digir.sourceforge.net/schema/conceptual/darwin/2003/1.0/darwin2.xsd")
032      .put(RECORD_SCHEMA_LOCATION_KEY,
033        "http://digir.sourceforge.net/schema/conceptual/darwin/full/2003/1.0/darwin2full.xsd")
034      .build();
035
036    ImmutableMap<String, String> manisDigir = ImmutableMap.<String, String>builder()
037      .put(SCHEMA_LOCATION_KEY, "http://bnhm.berkeley.edu/manis/DwC/darwin2jrw030315.xsd")
038      .put(RECORD_SCHEMA_LOCATION_KEY, "http://bnhm.berkeley.edu/manis/DwC/darwin2resultfull.xsd")
039      .build();
040
041    SCHEMA_INFOS = ImmutableMap.<Boolean, ImmutableMap<String, String>>builder()
042      .put(true, manisDigir)
043      .put(false, normalDigir)
044      .build();
045  }
046
047  private final ImmutableMap<String, Object> defaultContext;
048
049  /**
050   * Initializes this request handler.
051   */
052  public DigirScientificNameRangeRequestHandler(DigirCrawlConfiguration configuration) {
053    super(configuration.getUrl(), TEMPLATE_LOCATION, REQUEST_PARAM_KEY);
054
055    defaultContext = ImmutableMap.<String, Object>builder()
056      .put("resource", configuration.getResourceCode())
057      .put("destination", configuration.getUrl().toString())
058      .put("maxResults", Integer.toString(MAX_RESULTS))
059      .put(SCHEMA_LOCATION_KEY, SCHEMA_INFOS.get(configuration.isManis()).get(SCHEMA_LOCATION_KEY))
060      .put(RECORD_SCHEMA_LOCATION_KEY, SCHEMA_INFOS.get(configuration.isManis()).get(RECORD_SCHEMA_LOCATION_KEY))
061      .build();
062  }
063
064  @Override
065  protected ImmutableMap<String, Object> getDefaultContext() {
066    return defaultContext;
067  }
068
069  @Override
070  public int getLimit() {
071    return MAX_RESULTS;
072  }
073}
