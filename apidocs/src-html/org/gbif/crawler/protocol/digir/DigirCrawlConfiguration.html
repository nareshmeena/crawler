001package org.gbif.crawler.protocol.digir;
002
003import org.gbif.crawler.CrawlConfiguration;
004
005import java.net.URI;
006import java.util.UUID;
007import javax.annotation.concurrent.ThreadSafe;
008
009import com.google.common.base.Objects;
010
011import static com.google.common.base.Preconditions.checkArgument;
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * Holds information necessary to crawl a DiGIR resource:
016 * <p/>
017 * <ul>
018 * <li>target URL</li>
019 * <li>resource code</li>
020 * <li>a flag indicating if this is a MANIS endpoint</li>
021 * </ul>
022 * <p/>
023 * This object is immutable.
024 */
025@ThreadSafe
026public class DigirCrawlConfiguration extends CrawlConfiguration {
027
028  private final String resourceCode;
029
030  private final boolean manis;
031
032  public DigirCrawlConfiguration(UUID datasetKey, int attempt, URI url, String resourceCode, boolean manis) {
033    super(datasetKey, url, attempt);
034
035    this.resourceCode = checkNotNull(resourceCode, "resourceCode can't be null");
036    checkArgument(!resourceCode.isEmpty(), "resourceCode can't be empty");
037
038    this.manis = manis;
039  }
040
041  public String getResourceCode() {
042    return resourceCode;
043  }
044
045  public boolean isManis() {
046    return manis;
047  }
048
049  @Override
050  public boolean equals(Object obj) {
051    if (obj == this) {
052      return true;
053    }
054    if (!(obj instanceof DigirCrawlConfiguration)) {
055      return false;
056    }
057
058    final DigirCrawlConfiguration other = (DigirCrawlConfiguration) obj;
059    return super.equals(other)
060           && Objects.equal(this.resourceCode, other.resourceCode)
061           && Objects.equal(this.manis, other.manis);
062  }
063
064  @Override
065  public int hashCode() {
066    return Objects.hashCode(super.hashCode(), resourceCode, manis);
067  }
068
069  @Override
070  public String toString() {
071    return toStringHelper().add("resourceCode", resourceCode).add("manis", manis).toString();
072  }
073
074}
