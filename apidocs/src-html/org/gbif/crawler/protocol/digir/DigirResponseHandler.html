001package org.gbif.crawler.protocol.digir;
002
003import org.gbif.crawler.exception.FatalCrawlException;
004import org.gbif.crawler.protocol.AbstractResponseHandler;
005
006import javax.annotation.concurrent.NotThreadSafe;
007import javax.xml.stream.XMLStreamConstants;
008import javax.xml.stream.XMLStreamException;
009
010import com.google.common.base.Optional;
011import org.codehaus.stax2.XMLStreamReader2;
012
013/**
014 * This class is not thread-safe.
015 * <p/>
016 * The latest <a href="http://digir.sourceforge.net/schema/protocol/2003/1.0/digir.xsd">DiGIR XSD</a> does not require
017 * any content or diagnostics elements.
018 * <p/>
019 * So with DiGIR it could be that we successfully parse a page and are in a valid state but don't know if we're at the
020 * end of the records or how many we got.
021 */
022@NotThreadSafe
023public class DigirResponseHandler extends AbstractResponseHandler {
024
025  private static final String CONTENT_ELEMENT = "content";
026  private static final String INFO_ELEMENT = "diagnostic";
027
028  private static final String DIAGNOSITC_ELEMENT_ATTRIBUTE_NAME = "code";
029
030  private static final String RESOURCE_NOT_FOUND = "RESOURCE_NOT_FOUND";
031  private static final String RECORD_COUNT = "RECORD_COUNT";
032  private static final String END_OF_RECORDS = "END_OF_RECORDS";
033
034  private boolean insideContent;
035  private boolean seenContent;
036
037  @Override
038  protected boolean shouldHash(XMLStreamReader2 reader) {
039    if (isElement(reader, CONTENT_ELEMENT, true)) {
040      insideContent = true;
041      return false;
042    }
043
044    if (insideContent && reader.getEventType() == XMLStreamConstants.START_ELEMENT) {
045      seenContent = true;
046    }
047
048    // We found the end element which shouldn't be hashed either
049    if (isElement(reader, CONTENT_ELEMENT, false)) {
050      insideContent = false;
051    }
052
053    return insideContent && seenContent;
054  }
055
056  @Override
057  protected void initialize() {
058    insideContent = false;
059    seenContent = false;
060  }
061
062  @Override
063  protected void process(XMLStreamReader2 reader) throws XMLStreamException, FatalCrawlException {
064    if (isElement(reader, INFO_ELEMENT, true)) {
065      if (reader.getAttributeValue(null, DIAGNOSITC_ELEMENT_ATTRIBUTE_NAME).equals(RECORD_COUNT)) {
066        setRecordCount(Optional.of(reader.getElementAsInt()));
067      } else if (reader.getAttributeValue(null, DIAGNOSITC_ELEMENT_ATTRIBUTE_NAME).equals(END_OF_RECORDS)) {
068        setEndOfRecords(Optional.of(reader.getElementAsBoolean()));
069      } else if (reader.getAttributeValue(null, DIAGNOSITC_ELEMENT_ATTRIBUTE_NAME).equals(RESOURCE_NOT_FOUND)) {
070        throw new FatalCrawlException("Requested resource could not be found");
071      }
072    }
073  }
074
075}
