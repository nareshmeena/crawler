001package org.gbif.crawler.protocol.tapir;
002
003import org.gbif.crawler.CrawlConfiguration;
004
005import java.net.URI;
006import java.util.UUID;
007import javax.annotation.concurrent.ThreadSafe;
008
009import com.google.common.base.Objects;
010import com.google.common.collect.ImmutableList;
011
012import static com.google.common.base.Preconditions.checkArgument;
013import static com.google.common.base.Preconditions.checkNotNull;
014
015/**
016 * Holds information necessary to crawl a TAPIR resource:
017 * <p/>
018 * <ul>
019 * <li>target URL</li>
020 * <li>content namespace, a list of supported namespaces is available in the {@link #VALID_SCHEMAS} field</li>
021 * </ul>
022 * <p/>
023 * This object is immutable.
024 */
025@ThreadSafe
026public class TapirCrawlConfiguration extends CrawlConfiguration {
027
028  public static final ImmutableList<String> VALID_SCHEMAS = ImmutableList.of(
029    "http://rs.tdwg.org/dwc/dwcore/",
030    "http://rs.tdwg.org/dwc/geospatial/",
031    "http://rs.tdwg.org/dwc/curatorial/",
032    "http://rs.tdwg.org/dwc/terms/",
033    "http://digir.net/schema/conceptual/darwin/2003/1.0",
034    "http://www.tdwg.org/schemas/abcd/1.2",
035    "http://www.tdwg.org/schemas/abcd/2.05",
036    "http://www.tdwg.org/schemas/abcd/2.06"
037  );
038
039  private final String contentNamespace;
040
041  public TapirCrawlConfiguration(UUID datasetKey, int attempt, URI url, String contentNamespace) {
042    super(datasetKey, url, attempt);
043
044    this.contentNamespace = checkNotNull(contentNamespace, "contentNamespace can't be null");
045
046    checkArgument(VALID_SCHEMAS.contains(contentNamespace),
047      "contentNamespace not supported: [" + contentNamespace + "]");
048  }
049
050  public String getContentNamespace() {
051    return contentNamespace;
052  }
053
054  @Override
055  public boolean equals(Object obj) {
056    if (this == obj) {
057      return true;
058    }
059    if (!(obj instanceof TapirCrawlConfiguration)) {
060      return false;
061    }
062
063    final TapirCrawlConfiguration other = (TapirCrawlConfiguration) obj;
064    return super.equals(other)
065           && Objects.equal(this.contentNamespace, other.contentNamespace);
066  }
067
068  @Override
069  public int hashCode() {
070    return Objects.hashCode(super.hashCode(), contentNamespace);
071  }
072
073  @Override
074  public String toString() {
075    return toStringHelper().add("contentNamespace", contentNamespace).toString();
076  }
077
078}
