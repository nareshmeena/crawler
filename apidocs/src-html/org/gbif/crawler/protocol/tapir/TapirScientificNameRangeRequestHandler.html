001package org.gbif.crawler.protocol.tapir;
002
003import org.gbif.crawler.RequestHandler;
004import org.gbif.crawler.strategy.ScientificNameRangeCrawlContext;
005
006import java.net.URI;
007import java.net.URISyntaxException;
008import java.util.HashMap;
009import java.util.Map;
010import javax.annotation.concurrent.NotThreadSafe;
011
012import com.google.common.collect.Maps;
013import org.apache.http.client.utils.URIBuilder;
014
015/**
016 * This is a request handler for TAPIR queries using a range of scientific names.
017 * <p/>
018 * It supports DwC, ABCD 1.2, ABCD 2.05 and ABCD 2.06.
019 * <p/>
020 * This class is not thread-safe.
021 */
022@NotThreadSafe
023public class TapirScientificNameRangeRequestHandler implements RequestHandler<ScientificNameRangeCrawlContext, String> {
024
025  // TAPIR supports multiple output formats, so we map a namespace to a search template with appropriate output model
026  private static final Map<String, String> TEMPLATE_MAPPING = new HashMap<String, String>();
027
028  private static final int MAX_RESULTS = 1000;
029
030  private final String contentNamespace;
031
032  private final URI targetUrl;
033
034  static {
035    // DwC
036    TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/dwcore/",
037                         "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
038    TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/geospatial/",
039                         "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
040    TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/curatorial/",
041                         "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
042    TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/terms/",
043                         "http://rs.tdwg.org/tapir/cs/dwc/terms/2009-09-23/template/dwc_sci_name_range.xml");
044    TEMPLATE_MAPPING.put("http://digir.net/schema/conceptual/darwin/2003/1.0",
045                         "http://rs.gbif.org/templates/tapir/dwc/1.0/sci_name_range.xml");
046    TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/1.2",
047                         "http://rs.gbif.org/templates/tapir/abcd/1.2/sci_name_range.xml");
048    TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/2.06",
049                         "http://rs.gbif.org/templates/tapir/abcd/206/sci_name_range.xml");
050    // ABCD (Default to 2.0.6 whenever 2.0.5 is encountered)
051    TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/2.05",
052                         TEMPLATE_MAPPING.get("http://www.tdwg.org/schemas/abcd/2.06"));
053
054  }
055
056  public TapirScientificNameRangeRequestHandler(TapirCrawlConfiguration configuration) {
057    contentNamespace = configuration.getContentNamespace();
058    targetUrl = configuration.getUrl();
059  }
060
061  /**
062   * @return The atoms for the TAPIR request
063   */
064  private Map<String, String> buildTapirRequestParameters(ScientificNameRangeCrawlContext context) {
065    Map<String, String> params = Maps.newHashMap();
066    params.put("op", "s");
067    params.put("t", TEMPLATE_MAPPING.get(contentNamespace));
068
069    params.put("start", String.valueOf(context.getOffset()));
070    params.put("limit", String.valueOf(MAX_RESULTS));
071
072    if (context.getLowerBound().isPresent()) {
073      params.put("lower", context.getLowerBound().get());
074    } else {
075      params.put("lower", null);
076    }
077
078    if (context.getUpperBound().isPresent()) {
079      params.put("upper", context.getUpperBound().get());
080    } else {
081      params.put("upper", null);
082    }
083
084    return params;
085  }
086
087  @Override
088  public String buildRequestUrl(ScientificNameRangeCrawlContext context) {
089    try {
090      URIBuilder uriBuilder = new URIBuilder(targetUrl);
091      for (Map.Entry<String, String> paramEntry : buildTapirRequestParameters(context).entrySet()) {
092        uriBuilder.addParameter(paramEntry.getKey(), paramEntry.getValue());
093      }
094
095      return uriBuilder.build().toString();
096    } catch (URISyntaxException e) {
097      // It's coming from a valid URL so it shouldn't ever throw this, so we swallow this
098      return null;
099    }
100  }
101
102  @Override
103  public int getLimit() {
104    return MAX_RESULTS;
105  }
106}
