001package org.gbif.crawler.protocol.tapir;
002
003import org.gbif.crawler.exception.ProtocolException;
004import org.gbif.crawler.protocol.AbstractResponseHandler;
005
006import javax.annotation.concurrent.NotThreadSafe;
007import javax.xml.stream.XMLStreamConstants;
008import javax.xml.stream.XMLStreamException;
009
010import com.google.common.base.Optional;
011import org.codehaus.stax2.XMLStreamReader2;
012
013/**
014 * This class handles a TAPIR response.
015 * <p/>
016 * Every TAPIR response has a few pieces of diagnostic information as defined in the <a
017 * href="http://www.tdwg.org/dav/subgroups/tapir/1.0/docs/tdwg_tapir_specification_2010-05-05.htm">specification</a>:
018 * <ul>
019 * <li>{@code start} index of the first element in this response</li>
020 * <li>{@code totalReturned} number of elements in this response</li>
021 * <li>{@code next} index of the next element that should be retrieved. Used for paging. This attribute is absent if
022 * there are no more records.</li>
023 * </ul>
024 * <p/>
025 * As an example. Sending a request with {@code start}=0 and {@code limit}=2 could return a response like this:
026 * <pre>
027 *   {@code <summary start='0' next='2' totalReturned='2'/>}
028 * </pre>
029 * <p/>
030 * This class is not thread-safe.
031 */
032@NotThreadSafe
033public class TapirResponseHandler extends AbstractResponseHandler {
034
035  private static final String CONTENT_ELEMENT = "search";
036  private static final String INFO_ELEMENT = "summary";
037
038  private boolean insideContent;
039  private boolean seenContent;
040
041  @Override
042  protected boolean shouldHash(XMLStreamReader2 reader) {
043    if (isElement(reader, CONTENT_ELEMENT, true)) {
044      insideContent = true;
045      return false;
046    }
047
048    boolean infoElement = isElement(reader, INFO_ELEMENT, true) || isElement(reader, INFO_ELEMENT, false);
049
050    if (insideContent && !infoElement && reader.getEventType() == XMLStreamConstants.START_ELEMENT) {
051      seenContent = true;
052    }
053
054    if (isElement(reader, CONTENT_ELEMENT, false)) {
055      insideContent = false;
056    }
057
058    return insideContent && seenContent && !infoElement;
059  }
060
061  @Override
062  protected void initialize() {
063    insideContent = false;
064    seenContent = false;
065  }
066
067  @Override
068  protected void process(XMLStreamReader2 reader) throws XMLStreamException, ProtocolException {
069    if (isElement(reader, INFO_ELEMENT, true)) {
070      int index = reader.getAttributeIndex(null, "totalReturned");
071      if (index == -1) {
072        setRecordCount(Optional.<Integer>absent());
073      } else {
074        setRecordCount(Optional.of(reader.getAttributeAsInt(index)));
075      }
076
077      if (reader.getAttributeIndex(null, "next") == -1) {
078        setEndOfRecords(Optional.<Boolean>absent());
079      } else {
080        setEndOfRecords(Optional.of(false));
081      }
082
083    }
084  }
085
086}
