001package org.gbif.crawler.protocol;
002
003import org.gbif.crawler.RequestHandler;
004import org.gbif.crawler.strategy.ScientificNameRangeCrawlContext;
005import org.gbif.crawler.util.TemplateUtils;
006
007import java.net.URI;
008import java.net.URISyntaxException;
009import java.util.HashMap;
010import java.util.Map;
011
012import com.google.common.collect.ImmutableMap;
013import com.google.common.io.Resources;
014import org.apache.http.client.utils.URIBuilder;
015
016import static com.google.common.base.Preconditions.checkNotNull;
017
018/**
019 * A abstract class that is extended by the BioCASe and DiGIR request builders because their request URLs are very
020 * similar to build.
021 * <p/>
022 * All they need to do is provide the correct constructor parameters and implement the {@link #getDefaultContext()}
023 * method.
024 */
025public abstract class AbstractScientificNameRangeRequestHandler
026  implements RequestHandler<ScientificNameRangeCrawlContext, String> {
027
028  private final String templateLocation;
029  private final String requestParamKey;
030  private final URI targetUrl;
031
032  /**
033   * All constructor parameters are mandatory and may not be null.
034   *
035   * @param targetUrl        of the dataset
036   * @param templateLocation the location on the classpath that contains the template to use for this request
037   * @param requestParamKey  the query parameter name that contains the serialized request
038   */
039  protected AbstractScientificNameRangeRequestHandler(URI targetUrl, String templateLocation, String requestParamKey) {
040    this.targetUrl = checkNotNull(targetUrl);
041    this.templateLocation = checkNotNull(templateLocation);
042    this.requestParamKey = checkNotNull(requestParamKey);
043
044    // Throws an IllegalArgumentException if it doesn't exist so serves as validation
045    Resources.getResource(templateLocation);
046  }
047
048  @Override
049  public String buildRequestUrl(ScientificNameRangeCrawlContext context) {
050    checkNotNull(context);
051
052    try {
053      return new URIBuilder(targetUrl).addParameter(requestParamKey, buildRequest(context)).build().toString();
054    } catch (URISyntaxException e) {
055      // It's coming from a valid URL so it shouldn't ever throw this, so we swallow this
056      return null;
057    }
058  }
059
060  /**
061   * @param context The request context
062   *
063   * @return the request, not the full URL
064   */
065  private String buildRequest(ScientificNameRangeCrawlContext context) {
066    Map<String, Object> templateContext = new HashMap<String, Object>(getDefaultContext());
067
068    templateContext.put("startAt", String.valueOf(context.getOffset()));
069
070    if (context.getLowerBound().isPresent()) {
071      templateContext.put("lower", context.getLowerBound().get());
072    }
073    if (context.getUpperBound().isPresent()) {
074      templateContext.put("upper", context.getUpperBound().get());
075    }
076
077    try {
078      return TemplateUtils.getAndProcess(templateLocation, templateContext);
079    } catch (Exception e) {
080      throw new IllegalArgumentException("Unable to generate request", e);
081    }
082  }
083
084  protected abstract ImmutableMap<String, Object> getDefaultContext();
085
086}
