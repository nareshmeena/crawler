001package org.gbif.crawler;
002
003import org.gbif.api.model.registry.Dataset;
004import org.gbif.common.messaging.AbstractMessageCallback;
005import org.gbif.common.messaging.ConnectionParameters;
006import org.gbif.common.messaging.DefaultMessageRegistry;
007import org.gbif.common.messaging.MessageListener;
008import org.gbif.common.messaging.api.messages.RegistryChangeMessage;
009
010import java.io.IOException;
011
012import org.codehaus.jackson.map.DeserializationConfig.Feature;
013import org.codehaus.jackson.map.ObjectMapper;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * This is a class that is very useful for testing during crawl.
019 * It will likely be deleted, or moved to a new location in the near future.
020 */
021public class TestListener {
022
023  private static final Logger LOG = LoggerFactory.getLogger(TestListener.class);
024
025  public static void main(String[] args) throws IOException {
026    ConnectionParameters connectionParameters = new ConnectionParameters("localhost", 5672, "guest", "guest", "/");
027    ObjectMapper objectMapper = new ObjectMapper().configure(Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
028
029    MessageListener listener =
030      new MessageListener(connectionParameters, new DefaultMessageRegistry(),
031        objectMapper);
032
033    listener.listen("testlistener", "registry.change.#", 100, new AbstractMessageCallback<RegistryChangeMessage>() {
034
035      @Override
036      public void handleMessage(RegistryChangeMessage message) {
037        switch (message.getChangeType()) {
038          case CREATED:
039            LOG.info("Message type: [{}], class [{}]", message.getChangeType(), message.getObjectClass());
040            if (message.getObjectClass().equals(Dataset.class)) {
041              Dataset dataset = (Dataset) message.getNewObject();
042              LOG.info("Created new Dataset with Key [{}] and Title [{}]", dataset.getKey(), dataset.getTitle());
043            }
044            break;
045          case UPDATED:
046            if (message.getObjectClass().equals(Dataset.class)) {
047              Dataset newDataset = (Dataset) message.getNewObject();
048              Dataset oldDataset = (Dataset) message.getOldObject();
049              LOG.info("Updated dataset with key [{}], old Title [{}], new Title [{}]",
050                newDataset.getKey(),
051                oldDataset.getTitle(),
052                newDataset.getTitle());
053            }
054            break;
055          case DELETED:
056            if (message.getObjectClass().equals(Dataset.class)) {
057              Dataset dataset = (Dataset) message.getOldObject();
058              LOG.info("Deleted Dataset with Key [{}] and Title [{}]", dataset.getKey(), dataset.getTitle());
059            }
060            break;
061        }
062      }
063    });
064  }
065}
