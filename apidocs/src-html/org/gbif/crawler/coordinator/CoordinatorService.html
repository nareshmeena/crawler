001package org.gbif.crawler.coordinator;
002
003import org.gbif.api.service.registry.DatasetService;
004import org.gbif.common.messaging.MessageListener;
005import org.gbif.common.messaging.api.MessageCallback;
006import org.gbif.common.messaging.api.messages.StartCrawlMessage;
007import org.gbif.crawler.CrawlerCoordinatorService;
008import org.gbif.crawler.CrawlerCoordinatorServiceImpl;
009import org.gbif.crawler.StartCrawlMessageCallback;
010
011import java.util.concurrent.Executors;
012
013import com.google.common.util.concurrent.AbstractIdleService;
014import com.google.inject.Injector;
015import org.apache.curator.framework.CuratorFramework;
016
017/**
018 * This services starts the Crawler Coordinator by listening for messages.
019 */
020public class CoordinatorService extends AbstractIdleService {
021
022  private final CoordinatorConfiguration configuration;
023
024  private MessageListener listener;
025
026  private CuratorFramework curator;
027
028  public CoordinatorService(CoordinatorConfiguration configuration) {
029    this.configuration = configuration;
030  }
031
032  @Override
033  protected void startUp() throws Exception {
034    // Create ZooKeeper and RabbitMQ connections
035    curator = configuration.zooKeeper.getCuratorFramework();
036
037    // Create Registry WS Client
038    Injector injector = configuration.registry.newRegistryInjector();
039    DatasetService datasetService = injector.getInstance(DatasetService.class);
040
041    CrawlerCoordinatorService coord = new CrawlerCoordinatorServiceImpl(curator, datasetService);
042    MessageCallback<StartCrawlMessage> callback = new StartCrawlMessageCallback(coord);
043
044    listener = new MessageListener(configuration.messaging.getConnectionParameters());
045    listener.listen(configuration.queueName, configuration.poolSize, callback);
046  }
047
048  @Override
049  protected void shutDown() throws Exception {
050    if (listener != null) {
051      listener.close();
052    }
053    if (curator != null) {
054      curator.close();
055    }
056  }
057}
