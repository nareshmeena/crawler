001package org.gbif.crawler.coordinator;
002
003import org.gbif.common.messaging.config.MessagingConfiguration;
004import org.gbif.crawler.common.RegistryConfiguration;
005import org.gbif.crawler.common.ZooKeeperConfiguration;
006
007import javax.validation.Valid;
008import javax.validation.constraints.Min;
009import javax.validation.constraints.NotNull;
010
011import com.beust.jcommander.Parameter;
012import com.beust.jcommander.ParametersDelegate;
013import com.google.common.base.Objects;
014
015public class CoordinatorConfiguration {
016
017  @ParametersDelegate
018  @Valid
019  @NotNull
020  public MessagingConfiguration messaging = new MessagingConfiguration();
021
022  @ParametersDelegate
023  @Valid
024  @NotNull
025  public ZooKeeperConfiguration zooKeeper = new ZooKeeperConfiguration();
026
027  @ParametersDelegate
028  @Valid
029  @NotNull
030  public RegistryConfiguration registry = new RegistryConfiguration();
031
032  @Parameter(names = "--pool-size")
033  @Min(1)
034  public int poolSize = 5;
035
036  @Parameter(names = "--queue-name")
037  @NotNull
038  public String queueName = "crawler_coordinator";
039
040  @Override
041  public String toString() {
042    return Objects.toStringHelper(this)
043      .add("messaging", messaging)
044      .add("zooKeeper", zooKeeper)
045      .add("registry", registry)
046      .add("poolSize", poolSize)
047      .add("queueName", queueName)
048      .toString();
049  }
050}
