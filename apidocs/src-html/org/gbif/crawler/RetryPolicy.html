001package org.gbif.crawler;
002
003/**
004 * Will be called by the crawler to determine if certain operations should be retried after errors during the crawl.
005 */
006public interface RetryPolicy {
007
008  /**
009   * Call this method after every {@link org.gbif.crawler.exception.ProtocolException} to determine if the operation
010   * should be retried.
011   *
012   * @return if the operation in question should be retried
013   */
014  boolean allowAfterProtocolException();
015
016  /**
017   * Call this method after every {@link org.gbif.crawler.exception.TransportException} to determine if the operation
018   * should be retried.
019   *
020   * @return if the operation in question should be retried
021   */
022  boolean allowAfterTransportException();
023
024  /**
025   * Called to register a successful request. This can be used by implementations to reset internal counters.
026   */
027  void successfulRequest();
028
029  /**
030   * Called to let the strategy know that we've given up the current request and will try the next if allowed.
031   */
032  void giveUpRequest();
033
034  /**
035   * Used to see if the crawl should be aborted.
036   *
037   * @return {@code true} if the crawl should be aborted
038   */
039  boolean abortCrawl();
040
041}
