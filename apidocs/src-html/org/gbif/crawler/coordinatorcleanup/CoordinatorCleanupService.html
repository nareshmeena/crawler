043/**
044 * This services starts the Crawler Coordinator by listening for messages.
045 */
046public class CoordinatorCleanupService extends AbstractScheduledService {
047
048  private static final Logger LOG = LoggerFactory.getLogger(CoordinatorCleanupService.class);
049  private static final ObjectMapper MAPPER = new ObjectMapper();
050  private final CoordinatorCleanupConfiguration configuration;
051  private CuratorFramework curator;
052  private DatasetProcessService service;
053  private DatasetProcessStatusService registryService;
054
055  public CoordinatorCleanupService(CoordinatorCleanupConfiguration configuration) {
056    this.configuration = configuration;
057  }
058
059  @Override
060  protected void runOneIteration() throws Exception {
061    LOG.info("Checking if any crawls have finished now.");
062    Set<DatasetProcessStatus> statuses;
063    try {
064      statuses = service.getRunningDatasetProcesses();
065      LOG.debug("All DatasetProcessStatuses retrieved: {}", statuses);
066    } catch (ServiceUnavailableException e) {
067      LOG.warn("Caught exception while trying to retrieve all running datasets, will try again", e);
068      return;
069    }
070
071    for (DatasetProcessStatus status : statuses) {
072      LOG.info("Checking DatasetProcessStatus with UUID [{}] now", status.getDatasetKey());
073
074      try {
075        updateRegistry(status);
076      } catch (Exception e) {
077        LOG.error("Unable to callback and update the registry. Aborting this cleanup round, will try again later.", e);
078        return;
079      }
080
081      if (!checkDoneProcessing(status)) {
082        continue;
083      }
084
085      LOG.debug(MAPPER.writeValueAsString(status));
086      try {
087        File file = new File(configuration.archiveDirectory,
088                             status.getDatasetKey() + "_" + status.getCrawlJob().getAttempt() + "_result.json");
089        Files.createParentDirs(file);
090        Files.write(MAPPER.writeValueAsBytes(status), file);
091      } catch (IOException e) {
092        LOG.warn("Could not write status due to exception. [{}]", status.getDatasetKey(), e);
093      }
094
095      // If all of these things are true we can delete this dataset from ZK and dump info to disc
096      delete(status.getDatasetKey());
097    }
098    LOG.info("Done checking for finished crawls");
099  }
100
101  @Override
102  protected void startUp() throws Exception {
103    Properties props = new Properties();
104    props.setProperty("registry.ws.url", configuration.registry.wsUrl);
105
106    Injector injector = Guice.createInjector(new CrawlerWsClientModule(ConfigUtils.toProperties(configuration)),
107                                             new RegistryWsClientModule(props),
108                                             new SingleUserAuthModule(configuration.registry.user,
109                                                                      configuration.registry.password));
110    service = injector.getInstance(DatasetProcessService.class);
111    registryService = injector.getInstance(DatasetProcessStatusService.class);
112    curator = configuration.zooKeeper.getCuratorFramework();
113    MAPPER.configure(SerializationConfig.Feature.INDENT_OUTPUT, true);
114  }
115
116  @Override
117  protected void shutDown() throws Exception {
118    if (curator != null) {
119      curator.close();
120    }
121  }
122
123  @Override
124  protected Scheduler scheduler() {
125    return Scheduler.newFixedDelaySchedule(0, configuration.interval, TimeUnit.MINUTES);
126  }
127
128  /**
129   * Updates the registry with up to date information about the status of processing this dataset.
130   *
131   * @throws Exception when there was an error updating the registry. This can happen when it's offline.
132   */
133  private void updateRegistry(DatasetProcessStatus status) throws Exception {
134    UUID datasetKey = status.getDatasetKey();
135    int attempt = status.getCrawlJob().getAttempt();
136    DatasetProcessStatus persisted = registryService.getDatasetProcessStatus(datasetKey, attempt);
137    LOG.info(MAPPER.writeValueAsString(status));
138    if (persisted == null) {
139      registryService.createDatasetProcessStatus(status);
140    } else {
141      registryService.updateDatasetProcessStatus(status);
142    }
143  }
144
145  /**
146   * This method checks whether a certain Dataset has been fully processed (crawled, occurrences processed etc.)
147   *
148   * @param status of the dataset in question to check
149   *
150   * @return {@code true} if we are done processing, {@code false} otherwise
151   */
152  private boolean checkDoneProcessing(DatasetProcessStatus status) {
153    if (status.getFinishedCrawling() == null) {
154      return false;
155    }
156
157    // Done fragmenting?
158    // We are done when we have as many pages fragmented (in error or successful) as we did crawl
159    if (status.getPagesCrawled() != status.getPagesFragmentedError() + status.getPagesFragmentedSuccessful()) {
160      return false;
161    }
162
163    // Done persisting fragments?
164    // We are done when we have processed as many fragments as the fragmenter emitted. During this processing we could
165    // generate more raw occurrence records than we got fragments due to ABCD2
166    if (status.getFragmentsProcessed() != status.getFragmentsEmitted()) {
167      return false;
168    }
169
170    // Are we done persisting verbatim occurrences?
171    // We are done when we have persisted (in error or successful) as many verbatim records as there were new or
172    // updated raw occurrences in the previous steps
173    if (status.getVerbatimOccurrencesPersistedSuccessful() + status.getVerbatimOccurrencesPersistedError()
174        != status.getRawOccurrencesPersistedNew() + status.getRawOccurrencesPersistedUpdated()) {
175      return false;
176    }
177
178    // Are we done interpreting occurrences?
179    // We are done when we have interpreted (in error or successful) as many occurrences as there were successful
180    // verbatim occurrences persisted
181    if (status.getInterpretedOccurrencesPersistedSuccessful() + status.getInterpretedOccurrencesPersistedError()
182        != status.getVerbatimOccurrencesPersistedSuccessful()) {
183      return false;
184    }
185
186    return true;
187  }
188
189  private void delete(UUID datasetKey) {
190    LOG.info("Done with [{}]", datasetKey);
191    String path = CrawlerNodePaths.getCrawlInfoPath(datasetKey);
192    try {
193      recursiveDelete(path, curator.getChildren().forPath(path));
194      curator.delete().forPath("crawls/" + datasetKey);
195    } catch (Exception e) {
196      LOG.info("Couldn't delete [{}]", datasetKey, e);
197    }
198  }
199
200  private void recursiveDelete(String parentPath, Collection<String> paths) throws Exception {
201    if (paths.isEmpty()) {
202      return;
203    }
204
205    for (String path : paths) {
206      String fullPath = parentPath + "/" + path;
207      recursiveDelete(fullPath, curator.getChildren().forPath(fullPath));
208      curator.delete().forPath(fullPath);
209    }
210  }
