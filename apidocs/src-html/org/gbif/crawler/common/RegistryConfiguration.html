001package org.gbif.crawler.common;
002
003import org.gbif.cli.ConfigUtils;
004import org.gbif.cli.PropertyName;
005import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
006import org.gbif.ws.client.guice.SingleUserAuthModule;
007
008import java.util.Properties;
009import javax.validation.constraints.NotNull;
010
011import com.beust.jcommander.Parameter;
012import com.google.common.base.Objects;
013import com.google.common.base.Strings;
014import com.google.inject.Guice;
015import com.google.inject.Injector;
016
017/**
018 * A configuration class which can be used to get all the details needed to create a writable connection to the
019 * GBIF registry.
020 */
021public class RegistryConfiguration {
022
023  @Parameter(names = "--registry-ws")
024  @PropertyName("registry.ws.url")
025  @NotNull
026  public String wsUrl = "http://api.gbif.org/";
027
028  @Parameter(names = "--registry-user")
029  @NotNull
030  public String user;
031
032  @Parameter(names = "--registry-password", password = true)
033  @NotNull
034  public String password;
035
036  @Override
037  public String toString() {
038    return Objects.toStringHelper(this)
039      .add("registryWsUrl", wsUrl)
040      .add("registryUser", user)
041      .add("registryPassword", Strings.repeat("*", Strings.nullToEmpty(password).length()))
042      .toString();
043  }
044
045  /**
046   * Convenience method to setup a guice injector with a writable registry client module using the configuration
047   * of this instance.
048   *
049   * @return guice injector with RegistryWsClientModule bound
050   */
051  public Injector newRegistryInjector() {
052    // setup writable registry client
053    Properties properties = ConfigUtils.toProperties(this);
054    RegistryWsClientModule regModule = new RegistryWsClientModule(properties);
055    SingleUserAuthModule authModule = new SingleUserAuthModule(user, password);
056
057    return Guice.createInjector(regModule, authModule);
058  }
059
060}
