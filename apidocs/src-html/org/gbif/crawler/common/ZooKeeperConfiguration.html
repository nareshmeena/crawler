001package org.gbif.crawler.common;
002
003import java.io.IOException;
004import javax.validation.constraints.Min;
005import javax.validation.constraints.NotNull;
006
007import com.beust.jcommander.Parameter;
008import com.fasterxml.jackson.annotation.JsonIgnore;
009import com.google.common.base.Objects;
010import org.apache.curator.framework.CuratorFramework;
011import org.apache.curator.framework.CuratorFrameworkFactory;
012import org.apache.curator.retry.ExponentialBackoffRetry;
013
014/**
015 * A configuration class which can be used to get all the details needed to create a connection to ZooKeeper needed by
016 * the Curator Framework.
017 * <p/>
018 * It provides a convenience method ({@link #getCuratorFramework()} ()} to actually get a {@link CuratorFramework}
019 * object when populated fully.
020 */
021@SuppressWarnings("PublicField")
022public class ZooKeeperConfiguration {
023
024  @Parameter(
025    names = "--zk-connection-string",
026    description = "The connection string to connect to ZooKeeper")
027  @NotNull
028  public String connectionString;
029
030  @Parameter(
031    names = "--zk-namespace",
032    description = "The namespace in ZooKeeper under which all data lives")
033  @NotNull
034  public String namespace;
035
036  @Parameter(
037    names = "--zk-sleep-time",
038    description = "Initial amount of time to wait between retries in ms")
039  @Min(1)
040  public int baseSleepTime = 1000;
041
042  @Parameter(
043    names = "--zk-max-retries",
044    description = "Max number of times to retry")
045  @Min(1)
046  public int maxRetries = 10;
047
048  /**
049   * This method returns a connection object to ZooKeeper with the provided settings and creates and starts a {@link
050   * CuratorFramework}. These settings are not validated in this method so only call it when the object has been
051   * validated.
052   *
053   * @return started CuratorFramework
054   *
055   * @throws IOException if connection fails
056   */
057  @JsonIgnore
058  public CuratorFramework getCuratorFramework() throws IOException {
059    CuratorFramework curator = CuratorFrameworkFactory.builder()
060      .namespace(namespace)
061      .retryPolicy(new ExponentialBackoffRetry(baseSleepTime, maxRetries))
062      .connectString(connectionString)
063      .build();
064    curator.start();
065    return curator;
066  }
067
068  @Override
069  public String toString() {
070    return Objects.toStringHelper(this)
071      .add("connectionString", connectionString)
072      .add("namespace", namespace)
073      .add("baseSleepTime", baseSleepTime)
074      .add("maxRetries", maxRetries)
075      .toString();
076
077  }
078}
