001package org.gbif.crawler.registry.metasync;
002
003import org.gbif.api.service.registry.DatasetService;
004import org.gbif.api.service.registry.InstallationService;
005import org.gbif.api.service.registry.MetasyncHistoryService;
006import org.gbif.common.messaging.AbstractMessageCallback;
007import org.gbif.common.messaging.MessageListener;
008import org.gbif.common.messaging.api.messages.StartMetasyncMessage;
009import org.gbif.registry.metasync.MetadataSynchroniserImpl;
010import org.gbif.registry.metasync.api.MetadataSynchroniser;
011import org.gbif.registry.metasync.api.SyncResult;
012import org.gbif.registry.metasync.protocols.biocase.BiocaseMetadataSynchroniser;
013import org.gbif.registry.metasync.protocols.digir.DigirMetadataSynchroniser;
014import org.gbif.registry.metasync.protocols.tapir.TapirMetadataSynchroniser;
015import org.gbif.registry.metasync.resulthandler.DebugHandler;
016import org.gbif.registry.metasync.resulthandler.RegistryUpdater;
017import org.gbif.registry.metasync.util.HttpClientFactory;
018import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
019import org.gbif.ws.client.guice.SingleUserAuthModule;
020
021import java.util.Properties;
022import java.util.concurrent.TimeUnit;
023
024import com.google.common.collect.ImmutableList;
025import com.google.common.util.concurrent.AbstractIdleService;
026import com.google.inject.Guice;
027import com.google.inject.Injector;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031public class MetasyncService extends AbstractIdleService {
032
033  private static final Logger LOG = LoggerFactory.getLogger(MetasyncService.class);
034  private final MetasyncConfiguration configuration;
035  private MessageListener listener;
036
037  public MetasyncService(MetasyncConfiguration configuration) {
038    this.configuration = configuration;
039  }
040
041  @Override
042  protected void startUp() throws Exception {
043    Properties props = new Properties();
044    props.setProperty("registry.ws.url", configuration.registry.wsUrl);
045    Injector injector = Guice.createInjector(new RegistryWsClientModule(props),
046      new SingleUserAuthModule(configuration.registry.user, configuration.registry.password));
047
048    DatasetService datasetService = injector.getInstance(DatasetService.class);
049    MetasyncHistoryService historyService = injector.getInstance(MetasyncHistoryService.class);
050    RegistryUpdater registryUpdater = new RegistryUpdater(datasetService, historyService);
051    InstallationService installationService = injector.getInstance(InstallationService.class);
052
053    listener = new MessageListener(configuration.messaging.getConnectionParameters(), 1);
054    listener.listen(configuration.queueName, configuration.threadCount,
055      new MetasyncCallback(registryUpdater, installationService));
056  }
057
058  @Override
059  protected void shutDown() throws Exception {
060    listener.close();
061  }
062
063  private static class MetasyncCallback extends AbstractMessageCallback<StartMetasyncMessage> {
064
065    private final RegistryUpdater registryUpdater;
066    private final MetadataSynchroniser synchroniser;
067
068    private MetasyncCallback(RegistryUpdater registryUpdater, InstallationService installationService) {
069      this.registryUpdater = registryUpdater;
070
071      HttpClientFactory clientFactory = new HttpClientFactory(120, TimeUnit.MINUTES);
072
073      MetadataSynchroniserImpl newSynchroniser = new MetadataSynchroniserImpl(installationService);
074
075      newSynchroniser.registerProtocolHandler(new DigirMetadataSynchroniser(clientFactory.provideHttpClient()));
076      newSynchroniser.registerProtocolHandler(new TapirMetadataSynchroniser(clientFactory.provideHttpClient()));
077      newSynchroniser.registerProtocolHandler(new BiocaseMetadataSynchroniser(clientFactory.provideHttpClient()));
078      this.synchroniser = newSynchroniser;
079
080    }
081
082    @Override
083    public void handleMessage(StartMetasyncMessage message) {
084      SyncResult syncResult = synchroniser.synchroniseInstallation(message.getInstallationKey());
085      LOG.info("Done syncing. Processing result.");
086      DebugHandler.processResult(syncResult);
087
088      registryUpdater.saveSyncResultsToRegistry(ImmutableList.of(syncResult));
089
090    }
091
092  }
093
094}
