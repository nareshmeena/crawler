001package org.gbif.crawler;
002
003import org.gbif.crawler.exception.FatalCrawlException;
004import org.gbif.crawler.exception.ProtocolException;
005import org.gbif.crawler.exception.TransportException;
006
007import com.google.common.base.Optional;
008
009/**
010 * Implementations of this interface will handle the response received by {@link CrawlClient}s and produce a result but
011 * it also plays a part in the paging through resources.
012 * <p/>
013 * They have three jobs:
014 * <ul>
015 * <li>Convert the response into a format that is suitable for shipping as a message</li>
016 * <li>Extract information from the response that is important for paging</li>
017 * <li>Extract enough information from the response to see if it is a duplicate response</li>
018 * </ul>
019 *
020 * @param <RESPONSE> the type of response to handle. This is dependent on the {@link CrawlClient} implementation
021 * @param <RESULT>   the type of result this response handler produces when processing a response
022 */
023public interface ResponseHandler<RESPONSE, RESULT> {
024
025  /**
026   * Handles a response and produces a result.
027   *
028   * @param response to be handled
029   *
030   * @throws ProtocolException  Implementation specific
031   * @throws TransportException Implementation specific
032   */
033  RESULT handleResponse(RESPONSE response) throws FatalCrawlException, ProtocolException, TransportException;
034
035  /**
036   * This can be used to see if this response handler is in an illegal state or not. This can happen after parsing a
037   * wrong response or if there has been a transport exception. A proper response needs to be handled for this to be
038   * cleared.
039   *
040   * @return false if the other getter methods will throw a {@link IllegalStateException}
041   */
042  boolean isValidState();
043
044  /**
045   * @return the number of records in this page
046   */
047  Optional<Integer> getRecordCount();
048
049  /**
050   * @return true if there are more records to extract
051   */
052  Optional<Boolean> isEndOfRecords();
053
054  /**
055   * @return a hash value for the current page. This can be used to detect duplicate pages.
056   */
057  Optional<Long> getContentHash();
058
059}
