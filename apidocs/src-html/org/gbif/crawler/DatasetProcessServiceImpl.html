075  private static final QueueSerializer<UUID> UUID_SERIALIZER = new UuidSerializer();

077  private final RetryPolicy counterRetryPolicy = new RetryNTimes(10, 100);

082  /**
083   * Creates a CrawlerMetricsService. Responsible for interacting with a ZooKeeper instance in a read-only fashion.
084   *
085   * @param curator to access ZooKeeper
086   * @param mapper  to deserialize CrawlJob JSON back into an object
087   */
088  @Inject
089  public DatasetProcessServiceImpl(CuratorFramework curator, ObjectMapper mapper, Executor executor) {
090    this.curator = checkNotNull(curator, "curator can't be null");
091    this.mapper = checkNotNull(mapper, "mapper can't be null");
092    this.executor = checkNotNull(executor, "executor can't be null");
093  }

095  // TODO: This message need to be much more lenient with bad data
096  @Override
097  public DatasetProcessStatus getDatasetProcessStatus(UUID datasetKey) {
098    checkNotNull(datasetKey, "datasetKey can't be null");
099
100    // Check if dataset is actually being processed right now
101    try {
102      if (curator.checkExists().forPath(getCrawlInfoPath(datasetKey)) == null) {
103        return null;
104      }
105    } catch (Exception e) {
106      throw new ServiceUnavailableException("Error communicating with ZooKeeper", e);
107    }
108
109    DatasetProcessStatus.Builder builder = DatasetProcessStatus.builder();
110
111    String crawlPath = getCrawlInfoPath(datasetKey);
112    builder.datasetKey(datasetKey);
113
114    // Here we're trying to load all information from Zookeeper into the DatasetProcessStatus object
115    try {
116      byte[] crawlJobBytes = curator.getData().forPath(crawlPath);
117      CrawlJob crawlJob = mapper.readValue(crawlJobBytes, CrawlJob.class);
118      builder.crawlJob(crawlJob);
119
120      String path = getCrawlInfoPath(datasetKey, CRAWL_CONTEXT);
121      if (curator.checkExists().forPath(path) != null) {
122        byte[] responseData = curator.getData().forPath(path);
123        if (responseData != null) {
124          builder.crawlContext(new String(responseData, Charsets.UTF_8));
125        }
126      }
127
128      path = getCrawlInfoPath(datasetKey, DECLARED_COUNT);
129      if (curator.checkExists().forPath(path) != null) {
130        byte[] responseData = curator.getData().forPath(path);
131        builder.declaredCount(Long.valueOf(new String(responseData)));
132      }
133
134      // should there be no started crawling, nothing else is read for now, later we might have to remove this
135      // optimization as we could pause and resume crawls
136      path = getCrawlInfoPath(datasetKey, STARTED_CRAWLING);
137      if (curator.checkExists().forPath(path) != null) {
138        byte[] responseData = curator.getData().forPath(path);
139        builder.startedCrawling(asDate(responseData));
140
141        path = getCrawlInfoPath(datasetKey, FINISHED_CRAWLING);
142        if (curator.checkExists().forPath(path) != null) {
143          responseData = curator.getData().forPath(path);
144          builder.finishedCrawling(asDate(responseData));
145
146          responseData = curator.getData().forPath(getCrawlInfoPath(datasetKey, FINISHED_REASON));
147          builder.finishReason(FinishReason.valueOf(new String(responseData)));
148        }
149
150        builder.pagesCrawled(getCounter(crawlPath, PAGES_CRAWLED).or(0L));
151        builder.pagesFragmentedSuccessful(getCounter(crawlPath, PAGES_FRAGMENTED_SUCCESSFUL).or(0L));
152        builder.pagesFragmentedError(getCounter(crawlPath, PAGES_FRAGMENTED_ERROR).or(0L));
153        builder.fragmentsEmitted(getCounter(crawlPath, FRAGMENTS_EMITTED).or(0L));
154        builder.fragmentsReceived(getCounter(crawlPath, FRAGMENTS_RECEIVED).or(0L));
155        builder.rawOccurrencesPersistedNew(getCounter(crawlPath, RAW_OCCURRENCES_PERSISTED_NEW).or(0L));
156        builder.rawOccurrencesPersistedUpdated(getCounter(crawlPath, RAW_OCCURRENCES_PERSISTED_UPDATED).or(0L));
157        builder.rawOccurrencesPersistedUnchanged(getCounter(crawlPath, RAW_OCCURRENCES_PERSISTED_UNCHANGED).or(0L));
158        builder.rawOccurrencesPersistedError(getCounter(crawlPath, RAW_OCCURRENCES_PERSISTED_ERROR).or(0L));
159        builder.fragmentsProcessed(getCounter(crawlPath, FRAGMENTS_PROCESSED).or(0L));
160        builder.verbatimOccurrencesPersistedSuccessful(getCounter(crawlPath,
161                                                                  VERBATIM_OCCURRENCES_PERSISTED_SUCCESSFUL).or(0L));
162        builder.verbatimOccurrencesPersistedError(getCounter(crawlPath, VERBATIM_OCCURRENCES_PERSISTED_ERROR).or(0L));
163        builder.interpretedOccurrencesPersistedSuccessful(getCounter(crawlPath,
164                                                                     INTERPRETED_OCCURRENCES_PERSISTED_SUCCESSFUL).or(0L));
165        builder.interpretedOccurrencesPersistedError(getCounter(crawlPath, INTERPRETED_OCCURRENCES_PERSISTED_ERROR).or(
166          0L));
167      }
168
169    } catch (Exception e) {
170      throw new ServiceUnavailableException("Error communicating with ZooKeeper, getting status for "
171                                            + datasetKey.toString(), e);
172    }
173    return builder.build();
174  }

176  @Override
177  public Set<DatasetProcessStatus> getRunningDatasetProcesses() {
178    List<UUID> pendingUuids = getPendingCrawlUuids(XML_CRAWL);
179    pendingUuids.addAll(getPendingCrawlUuids(DWCA_CRAWL));
180    List<String> allCrawls = getChildren(CRAWL_INFO, false);
181    List<UUID> allCrawlUuids = Lists.newArrayList();
182    for (String crawl : allCrawls) {
183      allCrawlUuids.add(UUID.fromString(crawl));
184    }
185
186    allCrawlUuids.removeAll(pendingUuids);
187
188    return Sets.newHashSet(getDatasetProcessStatuses(allCrawlUuids));
189  }

191  @Override
192  public List<DatasetProcessStatus> getPendingXmlDatasetProcesses() {
193    List<UUID> pendingCrawlUuids = getPendingCrawlUuids(XML_CRAWL);
194    return getDatasetProcessStatuses(pendingCrawlUuids);
195  }

203  /**
204   * Gets all pending but not running crawls from the specified path (either XML or DwC-A queue usually)
205   *
206   * @param path which needs two sub nodes: one for queued and one for running crawls
207   */
208  private List<UUID> getPendingCrawlUuids(String path) {
209    checkNotNull(path, "path can't be null");
210    LOG.debug("Requested a list of all queued datasets");
211
212    // Gets the list of running and queued queue IDs
213    List<String> runningIdentifiers = getChildren(buildPath(path, RUNNING_CRAWLS), false);
214    List<String> queueIdentifiers = getChildren(buildPath(path, QUEUED_CRAWLS), true);
215
216    // retain identifiers that are only present in the queued-jobs queue, so we are left with all queue identifiers
217    // that are _not_ currently being worked on
218    queueIdentifiers.removeAll(runningIdentifiers);
219
220    LOG.debug("[{}] queued datasets, retrieving UUIDs", queueIdentifiers.size());
221    List<UUID> queueKeys = getQueueKeys(buildPath(path, QUEUED_CRAWLS), queueIdentifiers);
222    LOG.debug("Retrieved UUIDs for queued datasets");
223
224    return queueKeys;
225  }

227  /**
228   * This method will retrieve the processing status for a list of UUIDs
229   *
230   * @param queueKeys uuids to query
231   *
232   * @return list of processing statuses, at the moment no guarantees are made for ordering
233   */
234  private List<DatasetProcessStatus> getDatasetProcessStatuses(Collection<UUID> queueKeys) {
235    CompletionService<DatasetProcessStatus> completionService =
236      new ExecutorCompletionService<DatasetProcessStatus>(executor);
237    for (final UUID queueKey : queueKeys) {
238      completionService.submit(new Callable<DatasetProcessStatus>() {
239        @Override
240        public DatasetProcessStatus call() throws Exception {
241          return getDatasetProcessStatus(queueKey);
242        }
243      });
244    }
245
246    List<DatasetProcessStatus> processStatuses = Lists.newArrayList();
247
248    for (int i = 0; i < queueKeys.size(); i++) {
249      try {
250        Future<DatasetProcessStatus> future = completionService.take();
251        DatasetProcessStatus status = future.get();
252        processStatuses.add(status);
253      } catch (InterruptedException ignored) {
254        Thread.currentThread().interrupt();
255      } catch (ExecutionException e) {
256        LOG.warn("Caught exception trying to retrieve dataset", e.getCause());
257        // TODO: Commented out because we don't fail if ZK is inconsistent for a second. On the other hand if we
258        // get only exceptions we want to capture that somehow and throw something in the end.
259        //throw new ServiceUnavailableException("Exception while getting dataset process status", e.getCause());
260      }
261    }
262    return processStatuses;
263  }

265  /**
266   * This gets a counter which has been written using a {@link DistributedAtomicLong} if it exists.
267   *
268   * @param rootPath where this counter is supposed to be in
269   * @param counter  to read
270   *
271   * @return the counter value or absent if it didn't exist
272   */
273  private Optional<Long> getCounter(String rootPath, String counter) {
274    DistributedAtomicLong dal = new DistributedAtomicLong(curator, rootPath + "/" + counter, counterRetryPolicy);
275    try {
276      return Optional.fromNullable(dal.get().preValue());
277    } catch (Exception ignored) {
278      return Optional.absent();
279    }
280  }

282  /**
283   * Gets a list of {@link UUID} keys belonging to datasets. Each dataset key is associated to one
284   * queue identifier.
285   *
286   * @param path             of the queue (the actual path to the queue not a parent node and not the running crawls)
287   * @param queueIdentifiers queue identifiers that might or might not contain a dataset {@link UUID} key
288   *
289   * @return list of dataset {@link UUID} keys
290   */
291  private List<UUID> getQueueKeys(final String path, Iterable<String> queueIdentifiers) {
292    CompletionService<UUID> completionService = new ExecutorCompletionService<UUID>(executor);
293    List<Future<UUID>> futures = Lists.newArrayList();
294    for (final String queueIdentifier : queueIdentifiers) {
295      Future<UUID> future = completionService.submit(new Callable<UUID>() {
296        @Override
297        public UUID call() throws Exception {
298          byte[] responseData = curator.getData().forPath(buildPath(path, queueIdentifier));
299          return QueueHelper.deserializeSingle(responseData, UUID_SERIALIZER);
300        }
301      });
302      futures.add(future);
303    }
304
305    List<UUID> keys = Lists.newArrayList();
306    for (Future<UUID> future : futures) {
307      try {
308        keys.add(future.get());
309      } catch (InterruptedException ignored) {
310        Thread.currentThread().interrupt();
311      } catch (ExecutionException e) {
312        throw new ServiceUnavailableException("Exception while getting dataset process status", e.getCause());
313      }
314    }
315    LOG.debug("Retrieved all queued dataset process statuses");
316    return keys;
317  }

319  /**
320   * Given a path, returns all the child nodes, optionally lexicographically sorted.
321   *
322   * @param path   to get a list of queue identifiers
323   * @param sorted whether to sort the child nodes lexicographically
324   *
325   * @return a list of child nodes
326   */
327  private List<String> getChildren(String path, boolean sorted) {
328    List<String> identifiers;
329    try {
330      identifiers = curator.getChildren().forPath(path);
331    } catch (Exception e) {
332      throw new ServiceUnavailableException("Error communicating with ZooKeeper", e);
333    }
334
335    if (sorted) {
336      Collections.sort(identifiers, String.CASE_INSENSITIVE_ORDER);
337    }
338    return identifiers;
339  }

341  /**
342   * Utility to convert a date as an array of bytes to proper {@link Date} format.
343   *
344   * @param dateAsBytes a date as a byte array
345   *
346   * @return The properly formatted date
347   */
348  private Date asDate(byte[] dateAsBytes) {
349    if (dateAsBytes == null) {
350      return null;
351    }
352
353    String dateAsString = new String(dateAsBytes, Charsets.UTF_8);
354    try {
355      // TODO: is the timezone right? Check with real data coming from zookeeper.
356      SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ");
357      return sdf.parse(dateAsString);
358    } catch (ParseException pe) {
359      LOG.warn("Date was not parsed successfully: [{}]: ", dateAsString, pe);
360    }
361    return null;
362  }
