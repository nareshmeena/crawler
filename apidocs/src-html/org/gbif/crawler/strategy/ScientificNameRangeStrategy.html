001package org.gbif.crawler.strategy;
002
003import java.util.NoSuchElementException;
004import javax.annotation.concurrent.NotThreadSafe;
005
006import static com.google.common.base.Preconditions.checkNotNull;
007
008/**
009 * This implements a crawling strategy that iterates over scientific name ranges (even though scientific name is not
010 * hardcoded in this class).
011 * <p/>
012 * The strategy is to first search everything before 'aaa' and from there we always increment
013 * the second character: {@code null} -> 'aaa', 'aaa' -> 'aba', 'aba' -> 'aca', ..., 'aza' -> 'baa', ..., 'zzz' ->
014 * {@code null}.
015 * <p/>
016 * We use three character strings even though we don't change the third because a lot of provides require at least that
017 * many characters for a search.
018 * <p/>
019 * <em>Note:</em> There is currently no way of stopping the job automatically when a certain range has been scanned.
020 * The initial context is just taken as a starting point buf from there on all the other ranges will be emitted.
021 * <p/>
022 * This class is not thread-safe.
023 */
024@NotThreadSafe
025public class ScientificNameRangeStrategy extends AbstractStrategy<ScientificNameRangeCrawlContext> {
026
027  private boolean isFirst = true;
028
029  private final ScientificNameRangeCrawlContext context;
030
031  public ScientificNameRangeStrategy(ScientificNameRangeCrawlContext context) {
032    this.context = checkNotNull(context, "context can't be null");
033  }
034
035  @Override
036  public ScientificNameRangeCrawlContext next() {
037    if (!hasNext()) {
038      throw new NoSuchElementException("There are no more elements in this crawl range");
039    }
040
041    // If this is our first request we just return the context as we got it because we don't want to lose that first
042    // step which usually is null..aaa
043    if (isFirst) {
044      isFirst = false;
045      return context;
046    }
047
048    // We start with the last upper bound
049    context.setLowerBound(context.getUpperBound().get());
050
051    // If we are at the very end we still need to search zzz..null
052    if (context.getLowerBound().get().equals("Zza")) {
053      context.setUpperBound(null);
054      return context;
055    }
056
057    // Otherwise increment normally as per Javadoc
058    char[] chars = context.getUpperBound().get().toCharArray();
059    for (int pos = 1; pos >= 0; pos--) {
060      if (chars[pos] == 'z') {
061        chars[pos] = 'a';
062      } else {
063        chars[pos]++;
064        break;
065      }
066    }
067
068    context.setUpperBound(new String(chars));
069    return context;
070  }
071
072  /**
073   * We are at the end when the lower bound is <em>Zzz</em> and the upper bound is {@code null}.
074   *
075   * @return {@code true} if there is more work pending
076   */
077  @Override
078  public boolean hasNext() {
079    return context.getUpperBound().isPresent();
080  }
081
082}
