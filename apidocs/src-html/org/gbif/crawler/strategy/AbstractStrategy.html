001package org.gbif.crawler.strategy;
002
003import org.gbif.crawler.CrawlContext;
004import org.gbif.crawler.CrawlStrategy;
005
006/**
007 * A base class for {@link CrawlStrategy} implementations.
008 * <p/>
009 * Because {@link CrawlStrategy} extends {@link Iterable} we need to implement the {@link #remove()} method but we
010 * don't expect to support it. This base class thus provides a default implementation.
011 *
012 * @param <CTX> the kind of context this strategy expects. It needs to be a subclass of {@link
013 *              org.gbif.crawler.CrawlContext} because that supports paging which is something every single request
014 *              might need no matter what the strategy is
015 */
016public abstract class AbstractStrategy<CTX extends CrawlContext> implements CrawlStrategy<CTX> {
017
018  @Override
019  public void remove() {
020    throw new UnsupportedOperationException("Not implemented yet");
021  }
022
023}
