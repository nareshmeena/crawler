001package org.gbif.crawler;
002
003import java.util.UUID;
004
005import com.google.common.base.Charsets;
006import org.apache.curator.framework.recipes.queue.QueueSerializer;
007
008/**
009 * Needed by the Curator Queue to serialize and deserialize UUIDs.
010 */
011public class UuidSerializer implements QueueSerializer<UUID> {
012
013  @Override
014  public byte[] serialize(UUID item) {
015    return item.toString().getBytes(Charsets.UTF_8);
016  }
017
018  @Override
019  public UUID deserialize(byte[] bytes) {
020    return UUID.fromString(new String(bytes, Charsets.UTF_8));
021  }
022
023}
