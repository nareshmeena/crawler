001package org.gbif.crawler;
002
003import org.gbif.api.exception.ServiceUnavailableException;
004
005import java.util.UUID;
006
007/**
008 * The public interface of the Crawler Coordinator. This allows to schedule a crawl.
009 */
010public interface CrawlerCoordinatorService {
011
012  /**
013   * Initiates a crawl of an existing dataset at a given priority.
014   *
015   * @param datasetKey of the dataset to crawl
016   * @param priority    of this crawl. Lower numbers mean higher priorities. This priority can be chosen arbitrarily.
017   *
018   * @throws ServiceUnavailableException if there are any problems communicating with the Registry or ZooKeeper.
019   *                                     ZooKeeper will already have been retried.
020   * @throws IllegalArgumentException    if the dataset doesn't exist, we don't support its protocol, it isn't eligible
021   *                                     for crawling
022   * @throws AlreadyCrawlingException    if the dataset is already being crawled
023   */
024  void initiateCrawl(UUID datasetKey, int priority);
025
026  /**
027   * Initiates a crawl of an existing dataset without any explicit priority. Implementations are free to chose a
028   * default priority.
029   *
030   * @param datasetKey of the dataset to crawl
031   *
032   * @throws ServiceUnavailableException if there are any problems communicating with the Registry or ZooKeeper.
033   *                                     ZooKeeper will already have been retried.
034   * @throws IllegalArgumentException    if the dataset doesn't exist, we don't support its protocol, it isn't eligible
035   *                                     for crawling
036   * @throws AlreadyCrawlingException    if the dataset is already being crawled
037   */
038  void initiateCrawl(UUID datasetKey);
039
040}
