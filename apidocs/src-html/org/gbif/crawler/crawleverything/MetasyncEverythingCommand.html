001package org.gbif.crawler.crawleverything;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.api.model.common.paging.PagingRequest;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.api.model.registry.Installation;
007import org.gbif.api.service.registry.InstallationService;
008import org.gbif.cli.BaseCommand;
009import org.gbif.cli.Command;
010import org.gbif.common.messaging.DefaultMessagePublisher;
011import org.gbif.common.messaging.api.Message;
012import org.gbif.common.messaging.api.MessagePublisher;
013import org.gbif.common.messaging.api.messages.StartMetasyncMessage;
014import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
015import org.gbif.ws.client.guice.AnonymousAuthModule;
016
017import java.io.IOException;
018import java.util.Properties;
019import java.util.concurrent.ExecutorService;
020import java.util.concurrent.Executors;
021import java.util.concurrent.TimeUnit;
022import java.util.concurrent.atomic.AtomicInteger;
023
024import com.google.common.base.Stopwatch;
025import com.google.common.base.Throwables;
026import com.google.inject.Guice;
027import com.google.inject.Injector;
028import org.kohsuke.MetaInfServices;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * This command iterates over all installations from the registry and sends a StartMetasyncMessage for each of them.
034 */
035@MetaInfServices(Command.class)
036public class MetasyncEverythingCommand extends BaseCommand {
037
038  private static final Logger LOG = LoggerFactory.getLogger(CrawlEverythingCommand.class);
039  private static final int LIMIT = 1000;
040  private final EverythingConfiguration config = new EverythingConfiguration();
041
042  public MetasyncEverythingCommand() {
043    super("metasynceverything");
044  }
045
046  @Override
047  protected Object getConfigurationObject() {
048    return config;
049  }
050
051  @Override
052  protected void doRun() {
053    try {
054      MessagePublisher publisher = new DefaultMessagePublisher(config.messaging.getConnectionParameters());
055
056      // Create Registry WS Client
057      Properties properties = new Properties();
058      properties.setProperty("registry.ws.url", config.registryWsUrl);
059
060      Injector injector = Guice.createInjector(new RegistryWsClientModule(properties), new AnonymousAuthModule());
061
062      int offset = 0;
063      boolean endOfRecords = true;
064      ExecutorService executor = Executors.newFixedThreadPool(20);
065      AtomicInteger totalCount = new AtomicInteger();
066      AtomicInteger scheduledCount = new AtomicInteger();
067      InstallationService installationService = injector.getInstance(InstallationService.class);
068      do {
069        Pageable request = new PagingRequest(offset, LIMIT);
070        Stopwatch stopwatch = Stopwatch.createStarted();
071        LOG.info("Requesting batch of installations starting at offset [{}]", request.getOffset());
072        PagingResponse<Installation> installations;
073        try {
074          installations = installationService.list(request);
075        } catch (Exception e) {
076          offset += LIMIT;
077          LOG.error("Got error requesting installations, skipping to offset [{}]", offset, e);
078          continue;
079        }
080        stopwatch.stop();
081        LOG.info("Received [{}] installations in [{}]s", installations.getResults().size(),
082          stopwatch.elapsed(TimeUnit.SECONDS));
083        executor
084          .submit(new InstallationSchedulingRunnable(installations, publisher, totalCount, scheduledCount, offset));
085
086        endOfRecords = installations.isEndOfRecords();
087        offset += installations.getResults().size();
088      } while (!endOfRecords);
089
090      executor.shutdown();
091      while (!executor.isTerminated()) {
092        try {
093          LOG.info("Waiting for completion of metasync...");
094          executor.awaitTermination(1, TimeUnit.MINUTES);
095        } catch (InterruptedException e) {
096          LOG.error("Waiting for completion interrupted", e);
097        }
098      }
099      publisher.close();
100      LOG.info("Done processing [{}] installations, [{}] were scheduled", totalCount.get(), scheduledCount.get());
101    } catch (IOException e) {
102      throw Throwables.propagate(e); // we're hosed
103    }
104  }
105
106  private static class InstallationSchedulingRunnable implements Runnable {
107
108    private final int offset;
109    private final PagingResponse<Installation> installations;
110    private final MessagePublisher publisher;
111    private final AtomicInteger count;
112    private final AtomicInteger scheduledCount;
113
114    private InstallationSchedulingRunnable(PagingResponse<Installation> installations, MessagePublisher publisher,
115      AtomicInteger count, AtomicInteger scheduledCount, int offset) {
116      this.installations = installations;
117      this.publisher = publisher;
118      this.count = count;
119      this.offset = offset;
120      this.scheduledCount = scheduledCount;
121    }
122
123    @Override
124    public void run() {
125      int registeredCount = 0;
126      for (Installation installation : installations.getResults()) {
127        count.incrementAndGet();
128        Message message = new StartMetasyncMessage(installation.getKey());
129        try {
130          publisher.send(message);
131          registeredCount++;
132          scheduledCount.incrementAndGet();
133          LOG.debug("Scheduled metasync of installation [{}]", installation.getKey());
134        } catch (IOException e) {
135          LOG.error("Caught exception while sending metasync message", e);
136        }
137      }
138      LOG.debug("[{}] installations out of [{}] for offset [{}] were registered and scheduled", registeredCount,
139        installations.getResults().size(), offset);
140    }
141  }
142}
