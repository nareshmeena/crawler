001package org.gbif.crawler;
002
003import org.gbif.crawler.exception.FatalCrawlException;
004import org.gbif.crawler.exception.ProtocolException;
005import org.gbif.crawler.exception.TransportException;
006
007/**
008 * Implementations of this interface take responsibility for interactions with endpoints. They take some form of
009 * request, a handler that processes the response of this request into a result.
010 *
011 * @param <REQUEST>  format of the request to process. This is expected to be mostly a String but could be a complex
012 *                   object if needed
013 * @param <RESPONSE> the type of the response that the CrawlClient produces which the ResponseHandler will have to
014 *                   process
015 */
016public interface CrawlClient<REQUEST, RESPONSE> {
017
018  /**
019   * Executes the request and lets a ResponseHandler handle it and returns its result.
020   *
021   * @param request that should be issued
022   * @param handler to process the response and that produces the result
023   *
024   * @throws TransportException on transport error, such as timeout, unable to connect etc.
025   * @throws ProtocolException  on error during handling of the response (e.g. such as a unexpected format, XML issues)
026   */
027  <RESULT> RESULT execute(REQUEST request, ResponseHandler<RESPONSE, RESULT> handler)
028    throws FatalCrawlException, TransportException, ProtocolException;
029
030}
