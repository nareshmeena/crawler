013package org.gbif.crawler.xml.crawlserver.util;
014
015import org.gbif.wrangler.lock.Lock;
016import org.gbif.wrangler.lock.LockFactory;
017
018import java.util.concurrent.TimeUnit;
019
020import com.google.common.base.Stopwatch;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024import static java.util.concurrent.TimeUnit.MILLISECONDS;
025
026import static com.google.common.base.Preconditions.checkArgument;
027import static com.google.common.base.Preconditions.checkNotNull;
028
029/**
030 * A lock factory that forwards all calls to another Lock Factory but delays unlocking for a certain amount of time.
031 * This can be used to make sure that we don't tax the resources of providers too much. The delay depends on three
032 * factors but by default it is the amount of time that the lock was taken out for. So if we held the lock for one
033 * second then we're going to delay unlocking for another second. A minimum (default = 100ms and maximum (default = 2s)
034 * delay must be specified and we're never going to delay less or more than those bounds.
035 */
036public class DynamicDelayingLockFactory extends ForwardingLockFactory {
037
038  private class DynamicDelayingLock extends ForwardingLock {
039
040    private final Lock delegateLock;
041
042    private final Stopwatch stopwatch = Stopwatch.createUnstarted();
043
044    private DynamicDelayingLock(Lock delegateLock) {
045      this.delegateLock = delegateLock;
046    }
047
048    @Override
049    public boolean tryLock(long time, TimeUnit unit) {
050      stopwatch.reset();
051      stopwatch.start();
052      return delegate().tryLock(time, unit);
053    }
054
055    @Override
056    public void unlock() {
057      stopwatch.stop();
058
059      // Make sure that we delay for a period of time between the given min and maximum delay
060      long localDelay = stopwatch.elapsed(MILLISECONDS) > maxDelay ? maxDelay : stopwatch.elapsed(MILLISECONDS);
061      localDelay = localDelay < minDelay ? minDelay : localDelay;
062
063      if (localDelay > 0) {
064        try {
065          Thread.sleep(localDelay);
066        } catch (InterruptedException ignored) {
067          LOG.warn("Interrupted while sleeping");
068        }
069      }
070
071      delegate().unlock();
072    }
073
074    @Override
075    protected Lock delegate() {
076      return delegateLock;
077    }
078
079  }
080
081  /**
082   * Default maximum value of milliseconds that unlock operations are delayed if no other value is given.
083   */
084  public static final long DEFAULT_MAX_DELAY = 2000;
085
086  /**
087   * Default minimum value of milliseconds that unlock operations are delayed if no other value is given.
088   */
089  private static final int DEFAULT_MIN_DELAY = 100;
090
091  private static final Logger LOG = LoggerFactory.getLogger(DynamicDelayingLockFactory.class);
092
093  private final LockFactory delegateFactory;
094  private final long maxDelay;
095
096  private final long minDelay;
097
098  /**
099   * Builds a delaying lock factory that delays every unlock operation by a certain amount of time. The exact delay
100   * depends on three factors: The time the lock was taken out (i.e. time between acquiring the lock and releasing it),
101   * but it will always be between the provided minimum and maximum delay.
102   * 
103   * @param delegateFactory to forward calls to
104   */
105  public DynamicDelayingLockFactory(LockFactory delegateFactory) {
106    this(delegateFactory, DEFAULT_MAX_DELAY, DEFAULT_MIN_DELAY);
107  }
108
109  /**
110   * Builds a delaying lock factory that delays every unlock operation by a user-specified amount of milliseconds.
111   * 
112   * @param delegateFactory to forward calls to
113   * @param maxDelay in milliseconds
114   */
115  public DynamicDelayingLockFactory(LockFactory delegateFactory, long minDelay, long maxDelay) {
116    this.delegateFactory = checkNotNull(delegateFactory, "delegateFactory can't be null");
117    checkArgument(minDelay >= 0, "minDelay must be greater than or equal to 0");
118    checkArgument(maxDelay >= 0, "maxDelay must be greater than or equal to 0");
119    checkArgument(maxDelay > minDelay, "maxDelay must be greater than minDelay");
120    this.minDelay = minDelay;
121    this.maxDelay = maxDelay;
122  }
123
124  @Override
125  public Lock makeLock(String name) {
126    return new DynamicDelayingLock(delegate().makeLock(name));
127  }
128
129  @Override
130  protected LockFactory delegate() {
131    return delegateFactory;
132  }
133
134}
