001package org.gbif.crawler.xml.crawlserver.util;
002
003import org.gbif.wrangler.lock.Lock;
004import org.gbif.wrangler.lock.LockFactory;
005
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009/**
010 * A lock factory that forwards all colls to another Lock Factory but delays unlocking for a configurable amount of
011 * time. This can be used to make sure that we don't tax the resources of providers too much.
012 */
013public class DelayingLockFactory extends ForwardingLockFactory {
014
015  /**
016   * Default value of milliseconds that unlock operations are delayed if no other value is given on construction.
017   */
018  public static final long DEFAULT_DELAY = 2000;
019
020  private static final Logger LOG = LoggerFactory.getLogger(DelayingLockFactory.class);
021
022  private final LockFactory delegateFactory;
023
024  private final long delay;
025
026  /**
027   * Builds a delaying lock factory that delays every unlock operation by a certain amount of time as specified in the
028   * default setting {@link DelayingLockFactory#DEFAULT_DELAY}.
029   *
030   * @param delegateFactory to forward calls to
031   */
032  public DelayingLockFactory(LockFactory delegateFactory) {
033    this(delegateFactory, DEFAULT_DELAY);
034  }
035
036  /**
037   * Builds a delaying lock factory that delays every unlock operation by a user-specified amount of milliseconds.
038   *
039   * @param delegateFactory to forward calls to
040   * @param delay           in milliseconds
041   */
042  public DelayingLockFactory(LockFactory delegateFactory, long delay) {
043    this.delegateFactory = delegateFactory;
044    this.delay = delay;
045  }
046
047  @Override
048  public Lock makeLock(String name) {
049    return new DelayingLock(delegate().makeLock(name), delay);
050  }
051
052  @Override
053  protected LockFactory delegate() {
054    return delegateFactory;
055  }
056
057  private class DelayingLock extends ForwardingLock {
058
059    private final Lock delegateLock;
060
061    private DelayingLock(Lock delegateLock, long delay) {
062      this.delegateLock = delegateLock;
063    }
064
065    @Override
066    public void unlock() {
067      try {
068        Thread.sleep(delay);
069      } catch (InterruptedException e) {
070        LOG.warn("Interrupted while sleeping");
071      }
072      delegate().unlock();
073    }
074
075    @Override
076    protected Lock delegate() {
077      return delegateLock;
078    }
079
080  }
081
082}
