001package org.gbif.crawler.xml.crawlserver.util;
002
003import org.gbif.crawler.client.HttpCrawlClient;
004
005import org.apache.http.client.HttpClient;
006import org.apache.http.client.params.ClientPNames;
007import org.apache.http.conn.scheme.PlainSocketFactory;
008import org.apache.http.conn.scheme.Scheme;
009import org.apache.http.conn.scheme.SchemeRegistry;
010import org.apache.http.impl.client.DecompressingHttpClient;
011import org.apache.http.impl.client.DefaultHttpClient;
012import org.apache.http.impl.conn.PoolingClientConnectionManager;
013import org.apache.http.params.BasicHttpParams;
014import org.apache.http.params.HttpConnectionParams;
015import org.apache.http.params.HttpParams;
016
017public class HttpCrawlClientProvider {
018
019  private static final int DEFAULT_HTTP_PORT = 80;
020
021  private static final int CONNECTION_TIMEOUT_MSEC = 600000; // 10 mins
022  private static final int MAX_TOTAL_CONNECTIONS = 500;
023  private static final int MAX_TOTAL_PER_ROUTE = 20;
024
025  public static HttpCrawlClient newHttpCrawlClient() {
026    SchemeRegistry schemeRegistry = new SchemeRegistry();
027    schemeRegistry.register(new Scheme("http", DEFAULT_HTTP_PORT, PlainSocketFactory.getSocketFactory()));
028
029    PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry);
030    connectionManager.setMaxTotal(MAX_TOTAL_CONNECTIONS);
031    connectionManager.setDefaultMaxPerRoute(MAX_TOTAL_PER_ROUTE);
032
033    HttpParams params = new BasicHttpParams();
034    HttpConnectionParams.setConnectionTimeout(params, CONNECTION_TIMEOUT_MSEC);
035    HttpConnectionParams.setSoTimeout(params, CONNECTION_TIMEOUT_MSEC);
036    params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, CONNECTION_TIMEOUT_MSEC);
037    HttpClient httpClient = new DecompressingHttpClient(new DefaultHttpClient(connectionManager, params));
038
039    return new HttpCrawlClient(connectionManager, httpClient);
040  }
041
042  private HttpCrawlClientProvider() {
043    throw new UnsupportedOperationException("Can't initialize class");
044  }
045
046}
