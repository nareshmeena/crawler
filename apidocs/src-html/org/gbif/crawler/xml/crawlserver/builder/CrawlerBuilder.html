040@SuppressWarnings("ReturnOfThis")
041public class CrawlerBuilder {
042
043  private RequestHandler<ScientificNameRangeCrawlContext, String> requestHandler;
044  private ResponseHandler<HttpResponse, List<Byte>> responseHandler;
045  private CrawlConfiguration crawlConfiguration;
046  private HttpCrawlClient client;
047  private ScientificNameRangeStrategy strategy;
048  private ScientificNameRangeCrawlContext context;
049  private RetryPolicy retryPolicy;
050  private LockFactory lockFactory;
051  private Optional<Long> minDelay = Optional.absent();
052  private Optional<Long> maxDelay = Optional.absent();
053  private static final String LOCKING_PATH = "/lockedUrls/";
054
055
056  public CrawlerBuilder(CrawlJob crawlJob) {
057    setupProtocolSpecifics(crawlJob);
058    setupCrawlClient();
059  }
060
061  public static CrawlerBuilder buildFor(CrawlJob crawlJob) {
062    return new CrawlerBuilder(crawlJob);
063  }
064
065  public Crawler<ScientificNameRangeCrawlContext, String, HttpResponse, List<Byte>> build() {
066    if (minDelay.isPresent()) {
067      lockFactory = new DynamicDelayingLockFactory(lockFactory, minDelay.get(), maxDelay.get());
068    }
069    Lock lock = lockFactory.makeLock(crawlConfiguration.getUrl().getHost());
070    return new Crawler<ScientificNameRangeCrawlContext, String, HttpResponse, List<Byte>>(strategy,
071      requestHandler,
072      responseHandler,
073      client,
074      retryPolicy,
075      lock);
076  }
077
078  public CrawlConfiguration getCrawlConfiguration() {
079    return crawlConfiguration;
080  }
081
082  public CrawlerBuilder withDefaultRetryStrategy() {
083    retryPolicy = new LimitedRetryPolicy(10, 2, 10, 2);
084    return this;
085  }
086
087  public CrawlerBuilder withDelayedLock(long minDelay, long maxDelay) {
088    this.minDelay = Optional.of(minDelay);
089    this.maxDelay = Optional.of(maxDelay);
090    return this;
091  }
092
093  public CrawlerBuilder withoutLocking() {
094    lockFactory = new NoLockFactory();
095    return this;
096  }
097
098  public CrawlerBuilder withScientificNameRangeCrawlContext() {
099    context = new ScientificNameRangeCrawlContext();
100    strategy = new ScientificNameRangeStrategy(context);
101    return this;
102  }
103
104  public CrawlerBuilder withScientificNameRangeCrawlContext(@Nullable String lowerBound, @Nullable String upperBound) {
105    context = new ScientificNameRangeCrawlContext(0, lowerBound, upperBound);
106    return this;
107  }
108
109
110  public CrawlerBuilder withZooKeeperLock(CuratorFramework curator) {
111    checkNotNull(curator, "curator can't be null");
112    lockFactory = new ZooKeeperLockFactory(curator, LOCKING_PATH);
113    return this;
114  }
115
116  public CrawlerBuilder withZooKeeperLock(CuratorFramework curator, int count) {
117    checkNotNull(curator, "curator can't be null");
118    lockFactory = new ZooKeeperLockFactory(curator, count, LOCKING_PATH);
119    return this;
120  }
121
122  private void setupBiocaseCrawl(CrawlJob crawlJob) {
123    checkNotNull(crawlJob.getProperty("conceptualSchema"));
124    checkNotNull(crawlJob.getProperty("datasetTitle"));
125
126    BiocaseCrawlConfiguration job = new BiocaseCrawlConfiguration(crawlJob.getDatasetKey(),
127      crawlJob.getAttempt(),
128      crawlJob.getTargetUrl(),
129      crawlJob.getProperty("conceptualSchema"),
130      crawlJob.getProperty("datasetTitle"));
131
132    crawlConfiguration = job;
133    requestHandler = new BiocaseScientificNameRangeRequestHandler(job);
134    responseHandler = new BiocaseResponseHandler();
135  }
136
137  private void setupCrawlClient() {
138    client = HttpCrawlClientProvider.newHttpCrawlClient();
139  }
140
141  private void setupDigirCrawl(CrawlJob crawlJob) {
142    checkNotNull(crawlJob.getProperty("code"));
143    checkNotNull(crawlJob.getProperty("manis"));
144
145    DigirCrawlConfiguration job = new DigirCrawlConfiguration(crawlJob.getDatasetKey(),
146      crawlJob.getAttempt(),
147      crawlJob.getTargetUrl(),
148      crawlJob.getProperty("code"),
149      Boolean.valueOf(crawlJob.getProperty("manis")));
150
151    crawlConfiguration = job;
152    requestHandler = new DigirScientificNameRangeRequestHandler(job);
153    responseHandler = new DigirResponseHandler();
154  }
155
156  private void setupProtocolSpecifics(CrawlJob crawlJob) {
157    switch (crawlJob.getEndpointType()) {
158      case TAPIR:
159        setupTapirCrawl(crawlJob);
160        break;
161      case BIOCASE:
162        setupBiocaseCrawl(crawlJob);
163        break;
164      case DIGIR:
165      case DIGIR_MANIS:
166        setupDigirCrawl(crawlJob);
167        break;
168      default:
169        // LOG.warn("Received crawl for unsupported endpoint, [{}]", crawlJob);
170    }
171  }
172
173  /**
174   * This extracts the properties from the crawl job specific to the TAPIR crawl.
175   * <p/>
176   * When used with the GBIF messaging service, this is populated in the private getCrawlJob() method in
177   * {@link CrawlerCoordinatorServiceImpl}. Note that in that implementation it selects a single
178   * <em>conceptualSchema</em> to use as the <em>contentNamespace</em> during crawling. Should logic be required to
179   * change the selection criteria, it must happen when the crawlJob is created, such as in
180   * {@link CrawlerCoordinatorServiceImpl}.
181   */
182  private void setupTapirCrawl(CrawlJob crawlJob) {
183    checkNotNull(crawlJob.getProperty("conceptualSchema"));
184
185    TapirCrawlConfiguration job = new TapirCrawlConfiguration(crawlJob.getDatasetKey(),
186      crawlJob.getAttempt(),
187      crawlJob.getTargetUrl(),
188      crawlJob.getProperty("conceptualSchema"));
189
190    crawlConfiguration = job;
191    requestHandler = new TapirScientificNameRangeRequestHandler(job);
192    responseHandler = new TapirResponseHandler();
193  }
