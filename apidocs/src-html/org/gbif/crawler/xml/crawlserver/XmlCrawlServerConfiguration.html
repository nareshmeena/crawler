016package org.gbif.crawler.xml.crawlserver;
017
018import org.gbif.crawler.common.crawlserver.CrawlServerConfiguration;
019
020import java.io.File;
021import javax.validation.constraints.Min;
022
023import com.beust.jcommander.Parameter;
024import com.google.common.base.Objects;
025
026@SuppressWarnings("PublicField")
027public class XmlCrawlServerConfiguration extends CrawlServerConfiguration {
028
029  @Parameter(names = "--min-lock-delay", description = "An optional minimum delay before releasing a lock on an URL.")
030  @Min(1)
031  public Long minLockDelay;
032
033  @Parameter(names = "--max-lock-delay", description = "An optional maximum delay before releasing a lock on an URL.")
034  @Min(1)
035  public Long maxLockDelay;
036
037  @Parameter(names = "--response-archive", description = "An optional directory where all the responses will be saved")
038  public File responseArchive;
039
040  @Override
041  public String toString() {
042    return Objects.toStringHelper(this)
043      .add("super", super.toString())
044      .add("minLockDelay", minLockDelay)
045      .add("maxLockDelay", maxLockDelay)
046      .add("responseArchive", responseArchive)
047      .toString();
048  }
049
050}
