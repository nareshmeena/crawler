016package org.gbif.crawler.xml.crawlserver.listener;
017
018import org.gbif.crawler.AbstractCrawlListener;
019import org.gbif.crawler.CrawlConfiguration;
020import org.gbif.crawler.strategy.ScientificNameRangeCrawlContext;
021
022import java.io.File;
023import java.io.IOException;
024import java.util.List;
025
026import com.google.common.base.Optional;
027import com.google.common.io.Files;
028import com.google.common.primitives.Bytes;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * This Crawl Listener will persist every single response we receive to disk.
034 *
035 * It will do so in sub folders of the given base path in a format like this:
036 * /<uuid>/<attempt>/<context>/<uuid>_<attempt>_<context>_<retry>.response
037 */
038public class ResultPersistingListener
039  extends AbstractCrawlListener<ScientificNameRangeCrawlContext, String, List<Byte>> {
040
041  private static final Logger LOG = LoggerFactory.getLogger(ResultPersistingListener.class);
042
043  private final File basePath;
044  private final CrawlConfiguration configuration;
045
046  public ResultPersistingListener(File basePath, CrawlConfiguration configuration) {
047    this.basePath = basePath;
048    this.configuration = configuration;
049  }
050
051  @Override
052  public void response(
053    List<Byte> response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords
054  ) {
055    String contextString =
056      getCurrentContext().getLowerBound().or("null") + "-" + getCurrentContext().getUpperBound().or("null");
057    if (response == null || response.isEmpty()) {
058      LOG.info("Received empty response for [{}] [{}] in retry [{}]",
059               configuration.getDatasetKey(),
060               contextString,
061               retry);
062    }
063    // This is guaranteed to be efficient as documented on the AbstractResponseHandler
064    byte[] bytes = Bytes.toArray(response);
065    try {
066      // Creates a format like this: /<uuid>/<attempt>/<context>/<uuid>_<attempt>_<context>_<retry>.response
067      StringBuilder path = new StringBuilder();
068      path.append(configuration.getDatasetKey())
069        .append(File.separator)
070        .append(configuration.getAttempt())
071        .append(File.separator)
072        .append(contextString)
073        .append(File.separator)
074        .append(configuration.getDatasetKey())
075        .append("_")
076        .append(configuration.getAttempt())
077        .append("_")
078        .append(contextString)
079        .append("_")
080        .append(getCurrentContext().getOffset())
081        .append("_")
082        .append(retry)
083        .append(".response");
084
085      File file = new File(basePath, path.toString());
086      Files.createParentDirs(file);
087      Files.write(bytes, file);
088    } catch (IOException e) {
089      LOG.warn("Could not write response due to exception. [{}]", configuration.getDatasetKey(), e);
090    }
091  }
092}
