001package org.gbif.crawler.xml.crawlserver.listener;
002
003import org.gbif.api.model.crawler.FinishReason;
004import org.gbif.crawler.CrawlConfiguration;
005import org.gbif.crawler.CrawlContext;
006import org.gbif.crawler.CrawlListener;
007
008import java.util.Date;
009import java.util.List;
010import java.util.concurrent.TimeUnit;
011
012import com.google.common.base.Optional;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015import org.slf4j.MDC;
016
017/**
018 * This class uses SLF4J to log events that occur during crawling. It logs everything at INFO level except errors which
019 * are at WARN. It also make two things available in the MDC: datasetKey and attempt.
020 */
021public class LoggingCrawlListener<CTX extends CrawlContext> implements CrawlListener<CTX, String, List<Byte>> {
022
023  private static final Logger LOG = LoggerFactory.getLogger(LoggingCrawlListener.class);
024
025  private CrawlContext lastContext;
026
027  private String lastRequest;
028
029  private int totalRecordCount;
030
031  private Date startDate;
032
033  public LoggingCrawlListener(CrawlConfiguration configuration) {
034    MDC.put("datasetKey", configuration.getDatasetKey().toString());
035    MDC.put("attempt", String.valueOf(configuration.getAttempt()));
036  }
037
038  @Override
039  public void startCrawl() {
040    this.startDate = new Date();
041    LOG.info("started crawl");
042  }
043
044  @Override
045  public void error(Throwable e) {
046    LOG.warn("error during crawling: [{}], last request [{}]", lastContext, lastRequest, e);
047  }
048
049  @Override
050  public void error(String msg) {
051    LOG.warn("error during crawling: [{}], last request [{}], message [{}]", lastContext, lastRequest, msg);
052  }
053
054  @Override
055  public void response(
056    List<Byte> response, int retry, long duration, Optional<Integer> recordCount, Optional<Boolean> endOfRecords
057  ) {
058    totalRecordCount += recordCount.or(0);
059    LOG.info("got response for [{}], records [{}], endOfRecords [{}], retry [{}], took [{}s]",
060             lastContext,
061             recordCount,
062             endOfRecords,
063             retry,
064             TimeUnit.MILLISECONDS.toSeconds(duration));
065  }
066
067  @Override
068  public void finishCrawlNormally() {
069    finishCrawl(FinishReason.NORMAL);
070  }
071
072  @Override
073  public void finishCrawlOnUserRequest() {
074    finishCrawl(FinishReason.USER_ABORT);
075  }
076
077  @Override
078  public void finishCrawlAbnormally() {
079    finishCrawl(FinishReason.ABORT);
080  }
081
082  private void finishCrawl(FinishReason reason) {
083    Date finishDate = new Date();
084    long minutes = (finishDate.getTime() - startDate.getTime()) / (60 * 1000);
085        LOG.info(
086      "finished crawling with a total of [{}] records, reason [{}], started at [{}], finished at [{}], took [{}] minutes",
087      totalRecordCount,
088      reason,
089      startDate,
090      finishDate,
091      minutes);
092
093    MDC.remove("datasetKey");
094    MDC.remove("attempt");
095  }
096
097  @Override
098  public void request(String req, int retry) {
099    LOG.info("requested page for [{}], retry [{}], request [{}]", lastContext, retry, req);
100    lastRequest = req;
101  }
102
103  @Override
104  public void progress(CrawlContext context) {
105    lastRequest = null;
106    lastContext = context;
107    LOG.info("now beginning to crawl [{}]", context);
108  }
109
110}
