001package org.gbif.crawler.exception;
002
003/**
004 * Indicates an issue with the transport layer during communication.
005 */
006public class TransportException extends Exception {
007
008  private static final long serialVersionUID = 5973132911743143191L;
009
010  public TransportException(String message) {
011    super(message);
012  }
013
014  public TransportException(Throwable cause) {
015    super(cause);
016  }
017
018  public TransportException(String message, Throwable cause) {
019    super(message, cause);
020  }
021
022}
