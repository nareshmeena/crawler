001package org.gbif.crawler.retry;
002
003import org.gbif.crawler.RetryPolicy;
004
005/**
006 * This implements a retry strategy that allows a certain fixed number of total and consecutive exceptions.
007 * <p/>
008 * The consecutive counts are reset with a successful request and the total number is never reset. The counts are
009 * combined in the sense that if one of the consecutive limits is hit neither is allowed to be retried and if any of
010 * the total limits have been hit the crawl should be aborted.
011 */
012public class LimitedRetryPolicy implements RetryPolicy {
013
014  // The maximum number of allowed total and consecutive Protocol and Transport exceptions
015  private final int maxTotalProtocolExceptions;
016  private final int maxConsecutiveProtocolExceptions;
017
018  private final int maxTotalTransportExceptions;
019  private final int maxConsecutiveTransportExceptions;
020
021  // The current counts of total and consecutive exceptions
022  private int totalProtocolExceptionCount;
023  private int totalTransportExceptionCount;
024
025  private int consecutiveProtocolExceptionCount;
026  private int consecutiveTransportExceptionCount;
027
028  public LimitedRetryPolicy(
029    int maxTotalProtocolExceptions,
030    int maxConsecutiveProtocolExceptions,
031    int maxTotalTransportExceptions,
032    int maxConsecutiveTransportExceptions
033  ) {
034    this.maxTotalProtocolExceptions = maxTotalProtocolExceptions;
035    this.maxConsecutiveProtocolExceptions = maxConsecutiveProtocolExceptions;
036    this.maxTotalTransportExceptions = maxTotalTransportExceptions;
037    this.maxConsecutiveTransportExceptions = maxConsecutiveTransportExceptions;
038  }
039
040  @Override
041  public boolean allowAfterProtocolException() {
042    totalProtocolExceptionCount++;
043    consecutiveProtocolExceptionCount++;
044
045    return allowRetry();
046  }
047
048  @Override
049  public boolean allowAfterTransportException() {
050    totalTransportExceptionCount++;
051    consecutiveTransportExceptionCount++;
052
053    return allowRetry();
054  }
055
056  @Override
057  public void successfulRequest() {
058    consecutiveProtocolExceptionCount = 0;
059    consecutiveTransportExceptionCount = 0;
060  }
061
062  @Override
063  public void giveUpRequest() {
064    consecutiveProtocolExceptionCount = 0;
065    consecutiveTransportExceptionCount = 0;
066  }
067
068  @Override
069  public boolean abortCrawl() {
070    return totalProtocolExceptionCount > maxTotalProtocolExceptions
071           || totalTransportExceptionCount > maxTotalTransportExceptions;
072
073  }
074
075  private boolean allowRetry() {
076    return !abortCrawl()
077           && consecutiveProtocolExceptionCount <= maxConsecutiveProtocolExceptions
078           && consecutiveTransportExceptionCount <= maxConsecutiveTransportExceptions;
079
080  }
081
082}
